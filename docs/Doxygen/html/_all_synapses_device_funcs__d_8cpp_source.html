51CUDA_CALLABLE void changeDSSynapsePSRDevice(AllDSSynapsesDeviceProperties *allEdgesDevice,
52 BGSIZE iEdg, const uint64_t simulationStep,
53 BGFLOAT deltaT)
54{
55 //assert( iEdg < allEdgesDevice->maxEdgesPerVertex * allEdgesDevice->countVertices_ );
56
57 uint64_t &lastSpike = allEdgesDevice->lastSpike_[iEdg];
58 BGFLOAT &r = allEdgesDevice->r_[iEdg];
59 BGFLOAT &u = allEdgesDevice->u_[iEdg];
60 BGFLOAT D = allEdgesDevice->D_[iEdg];
61 BGFLOAT F = allEdgesDevice->F_[iEdg];
62 BGFLOAT U = allEdgesDevice->U_[iEdg];
63 BGFLOAT W = allEdgesDevice->W_[iEdg];
64 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
65 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
66
67 // adjust synapse parameters
68 if (lastSpike != ULONG_MAX) {
69 BGFLOAT isi = (simulationStep - lastSpike) * deltaT;
70 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
71 u = U + u * (1 - U) * exp(-isi / F);
72 }
73 psr += ((W / decay) * u * r); // calculate psr
74 lastSpike = simulationStep; // record the time of the spike
75}

432CUDA_CALLABLE void addSpikingSynapse(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
433 edgeType type, int srcVertex, int destVertex, int sourceIndex,
434 int destIndex, BGFLOAT deltaT, BGFLOAT *W_d, int numVertices)
435{
436 if (allEdgesDevice->edgeCounts_[destVertex] >= allEdgesDevice->maxEdgesPerVertex_) {
437 return; // TODO: ERROR!
438 }
439
440 // add it to the list
441 BGSIZE synapseIndex;
442 BGSIZE maxEdges = allEdgesDevice->maxEdgesPerVertex_;
443 BGSIZE synapseBegin = maxEdges * destVertex;
444 for (synapseIndex = 0; synapseIndex < maxEdges; synapseIndex++) {
445 if (!allEdgesDevice->inUse_[synapseBegin + synapseIndex]) {
446 break;
447 }
448 }
449
450 allEdgesDevice->edgeCounts_[destVertex]++;
451
452 // create a synapse
453 switch (classSynapses_d) {
454 case enumClassSynapses::classAllSpikingSynapses:
455 createSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, sourceIndex, destIndex,
456 deltaT, type);
457 break;
458 case enumClassSynapses::classAllDSSynapses:
459 createDSSynapse(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
460 synapseIndex, sourceIndex, destIndex, deltaT, type);
461 break;
462 case enumClassSynapses::classAllSTDPSynapses:
463 createSTDPSynapse(static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice),
464 destVertex, synapseIndex, sourceIndex, destIndex, deltaT, type);
465 break;
466 case enumClassSynapses::classAllDynamicSTDPSynapses:
467 createDynamicSTDPSynapse(
468 static_cast<AllDynamicSTDPSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
469 synapseIndex, sourceIndex, destIndex, deltaT, type);
470 break;
471 default:
472 assert(false);
473 }
474 allEdgesDevice->W_[synapseBegin + synapseIndex] = W_d[srcVertex * numVertices + destVertex]
475 * edgSign(type)
476 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
477}

500CUDA_CALLABLE edgeType edgType(vertexType *neuronTypeMap_d, int srcVertex, int destVertex)
501{
502 if (neuronTypeMap_d[srcVertex] == vertexType::INH
503 && neuronTypeMap_d[destVertex] == vertexType::INH)
504 return edgeType::II;
505 else if (neuronTypeMap_d[srcVertex] == vertexType::INH
506 && neuronTypeMap_d[destVertex] == vertexType::EXC)
507 return edgeType::IE;
508 else if (neuronTypeMap_d[srcVertex] == vertexType::EXC
509 && neuronTypeMap_d[destVertex] == vertexType::INH)
510 return edgeType::EI;
511 else if (neuronTypeMap_d[srcVertex] == vertexType::EXC
512 && neuronTypeMap_d[destVertex] == vertexType::EXC)
513 return edgeType::EE;
514
515 return edgeType::ETYPE_UNDEF;
516}

533__global__ void updateSynapsesWeightsDevice(int numVertices, BGFLOAT deltaT, BGFLOAT *W_d,
534 int maxEdges,
535 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
536 AllSpikingSynapsesDeviceProperties *allEdgesDevice,
537 vertexType *neuronTypeMap_d)
538{
539 int idx = blockIdx.x * blockDim.x + threadIdx.x;
540 if (idx >= numVertices)
541 return;
542
543 int adjusted = 0;
544 //int could_have_been_removed = 0; // TODO: use this value
545 int removed = 0;
546 int added = 0;
547
548 // Scale and add sign to the areas
549 // visit each neuron 'a'
550 int destVertex = idx;
551
552 // and each destination neuron 'b'
553 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
554 if (destVertex == srcVertex) {
555 // we don't create a synapse between the same neuron
556 continue;
557 }
558
559 // visit each synapse at (xa,ya)
560 bool connected = false;
561 edgeType type = edgType(neuronTypeMap_d, srcVertex, destVertex);
562
563 // for each existing synapse
564 BGSIZE existing_synapses = allEdgesDevice->edgeCounts_[destVertex];
565 int existingSynapsesChecked = 0;
566 for (BGSIZE synapseIndex = 0; (existingSynapsesChecked < existing_synapses) && !connected;
567 synapseIndex++) {
568 BGSIZE iEdg = maxEdges * destVertex + synapseIndex;
569 if (allEdgesDevice->inUse_[iEdg]) {
570 // if there is a synapse between a and b
571 if (allEdgesDevice->sourceVertexIndex_[iEdg] == srcVertex) {
572 connected = true;
573 adjusted++;
574
575 // adjust the strength of the synapse or remove
576 // it from the synapse map if it has gone below
577 // zero.
578 if (W_d[srcVertex * numVertices + destVertex] <= 0) {
579 removed++;
580 eraseSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, maxEdges);
581 } else {
582 // adjust
583 // g_synapseStrengthAdjustmentConstant is 1.0e-8;
584 allEdgesDevice->W_[iEdg] = W_d[srcVertex * numVertices + destVertex]
585 * edgSign(type)
586 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
587 }
588 }
589 existingSynapsesChecked++;
590 }
591 }
592
593 // if not connected and weight(a,b) > 0, add a new synapse from a to b
594 if (!connected && (W_d[srcVertex * numVertices + destVertex] > 0)) {
595 added++;
596
597 addSpikingSynapse(allEdgesDevice, type, srcVertex, destVertex, srcVertex, destVertex,
598 deltaT, W_d, numVertices);
599 }
600 }
601}
