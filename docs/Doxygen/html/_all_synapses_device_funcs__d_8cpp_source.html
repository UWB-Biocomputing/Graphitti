51CUDA_CALLABLE void changeDSSynapsePSRDevice(AllDSSynapsesDeviceProperties *allEdgesDevice,
52 BGSIZE iEdg, const uint64_t simulationStep,
53 BGFLOAT deltaT)
54{
55 //assert( iEdg < allEdgesDevice->maxEdgesPerVertex * allEdgesDevice->countVertices_ );
56
57 uint64_t &lastSpike = allEdgesDevice->lastSpike_[iEdg];
58 BGFLOAT &r = allEdgesDevice->r_[iEdg];
59 BGFLOAT &u = allEdgesDevice->u_[iEdg];
60 BGFLOAT D = allEdgesDevice->D_[iEdg];
61 BGFLOAT F = allEdgesDevice->F_[iEdg];
62 BGFLOAT U = allEdgesDevice->U_[iEdg];
63 BGFLOAT W = allEdgesDevice->W_[iEdg];
64 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
65 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
66
67 // adjust synapse parameters
68 if (lastSpike != ULONG_MAX) {
69 BGFLOAT isi = (simulationStep - lastSpike) * deltaT;
70 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
71 u = U + u * (1 - U) * exp(-isi / F);
72 }
73 psr += ((W / decay) * u * r); // calculate psr
74 lastSpike = simulationStep; // record the time of the spike
75}

432CUDA_CALLABLE void addSpikingSynapse(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
433 edgeType type, int srcVertex, int destVertex, int sourceIndex,
434 int destIndex, BGFLOAT deltaT, BGFLOAT *W_d, int numVertices)
435{
436 if (allEdgesDevice->edgeCounts_[destVertex] >= allEdgesDevice->maxEdgesPerVertex_) {
437 return; // TODO: ERROR!
438 }
439
440 // add it to the list
441 BGSIZE synapseIndex;
442 BGSIZE maxEdges = allEdgesDevice->maxEdgesPerVertex_;
443 BGSIZE synapseBegin = maxEdges * destVertex;
444 for (synapseIndex = 0; synapseIndex < maxEdges; synapseIndex++) {
445 if (!allEdgesDevice->inUse_[synapseBegin + synapseIndex]) {
446 break;
447 }
448 }
449
450 allEdgesDevice->edgeCounts_[destVertex]++;
451
452 // create a synapse
453 switch (classSynapses_d) {
454 case classAllSpikingSynapses:
455 createSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, sourceIndex, destIndex,
456 deltaT, type);
457 break;
458 case classAllDSSynapses:
459 createDSSynapse(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
460 synapseIndex, sourceIndex, destIndex, deltaT, type);
461 break;
462 case classAllSTDPSynapses:
463 createSTDPSynapse(static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice),
464 destVertex, synapseIndex, sourceIndex, destIndex, deltaT, type);
465 break;
466 case classAllDynamicSTDPSynapses:
467 createDynamicSTDPSynapse(
468 static_cast<AllDynamicSTDPSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
469 synapseIndex, sourceIndex, destIndex, deltaT, type);
470 break;
471 default:
472 assert(false);
473 }
474 allEdgesDevice->W_[synapseBegin + synapseIndex] = W_d[srcVertex * numVertices + destVertex]
475 * edgSign(type)
476 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
477}

500CUDA_CALLABLE edgeType edgType(vertexType *neuronTypeMap_d, int srcVertex, int destVertex)
501{
502 if (neuronTypeMap_d[srcVertex] == INH && neuronTypeMap_d[destVertex] == INH)
503 return II;
504 else if (neuronTypeMap_d[srcVertex] == INH && neuronTypeMap_d[destVertex] == EXC)
505 return IE;
506 else if (neuronTypeMap_d[srcVertex] == EXC && neuronTypeMap_d[destVertex] == INH)
507 return EI;
508 else if (neuronTypeMap_d[srcVertex] == EXC && neuronTypeMap_d[destVertex] == EXC)
509 return EE;
510
511 return ETYPE_UNDEF;
512}

529__global__ void updateSynapsesWeightsDevice(int numVertices, BGFLOAT deltaT, BGFLOAT *W_d,
530 int maxEdges,
531 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
532 AllSpikingSynapsesDeviceProperties *allEdgesDevice,
533 vertexType *neuronTypeMap_d)
534{
535 int idx = blockIdx.x * blockDim.x + threadIdx.x;
536 if (idx >= numVertices)
537 return;
538
539 int adjusted = 0;
540 //int could_have_been_removed = 0; // TODO: use this value
541 int removed = 0;
542 int added = 0;
543
544 // Scale and add sign to the areas
545 // visit each neuron 'a'
546 int destVertex = idx;
547
548 // and each destination neuron 'b'
549 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
550 if (destVertex == srcVertex) {
551 // we don't create a synapse between the same neuron
552 continue;
553 }
554
555 // visit each synapse at (xa,ya)
556 bool connected = false;
557 edgeType type = edgType(neuronTypeMap_d, srcVertex, destVertex);
558
559 // for each existing synapse
560 BGSIZE existing_synapses = allEdgesDevice->edgeCounts_[destVertex];
561 int existingSynapsesChecked = 0;
562 for (BGSIZE synapseIndex = 0; (existingSynapsesChecked < existing_synapses) && !connected;
563 synapseIndex++) {
564 BGSIZE iEdg = maxEdges * destVertex + synapseIndex;
565 if (allEdgesDevice->inUse_[iEdg]) {
566 // if there is a synapse between a and b
567 if (allEdgesDevice->sourceVertexIndex_[iEdg] == srcVertex) {
568 connected = true;
569 adjusted++;
570
571 // adjust the strength of the synapse or remove
572 // it from the synapse map if it has gone below
573 // zero.
574 if (W_d[srcVertex * numVertices + destVertex] <= 0) {
575 removed++;
576 eraseSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, maxEdges);
577 } else {
578 // adjust
579 // g_synapseStrengthAdjustmentConstant is 1.0e-8;
580 allEdgesDevice->W_[iEdg] = W_d[srcVertex * numVertices + destVertex]
581 * edgSign(type)
582 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
583 }
584 }
585 existingSynapsesChecked++;
586 }
587 }
588
589 // if not connected and weight(a,b) > 0, add a new synapse from a to b
590 if (!connected && (W_d[srcVertex * numVertices + destVertex] > 0)) {
591 added++;
592
593 addSpikingSynapse(allEdgesDevice, type, srcVertex, destVertex, srcVertex, destVertex,
594 deltaT, W_d, numVertices);
595 }
596 }
597}
