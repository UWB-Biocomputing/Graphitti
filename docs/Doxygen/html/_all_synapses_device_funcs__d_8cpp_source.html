51CUDA_CALLABLE void changeDSSynapsePSRDevice(AllDSSynapsesDeviceProperties *allEdgesDevice,
52 const BGSIZE iEdg, const uint64_t simulationStep,
53 const BGFLOAT deltaT)
54{
55 //assert( iEdg < allEdgesDevice->maxEdgesPerVertex * allEdgesDevice->countVertices_ );
56
57 uint64_t &lastSpike = allEdgesDevice->lastSpike_[iEdg];
58 BGFLOAT &r = allEdgesDevice->r_[iEdg];
59 BGFLOAT &u = allEdgesDevice->u_[iEdg];
60 BGFLOAT D = allEdgesDevice->D_[iEdg];
61 BGFLOAT F = allEdgesDevice->F_[iEdg];
62 BGFLOAT U = allEdgesDevice->U_[iEdg];
63 BGFLOAT W = allEdgesDevice->W_[iEdg];
64 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
65 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
66
67 // adjust synapse parameters
68 if (lastSpike != ULONG_MAX) {
69 BGFLOAT isi = (simulationStep - lastSpike) * deltaT;
70 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
71 u = U + u * (1 - U) * exp(-isi / F);
72 }
73 psr += ((W / decay) * u * r); // calculate psr
74 lastSpike = simulationStep; // record the time of the spike
75}

442CUDA_CALLABLE void addSpikingSynapse(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
443 edgeType type, const int srcVertex, const int destVertex,
444 int sourceIndex, int destIndex, BGFLOAT *sumPoint,
445 const BGFLOAT deltaT, BGFLOAT *W_d, int numVertices)
446{
447 if (allEdgesDevice->edgeCounts_[destVertex] >= allEdgesDevice->maxEdgesPerVertex_) {
448 return; // TODO: ERROR!
449 }
450
451 // add it to the list
452 BGSIZE synapseIndex;
453 BGSIZE maxEdges = allEdgesDevice->maxEdgesPerVertex_;
454 BGSIZE synapseBegin = maxEdges * destVertex;
455 for (synapseIndex = 0; synapseIndex < maxEdges; synapseIndex++) {
456 if (!allEdgesDevice->inUse_[synapseBegin + synapseIndex]) {
457 break;
458 }
459 }
460
461 allEdgesDevice->edgeCounts_[destVertex]++;
462
463 // create a synapse
464 switch (classSynapses_d) {
465 case classAllSpikingSynapses:
466 createSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, sourceIndex, destIndex,
467 sumPoint, deltaT, type);
468 break;
469 case classAllDSSynapses:
470 createDSSynapse(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
471 synapseIndex, sourceIndex, destIndex, sumPoint, deltaT, type);
472 break;
473 case classAllSTDPSynapses:
474 createSTDPSynapse(static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice),
475 destVertex, synapseIndex, sourceIndex, destIndex, sumPoint, deltaT,
476 type);
477 break;
478 case classAllDynamicSTDPSynapses:
479 createDynamicSTDPSynapse(
480 static_cast<AllDynamicSTDPSynapsesDeviceProperties *>(allEdgesDevice), destVertex,
481 synapseIndex, sourceIndex, destIndex, sumPoint, deltaT, type);
482 break;
483 default:
484 assert(false);
485 }
486 allEdgesDevice->W_[synapseBegin + synapseIndex] = W_d[srcVertex * numVertices + destVertex]
487 * edgSign(type)
488 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
489}

512CUDA_CALLABLE edgeType edgType(vertexType *neuronTypeMap_d, const int srcVertex,
513 const int destVertex)
514{
515 if (neuronTypeMap_d[srcVertex] == INH && neuronTypeMap_d[destVertex] == INH)
516 return II;
517 else if (neuronTypeMap_d[srcVertex] == INH && neuronTypeMap_d[destVertex] == EXC)
518 return IE;
519 else if (neuronTypeMap_d[srcVertex] == EXC && neuronTypeMap_d[destVertex] == INH)
520 return EI;
521 else if (neuronTypeMap_d[srcVertex] == EXC && neuronTypeMap_d[destVertex] == EXC)
522 return EE;
523
524 return ETYPE_UNDEF;
525}

542__global__ void updateSynapsesWeightsDevice(int numVertices, BGFLOAT deltaT, BGFLOAT *W_d,
543 int maxEdges,
544 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
545 AllSpikingSynapsesDeviceProperties *allEdgesDevice,
546 vertexType *neuronTypeMap_d)
547{
548 int idx = blockIdx.x * blockDim.x + threadIdx.x;
549 if (idx >= numVertices)
550 return;
551
552 int adjusted = 0;
553 //int could_have_been_removed = 0; // TODO: use this value
554 int removed = 0;
555 int added = 0;
556
557 // Scale and add sign to the areas
558 // visit each neuron 'a'
559 int destVertex = idx;
560
561 // and each destination neuron 'b'
562 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
563 // visit each synapse at (xa,ya)
564 bool connected = false;
565 edgeType type = edgType(neuronTypeMap_d, srcVertex, destVertex);
566
567 // for each existing synapse
568 BGSIZE existing_synapses = allEdgesDevice->edgeCounts_[destVertex];
569 int existingSynapsesChecked = 0;
570 for (BGSIZE synapseIndex = 0; (existingSynapsesChecked < existing_synapses) && !connected;
571 synapseIndex++) {
572 BGSIZE iEdg = maxEdges * destVertex + synapseIndex;
573 if (allEdgesDevice->inUse_[iEdg] == true) {
574 // if there is a synapse between a and b
575 if (allEdgesDevice->sourceVertexIndex_[iEdg] == srcVertex) {
576 connected = true;
577 adjusted++;
578
579 // adjust the strength of the synapse or remove
580 // it from the synapse map if it has gone below
581 // zero.
582 if (W_d[srcVertex * numVertices + destVertex] <= 0) {
583 removed++;
584 eraseSpikingSynapse(allEdgesDevice, destVertex, synapseIndex, maxEdges);
585 } else {
586 // adjust
587 // g_synapseStrengthAdjustmentConstant is 1.0e-8;
588 allEdgesDevice->W_[iEdg] = W_d[srcVertex * numVertices + destVertex]
589 * edgSign(type)
590 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
591 }
592 }
593 existingSynapsesChecked++;
594 }
595 }
596
597 // if not connected and weight(a,b) > 0, add a new synapse from a to b
598 if (!connected && (W_d[srcVertex * numVertices + destVertex] > 0)) {
599 // locate summation point
600 BGFLOAT *sumPoint = &(allVerticesDevice->summationMap_[destVertex]);
601 added++;
602
603 addSpikingSynapse(allEdgesDevice, type, srcVertex, destVertex, srcVertex, destVertex,
604 sumPoint, deltaT, W_d, numVertices);
605 }
606 }
607}
