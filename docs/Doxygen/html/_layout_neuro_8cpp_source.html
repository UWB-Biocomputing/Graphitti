47void LayoutNeuro::generateVertexTypeMap()
48{
49 LOG4CPLUS_DEBUG(fileLogger_, "\nInitializing vertex type map" << endl);
50
51 int numInhibitoryNeurons;
52 int numExcititoryNeurons;
53
54 // Set Neuron Type from GraphML File
55 GraphManager<NeuralVertexProperties>::VertexIterator vi, vi_end;
56 GraphManager<NeuralVertexProperties> &gm = GraphManager<NeuralVertexProperties>::getInstance();
57
58 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
59 assert(*vi < numVertices_);
60 if (gm[*vi].type == "INH") {
61 vertexTypeMap_[*vi] = vertexType::INH;
62 numInhibitoryNeurons++;
63 }
64 // Default Type is Excitatory
65 else {
66 vertexTypeMap_[*vi] = vertexType::EXC;
67 }
68 }
69
70 // Register vertexTypeMap to be recorded
71 Recorder &recorder = Simulator::getInstance().getModel().getRecorder();
72 recorder.registerVariable("vertexTypeMap", vertexTypeMap_, Recorder::UpdatedType::CONSTANT);
73
74 numExcititoryNeurons = numVertices_ - numInhibitoryNeurons;
75
76 LOG4CPLUS_DEBUG(fileLogger_, "\nVERTEX TYPE MAP"
77 << endl
78 << "\tTotal vertices: " << numVertices_ << endl
79 << "\tInhibitory Neurons: " << numInhibitoryNeurons << endl
80 << "\tExcitatory Neurons: " << numExcititoryNeurons << endl);
81 LOG4CPLUS_INFO(fileLogger_, "Finished initializing vertex type map");
82}

109edgeType LayoutNeuro::edgType(int srcVertex, int destVertex)
110{
111 if (vertexTypeMap_[srcVertex] == vertexType::INH
112 && vertexTypeMap_[destVertex] == vertexType::INH)
113 return edgeType::II;
114 else if (vertexTypeMap_[srcVertex] == vertexType::INH
115 && vertexTypeMap_[destVertex] == vertexType::EXC)
116 return edgeType::IE;
117 else if (vertexTypeMap_[srcVertex] == vertexType::EXC
118 && vertexTypeMap_[destVertex] == vertexType::INH)
119 return edgeType::EI;
120 else if (vertexTypeMap_[srcVertex] == vertexType::EXC
121 && vertexTypeMap_[destVertex] == vertexType::EXC)
122 return edgeType::EE;
123
124 return edgeType::ETYPE_UNDEF;
125}
