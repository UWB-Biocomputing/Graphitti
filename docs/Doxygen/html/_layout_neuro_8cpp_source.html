47void LayoutNeuro::generateVertexTypeMap()
48{
49 LOG4CPLUS_DEBUG(fileLogger_, "\nInitializing vertex type map" << endl);
50
51 int numInhibitoryNeurons;
52 int numExcititoryNeurons;
53
54 // Set Neuron Type from GraphML File
55 GraphManager::VertexIterator vi, vi_end;
56 GraphManager &gm = GraphManager::getInstance();
57
58 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
59 assert(*vi < numVertices_);
60 if (gm[*vi].type == "INH") {
61 vertexTypeMap_[*vi] = INH;
62 numInhibitoryNeurons++;
63 }
64 // Default Type is Excitatory
65 else {
66 vertexTypeMap_[*vi] = EXC;
67 }
68 }
69
70 numExcititoryNeurons = numVertices_ - numInhibitoryNeurons;
71
72 LOG4CPLUS_DEBUG(fileLogger_, "\nVERTEX TYPE MAP"
73 << endl
74 << "\tTotal vertices: " << numVertices_ << endl
75 << "\tInhibitory Neurons: " << numInhibitoryNeurons << endl
76 << "\tExcitatory Neurons: " << numExcititoryNeurons << endl);
77 LOG4CPLUS_INFO(fileLogger_, "Finished initializing vertex type map");
78}

105edgeType LayoutNeuro::edgType(int srcVertex, int destVertex)
106{
107 if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == INH)
108 return II;
109 else if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == EXC)
110 return IE;
111 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == INH)
112 return EI;
113 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == EXC)
114 return EE;
115
116 return ETYPE_UNDEF;
117}
