475CUDA_CALLABLE void stdpLearningDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice,
476 const BGSIZE iEdg, double delta, double epost, double epre)
477{
478 BGFLOAT STDPgap = allEdgesDevice->STDPgap_[iEdg];
479 BGFLOAT muneg = allEdgesDevice->muneg_[iEdg];
480 BGFLOAT mupos = allEdgesDevice->mupos_[iEdg];
481 BGFLOAT tauneg = allEdgesDevice->tauneg_[iEdg];
482 BGFLOAT taupos = allEdgesDevice->taupos_[iEdg];
483 BGFLOAT Aneg = allEdgesDevice->Aneg_[iEdg];
484 BGFLOAT Apos = allEdgesDevice->Apos_[iEdg];
485 BGFLOAT Wex = allEdgesDevice->Wex_[iEdg];
486 BGFLOAT &W = allEdgesDevice->W_[iEdg];
487 edgeType type = allEdgesDevice->type_[iEdg];
488 BGFLOAT dw;
489 if (delta < -STDPgap) {
490 // Depression
491 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
492 } else if (delta > STDPgap) {
493 // Potentiation
494 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
495 } else {
496 return;
497 }
498 // dw is the percentage change in synaptic strength; add 1.0 to become the scaling ratio
499 dw = 1.0 + dw * epre * epost;
500
501 // if scaling ratio is less than zero, set it to zero so this synapse, its strength is always zero
502 if (dw < 0) {
503 dw = 0;
504 }
505
506 // current weight multiplies dw (scaling ratio) to generate new weight
507 W *= dw;
508
509 // if new weight is bigger than Wex_ (maximum allowed weight), then set it to Wex_
510 if (fabs(W) > Wex) {
511 W = edgSign(type) * Wex;
512 }
513}

522CUDA_CALLABLE bool
523 isSTDPSynapseSpikeQueuePostDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice, BGSIZE iEdg)
524{
525 uint32_t &delayQueue = allEdgesDevice->delayQueuePost_[iEdg];
526 int &delayIndex = allEdgesDevice->delayIndexPost_[iEdg];
527 int delayQueueLength = allEdgesDevice->delayQueuePostLength_[iEdg];
528 uint32_t delayMask = (0x1 << delayIndex);
529 bool isFired = delayQueue & (delayMask);
530 delayQueue &= ~(delayMask);
531 if (++delayIndex >= delayQueueLength) {
532 delayIndex = 0;
533 }
534 return isFired;
535}

566__global__ void advanceSTDPSynapsesDevice(int totalSynapseCount,
567 EdgeIndexMapDevice *edgeIndexMapDevice,
568 uint64_t simulationStep, const BGFLOAT deltaT,
569 AllSTDPSynapsesDeviceProperties *allEdgesDevice,
570 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
571 int maxSpikes)
572{
573 int idx = blockIdx.x * blockDim.x + threadIdx.x;
574 if (idx >= totalSynapseCount)
575 return;
576
577 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
578
579 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
580 BGFLOAT &W = allEdgesDevice->W_[iEdg];
581 if (W <= 0.0) {
582 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
583 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
584
585 // Checks if there is an input spike in the queue.
586 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
587
588 // is an input in the queue?
589 if (isFired) {
590 switch (classSynapses_d) {
591 case classAllSTDPSynapses:
592 changeSpikingSynapsesPSRDevice(
593 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
594 simulationStep, deltaT);
595 break;
596 case classAllDynamicSTDPSynapses:
597 // Note: we cast void * over the allEdgesDevice, then recast it,
598 // because AllDSSynapsesDeviceProperties inherited properties from
599 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
600 changeDSSynapsePSRDevice(
601 static_cast<AllDSSynapsesDeviceProperties *>((void *)allEdgesDevice), iEdg,
602 simulationStep, deltaT);
603 break;
604 default:
605 assert(false);
606 }
607 }
608 // decay the post spike response
609 psr *= decay;
610 return;
611 }
612
613 BGFLOAT &decay = allEdgesDevice->decay_[iEdg];
614 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
615
616 // is an input in the queue?
617 bool fPre = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
618 bool fPost = isSTDPSynapseSpikeQueuePostDevice(allEdgesDevice, iEdg);
619 if (fPre || fPost) {
620 BGFLOAT &tauspre = allEdgesDevice->tauspre_[iEdg];
621 BGFLOAT &tauspost = allEdgesDevice->tauspost_[iEdg];

643 } else {

673 }

679 simulationStep, deltaT);

681 case classAllDynamicSTDPSynapses:

697 // just one before the last spike.

702 epost = 1.0 - exp(-delta / tauspost);
703 } else {
704 epost = 1.0;
705 }

707 // call the learning function stdpLearning() for each pair of
708 // post-pre spikes
709 int offIndex = -1; // last spike
710 while (true) {
711 spikeHistory
712 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, offIndex, maxSpikes);
713 if (spikeHistory == ULONG_MAX)
714 break;
715
716 if (spikeHistory + totalDelay > simulationStep) {
717 --offIndex;
718 continue;
719 }
720 // delta is the spike interval between post-pre spikes
721 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory - totalDelay) * deltaT;
722
723
724 if (delta >= 3.0 * taupos)
725 break;
726 if (useFroemkeDanSTDP) {
727 spikeHistory2 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre,
728 offIndex - 1, maxSpikes);
729 if (spikeHistory2 == ULONG_MAX)
730 break;
731 epre
732 = 1.0
733 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT) / tauspre);
734 } else {
735 epre = 1.0;
736 }
737 stdpLearningDevice(allEdgesDevice, iEdg, delta, epost, epre);
738 --offIndex;
739 }
740 }
741 }

744 psr *= decay;
