477CUDA_CALLABLE void stdpLearningDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice, BGSIZE iEdg,
478 double delta, double epost, double epre)
479{
480 BGFLOAT STDPgap = allEdgesDevice->STDPgap_[iEdg];
481 BGFLOAT muneg = allEdgesDevice->muneg_[iEdg];
482 BGFLOAT mupos = allEdgesDevice->mupos_[iEdg];
483 BGFLOAT tauneg = allEdgesDevice->tauneg_[iEdg];
484 BGFLOAT taupos = allEdgesDevice->taupos_[iEdg];
485 BGFLOAT Aneg = allEdgesDevice->Aneg_[iEdg];
486 BGFLOAT Apos = allEdgesDevice->Apos_[iEdg];
487 BGFLOAT Wex = allEdgesDevice->Wex_[iEdg];
488 BGFLOAT &W = allEdgesDevice->W_[iEdg];
489 edgeType type = allEdgesDevice->type_[iEdg];
490 BGFLOAT dw;
491 if (delta < -STDPgap) {
492 // Depression
493 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
494 } else if (delta > STDPgap) {
495 // Potentiation
496 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
497 } else {
498 return;
499 }
500 // dw is the percentage change in synaptic strength; add 1.0 to become the scaling ratio
501 dw = 1.0 + dw * epre * epost;
502
503 // if scaling ratio is less than zero, set it to zero so this synapse, its strength is always zero
504 if (dw < 0) {
505 dw = 0;
506 }
507
508 // current weight multiplies dw (scaling ratio) to generate new weight
509 W *= dw;
510
511 // if new weight is bigger than Wex_ (maximum allowed weight), then set it to Wex_
512 if (fabs(W) > Wex) {
513 W = edgSign(type) * Wex;
514 }
515}

524CUDA_CALLABLE bool
525 isSTDPSynapseSpikeQueuePostDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice, BGSIZE iEdg)
526{
527 uint32_t &delayQueue = allEdgesDevice->delayQueuePost_[iEdg];
528 int &delayIndex = allEdgesDevice->delayIndexPost_[iEdg];
529 int delayQueueLength = allEdgesDevice->delayQueuePostLength_[iEdg];
530 uint32_t delayMask = (0x1 << delayIndex);
531 bool isFired = delayQueue & (delayMask);
532 delayQueue &= ~(delayMask);
533 if (++delayIndex >= delayQueueLength) {
534 delayIndex = 0;
535 }
536 return isFired;
537}

568__global__ void advanceSTDPSynapsesDevice(int totalSynapseCount,
569 EdgeIndexMapDevice *edgeIndexMapDevice,
570 uint64_t simulationStep, BGFLOAT deltaT,
571 AllSTDPSynapsesDeviceProperties *allEdgesDevice,
572 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
573 int maxSpikes)
574{
575 int idx = blockIdx.x * blockDim.x + threadIdx.x;
576 if (idx >= totalSynapseCount)
577 return;
578
579 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
580
581 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
582 BGFLOAT &W = allEdgesDevice->W_[iEdg];
583 if (W <= 0.0) {
584 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
585 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
586
587 // Checks if there is an input spike in the queue.
588 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
589
590 // is an input in the queue?
591 if (isFired) {
592 switch (classSynapses_d) {
593 case classAllSTDPSynapses:
594 changeSpikingSynapsesPSRDevice(
595 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
596 simulationStep, deltaT);
597 break;
598 case classAllDynamicSTDPSynapses:
599 // Note: we cast void * over the allEdgesDevice, then recast it,
600 // because AllDSSynapsesDeviceProperties inherited properties from
601 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
602 changeDSSynapsePSRDevice(
603 static_cast<AllDSSynapsesDeviceProperties *>((void *)allEdgesDevice), iEdg,
604 simulationStep, deltaT);
605 break;
606 default:
607 assert(false);
608 }
609 }
610 // decay the post spike response
611 psr *= decay;
612 return;
613 }
614
615 BGFLOAT &decay = allEdgesDevice->decay_[iEdg];
616 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
617
618 // is an input in the queue?
619 bool fPre = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
620 bool fPost = isSTDPSynapseSpikeQueuePostDevice(allEdgesDevice, iEdg);

647 }

669 / tauspost);
670 } else {

675 }

681 simulationStep, deltaT);
682 break;

688 static_cast<AllDSSynapsesDeviceProperties *>((void *)allEdgesDevice), iEdg,

698 // so the getSpikeHistory w/offset = -2 will return the spike time
699 // just one before the last spike.
700 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPost, -2, maxSpikes);
701 if (spikeHistory > 0 && useFroemkeDanSTDP) {
702 // delta will include the transmission delay
703 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory) * deltaT;
704 epost = 1.0 - exp(-delta / tauspost);
705 } else {
706 epost = 1.0;
707 }
708
709 // call the learning function stdpLearning() for each pair of
710 // post-pre spikes
711 int offIndex = -1; // last spike
712 while (true) {
713 spikeHistory
714 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, offIndex, maxSpikes);
715 if (spikeHistory == ULONG_MAX)
716 break;
717
718 if (spikeHistory + totalDelay > simulationStep) {
719 --offIndex;
720 continue;
721 }
722 // delta is the spike interval between post-pre spikes
723 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory - totalDelay) * deltaT;
724
725
726 if (delta >= 3.0 * taupos)
727 break;
728 if (useFroemkeDanSTDP) {
729 spikeHistory2 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre,
730 offIndex - 1, maxSpikes);
731 if (spikeHistory2 == ULONG_MAX)
732 break;
733 epre
734 = 1.0
735 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT) / tauspre);
736 } else {
737 epre = 1.0;
738 }
739 stdpLearningDevice(allEdgesDevice, iEdg, delta, epost, epre);
740 --offIndex;
741 }
742 }
743 }
744
745 // decay the post spike response
746 psr *= decay;
