472CUDA_CALLABLE void stdpLearningDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice, BGSIZE iEdg,
473 double delta, double epost, double epre)
474{
475 BGFLOAT STDPgap = allEdgesDevice->STDPgap_[iEdg];
476 BGFLOAT muneg = allEdgesDevice->muneg_[iEdg];
477 BGFLOAT mupos = allEdgesDevice->mupos_[iEdg];
478 BGFLOAT tauneg = allEdgesDevice->tauneg_[iEdg];
479 BGFLOAT taupos = allEdgesDevice->taupos_[iEdg];
480 BGFLOAT Aneg = allEdgesDevice->Aneg_[iEdg];
481 BGFLOAT Apos = allEdgesDevice->Apos_[iEdg];
482 BGFLOAT Wex = allEdgesDevice->Wex_[iEdg];
483 BGFLOAT &W = allEdgesDevice->W_[iEdg];
484 edgeType type = allEdgesDevice->type_[iEdg];
485 BGFLOAT dw;
486 if (delta < -STDPgap) {
487 // Depression
488 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
489 } else if (delta > STDPgap) {
490 // Potentiation
491 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
492 } else {
493 return;
494 }
495 // dw is the percentage change in synaptic strength; add 1.0 to become the scaling ratio
496 dw = 1.0 + dw * epre * epost;
497
498 // if scaling ratio is less than zero, set it to zero so this synapse, its strength is always zero
499 if (dw < 0) {
500 dw = 0;
501 }
502
503 // current weight multiplies dw (scaling ratio) to generate new weight
504 W *= dw;
505
506 // if new weight is bigger than Wex_ (maximum allowed weight), then set it to Wex_
507 if (fabs(W) > Wex) {
508 W = edgSign(type) * Wex;
509 }
510}

519CUDA_CALLABLE bool
520 isSTDPSynapseSpikeQueuePostDevice(AllSTDPSynapsesDeviceProperties *allEdgesDevice, BGSIZE iEdg)
521{
522 uint32_t &delayQueue = allEdgesDevice->delayQueuePost_[iEdg];
523 int &delayIndex = allEdgesDevice->delayIndexPost_[iEdg];
524 int delayQueueLength = allEdgesDevice->delayQueuePostLength_[iEdg];
525 uint32_t delayMask = (0x1 << delayIndex);
526 bool isFired = delayQueue & (delayMask);
527 delayQueue &= ~(delayMask);
528 if (++delayIndex >= delayQueueLength) {
529 delayIndex = 0;
530 }
531 return isFired;
532}

563__global__ void advanceSTDPSynapsesDevice(int totalSynapseCount,
564 EdgeIndexMapDevice *edgeIndexMapDevice,
565 uint64_t simulationStep, BGFLOAT deltaT,
566 AllSTDPSynapsesDeviceProperties *allEdgesDevice,
567 AllSpikingNeuronsDeviceProperties *allVerticesDevice,
568 int maxSpikes)
569{
570 int idx = blockIdx.x * blockDim.x + threadIdx.x;
571 if (idx >= totalSynapseCount)
572 return;
573
574 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
575
576 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
577 BGFLOAT &W = allEdgesDevice->W_[iEdg];
578 if (W <= 0.0) {
579 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
580 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
581
582 // Checks if there is an input spike in the queue.
583 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
584
585 // is an input in the queue?
586 if (isFired) {
587 switch (classSynapses_d) {
588 case enumClassSynapses::classAllSTDPSynapses:
589 changeSpikingSynapsesPSRDevice(
590 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
591 simulationStep, deltaT);
592 break;
593 case enumClassSynapses::classAllDynamicSTDPSynapses:
594 // Note: we cast void * over the allEdgesDevice, then recast it,
595 // because AllDSSynapsesDeviceProperties inherited properties from
596 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
597 changeDSSynapsePSRDevice(

606 psr *= decay;

615 bool fPost = isSTDPSynapseSpikeQueuePostDevice(allEdgesDevice, iEdg);
616 if (fPre || fPost) {

618 BGFLOAT &tauspost = allEdgesDevice->tauspost_[iEdg];
619 BGFLOAT &taupos = allEdgesDevice->taupos_[iEdg];

628 BGFLOAT delta;

632 // spikeCount points to the next available position of spike_history,
633 // so the getSpikeHistory w/offset = -2 will return the spike time
634 // just one before the last spike.
635 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, -2, maxSpikes);
636 if (spikeHistory > 0 && useFroemkeDanSTDP) {

639 epre = 1.0 - exp(-delta / tauspre);
640 } else {
641 epre = 1.0;
642 }

645 // pre-post spikes
646 int offIndex = -1; // last spike
647 while (true) {
648 spikeHistory

650 if (spikeHistory == ULONG_MAX)
651 break;

656 break;

659 offIndex - 1, maxSpikes);
660 if (spikeHistory2 == ULONG_MAX)
661 break;
662 epost = 1.0
663 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT)
664 / tauspost);
665 } else {
666 epost = 1.0;
667 }

669 --offIndex;
670 }

674 changeSpikingSynapsesPSRDevice(
675 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
676 simulationStep, deltaT);
677 break;
678 case enumClassSynapses::classAllDynamicSTDPSynapses:
679 // Note: we cast void * over the allEdgesDevice, then recast it,
680 // because AllDSSynapsesDeviceProperties inherited properties from
681 // the AllDSSynapsesDeviceProperties and the AllSTDPSynapsesDeviceProperties.
682 changeDSSynapsePSRDevice(
683 static_cast<AllDSSynapsesDeviceProperties *>((void *)allEdgesDevice), iEdg,
684 simulationStep, deltaT);
685 break;
686 default:
687 assert(false);
688 }

693 // so the getSpikeHistory w/offset = -2 will return the spike time
694 // just one before the last spike.
695 spikeHistory = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPost, -2, maxSpikes);
696 if (spikeHistory > 0 && useFroemkeDanSTDP) {
697 // delta will include the transmission delay
698 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory) * deltaT;
699 epost = 1.0 - exp(-delta / tauspost);
700 } else {
701 epost = 1.0;
702 }
703
704 // call the learning function stdpLearning() for each pair of
705 // post-pre spikes
706 int offIndex = -1; // last spike
707 while (true) {
708 spikeHistory
709 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre, offIndex, maxSpikes);
710 if (spikeHistory == ULONG_MAX)
711 break;
712
713 if (spikeHistory + totalDelay > simulationStep) {
714 --offIndex;
715 continue;
716 }
717 // delta is the spike interval between post-pre spikes
718 delta = static_cast<BGFLOAT>(simulationStep - spikeHistory - totalDelay) * deltaT;
719
720
721 if (delta >= 3.0 * taupos)
722 break;
723 if (useFroemkeDanSTDP) {
724 spikeHistory2 = getSTDPSynapseSpikeHistoryDevice(allVerticesDevice, idxPre,
725 offIndex - 1, maxSpikes);
726 if (spikeHistory2 == ULONG_MAX)
727 break;
728 epre
729 = 1.0
730 - exp(-(static_cast<BGFLOAT>(spikeHistory - spikeHistory2) * deltaT) / tauspre);
731 } else {
732 epre = 1.0;
733 }
734 stdpLearningDevice(allEdgesDevice, iEdg, delta, epost, epre);
735 --offIndex;
736 }
737 }
738 }
739
740 // decay the post spike response
741 psr *= decay;
