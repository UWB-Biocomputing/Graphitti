220__global__ void advanceIZHNeuronsDevice(int totalVertices, int maxEdges, int maxSpikes,
221 const BGFLOAT deltaT, uint64_t simulationStep,
222 float *randNoise,
223 AllIZHNeuronsDeviceProperties *allVerticesDevice,
224 AllSpikingSynapsesDeviceProperties *allEdgesDevice,
225 EdgeIndexMapDevice *edgeIndexMapDevice,
226 bool fAllowBackPropagation)
227{
228 // determine which neuron this thread is processing
229 int idx = blockIdx.x * blockDim.x + threadIdx.x;
230 if (idx >= totalVertices)
231 return;
232
233 allVerticesDevice->hasFired_[idx] = false;
234 BGFLOAT &sp = allVerticesDevice->summationMap_[idx];
235 BGFLOAT &vm = allVerticesDevice->Vm_[idx];
236 BGFLOAT &a = allVerticesDevice->Aconst_[idx];
237 BGFLOAT &b = allVerticesDevice->Bconst_[idx];
238 BGFLOAT &u = allVerticesDevice->u_[idx];
239 BGFLOAT r_sp = sp;
240 BGFLOAT r_vm = vm;
241 BGFLOAT r_a = a;
242 BGFLOAT r_b = b;
243 BGFLOAT r_u = u;
244
245 if (allVerticesDevice->numStepsInRefractoryPeriod_[idx] > 0) { // is neuron refractory?
246 --allVerticesDevice->numStepsInRefractoryPeriod_[idx];
247 } else if (r_vm >= allVerticesDevice->Vthresh_[idx]) { // should it fire?
248 int &spikeCount = allVerticesDevice->numEventsInEpoch_[idx];
249 // Note that the neuron has fired!
250 allVerticesDevice->hasFired_[idx] = true;
251 // record spike time
252 int &queueEnd = allVerticesDevice->queueEnd_[idx];
253 allVerticesDevice->spikeHistory_[idx][queueEnd] = simulationStep;
254 spikeCount++;
255
256 queueEnd = (queueEnd + 1) % maxSpikes;
257
258 // calculate the number of steps in the absolute refractory period
259 allVerticesDevice->numStepsInRefractoryPeriod_[idx]
260 = static_cast<int>(allVerticesDevice->Trefract_[idx] / deltaT + 0.5);
261
262 // reset to 'Vreset'
263 vm = allVerticesDevice->Cconst_[idx] * 0.001;
264 u = r_u + allVerticesDevice->Dconst_[idx];
265
266 // notify outgoing synapses of spike
267 BGSIZE synapseCounts = edgeIndexMapDevice->outgoingEdgeCount_[idx];
268 if (synapseCounts != 0) {
269 // get the index of where this neuron's list of synapses are
270 BGSIZE beginIndex = edgeIndexMapDevice->outgoingEdgeBegin_[idx];
271 // get the memory location of where that list begins
272 BGSIZE *outgoingMapBegin = &(edgeIndexMapDevice->outgoingEdgeIndexMap_[beginIndex]);
273
274 // for each synapse, let them know we have fired
275 for (BGSIZE i = 0; i < synapseCounts; i++) {
276 preSpikingSynapsesSpikeHitDevice(outgoingMapBegin[i], allEdgesDevice);
277 }
278 }
279
280 // notify incomming synapses of spike
281 synapseCounts = edgeIndexMapDevice->incomingEdgeCount_[idx];
282 if (fAllowBackPropagation && synapseCounts != 0) {
283 // get the index of where this neuron's list of synapses are
284 BGSIZE beginIndex = edgeIndexMapDevice->incomingEdgeBegin_[idx];
285 // get the memory location of where that list begins
286 BGSIZE *incomingMapBegin = &(edgeIndexMapDevice->incomingEdgeIndexMap_[beginIndex]);
287
288 // for each synapse, let them know we have fired
289 switch (classSynapses_d) {
290 case classAllSTDPSynapses:
291 case classAllDynamicSTDPSynapses:
292 for (BGSIZE i = 0; i < synapseCounts; i++) {
293 postSTDPSynapseSpikeHitDevice(
294 incomingMapBegin[i],
295 static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice));
296 } // end for
297 break;
298
299 case classAllSpikingSynapses:
300 case classAllDSSynapses:
301 for (BGSIZE i = 0; i < synapseCounts; i++) {
302 postSpikingSynapsesSpikeHitDevice(incomingMapBegin[i], allEdgesDevice);
303 } // end for
304 break;
305
306 default:
307 assert(false);
308 } // end switch
309 }
310 } else {
311 r_sp += allVerticesDevice->I0_[idx]; // add IO
312
313 // Random number alg. goes here
314 r_sp += (randNoise[idx] * allVerticesDevice->Inoise_[idx]); // add cheap noise
315
316 BGFLOAT Vint = r_vm * 1000;
317
318 // Izhikevich model integration step
319 BGFLOAT Vb = Vint + allVerticesDevice->C3_[idx] * (0.04 * Vint * Vint + 5 * Vint + 140 - u);
320 u = r_u + allVerticesDevice->C3_[idx] * r_a * (r_b * Vint - r_u);
321
322 vm = Vb * 0.001 + allVerticesDevice->C2_[idx] * r_sp; // add inputs
323 }
324
325 // clear synaptic input for next time step
326 sp = 0;
327}
