219__global__ void advanceIZHNeuronsDevice(int totalVertices, int maxEdges, int maxSpikes,
220 BGFLOAT deltaT, uint64_t simulationStep, float randNoise[],
221 AllIZHNeuronsDeviceProperties *allVerticesDevice,
222 AllSpikingSynapsesDeviceProperties *allEdgesDevice,
223 EdgeIndexMapDevice *edgeIndexMapDevice,
224 bool fAllowBackPropagation)
225{
226 // determine which neuron this thread is processing
227 int idx = blockIdx.x * blockDim.x + threadIdx.x;
228 if (idx >= totalVertices)
229 return;
230
231 allVerticesDevice->hasFired_[idx] = false;
232 BGFLOAT &sp = allVerticesDevice->summationPoints_[idx];
233 BGFLOAT &vm = allVerticesDevice->Vm_[idx];
234 BGFLOAT &a = allVerticesDevice->Aconst_[idx];
235 BGFLOAT &b = allVerticesDevice->Bconst_[idx];
236 BGFLOAT &u = allVerticesDevice->u_[idx];
237 BGFLOAT r_sp = sp;
238 BGFLOAT r_vm = vm;
239 BGFLOAT r_a = a;
240 BGFLOAT r_b = b;
241 BGFLOAT r_u = u;
242
243 if (allVerticesDevice->numStepsInRefractoryPeriod_[idx] > 0) { // is neuron refractory?
244 --allVerticesDevice->numStepsInRefractoryPeriod_[idx];
245 } else if (r_vm >= allVerticesDevice->Vthresh_[idx]) { // should it fire?
246 int &spikeCount = allVerticesDevice->numElementsInEpoch_[idx];
247 // Note that the neuron has fired!
248 allVerticesDevice->hasFired_[idx] = true;
249 // record spike time
250 int &queueEnd = allVerticesDevice->bufferEnd_[idx];
251 allVerticesDevice->spikeHistory_[idx][queueEnd] = simulationStep;
252 spikeCount++;
253
254 queueEnd = (queueEnd + 1) % maxSpikes;
255
256 // calculate the number of steps in the absolute refractory period
257 allVerticesDevice->numStepsInRefractoryPeriod_[idx]
258 = static_cast<int>(allVerticesDevice->Trefract_[idx] / deltaT + 0.5);
259
260 // reset to 'Vreset'
261 vm = allVerticesDevice->Cconst_[idx] * 0.001;
262 u = r_u + allVerticesDevice->Dconst_[idx];
263
264 // notify outgoing synapses of spike
265 BGSIZE synapseCounts = edgeIndexMapDevice->outgoingEdgeCount_[idx];
266 if (synapseCounts != 0) {
267 // get the index of where this neuron's list of synapses are
268 BGSIZE beginIndex = edgeIndexMapDevice->outgoingEdgeBegin_[idx];
269 // get the memory location of where that list begins
270 BGSIZE *outgoingMapBegin = &(edgeIndexMapDevice->outgoingEdgeIndexMap_[beginIndex]);
271
272 // for each synapse, let them know we have fired
273 for (BGSIZE i = 0; i < synapseCounts; i++) {
274 preSpikingSynapsesSpikeHitDevice(outgoingMapBegin[i], allEdgesDevice);
275 }
276 }
277
278 // notify incomming synapses of spike
279 synapseCounts = edgeIndexMapDevice->incomingEdgeCount_[idx];
280 if (fAllowBackPropagation && synapseCounts != 0) {
281 // get the index of where this neuron's list of synapses are
282 BGSIZE beginIndex = edgeIndexMapDevice->incomingEdgeBegin_[idx];
283 // get the memory location of where that list begins
284 BGSIZE *incomingMapBegin = &(edgeIndexMapDevice->incomingEdgeIndexMap_[beginIndex]);
285
286 // for each synapse, let them know we have fired
287 switch (classSynapses_d) {
288 case classAllSTDPSynapses:
289 case classAllDynamicSTDPSynapses:
290 for (BGSIZE i = 0; i < synapseCounts; i++) {
291 postSTDPSynapseSpikeHitDevice(
292 incomingMapBegin[i],
293 static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice));
294 } // end for
295 break;
296
297 case classAllSpikingSynapses:
298 case classAllDSSynapses:
299 for (BGSIZE i = 0; i < synapseCounts; i++) {
300 postSpikingSynapsesSpikeHitDevice(incomingMapBegin[i], allEdgesDevice);
301 } // end for
302 break;
303
304 default:
305 assert(false);
306 } // end switch
307 }
308 } else {
309 r_sp += allVerticesDevice->I0_[idx]; // add IO
310
311 // Random number alg. goes here
312 r_sp += (randNoise[idx] * allVerticesDevice->Inoise_[idx]); // add cheap noise
313
314 BGFLOAT Vint = r_vm * 1000;
315
316 // Izhikevich model integration step
317 BGFLOAT Vb = Vint + allVerticesDevice->C3_[idx] * (0.04 * Vint * Vint + 5 * Vint + 140 - u);
318 u = r_u + allVerticesDevice->C3_[idx] * r_a * (r_b * Vint - r_u);
319
320 vm = Vb * 0.001 + allVerticesDevice->C2_[idx] * r_sp; // add inputs
321 }
322
323 // clear synaptic input for next time step
324 sp = 0;
325}
