111 // Write MT_RNG_COUNT vertical lanes of nPerRng random numbers to *d_Random.
112 // For coalesced global writes MT_RNG_COUNT should be a multiple of warp size.
113 // Initial states for each generator are the same, since the states are
114 // initialized from the global seed. In order to improve distribution properties
115 // on small NPerRng supply dedicated (local) seed to each twister.
116 // The local seeds, in their turn, can be extracted from global seed
117 // by means of any simple random number generator, like LCG.
119 __global__ void RandomGPU(
120  float *d_Random,
121  int nPerRng, int mt_rng_count)
122 {
123  const int tid = blockDim.x * blockIdx.x + threadIdx.x;
124  int iState, iState1, iStateM, iOut;
125  unsigned int mti, mti1, mtiM, x;
126  unsigned int matrix_a, mask_b, mask_c;
127 
128  //Load bit-vector Mersenne Twister parameters
129  matrix_a = ds_MT[tid].matrix_a;
130  mask_b = ds_MT[tid].mask_b;
131  mask_c = ds_MT[tid].mask_c;
132 
133  iState = ds_MT[tid].iState;
134  mti1 = mt[iState];
135  for (iOut = 0; iOut < nPerRng; iOut++) {
136  iState1 = iState + 1;
137  iStateM = iState + MT_MM;
138  if(iState1 >= MT_NN*(1+tid)) iState1 -= MT_NN;
139  if(iStateM >= MT_NN*(1+tid)) iStateM -= MT_NN;
140  mti = mti1;
141  mti1 = mt[iState1];
142  mtiM = mt[iStateM];
143 
144  // MT recurrence
145  x = (mti & MT_UMASK) | (mti1 & MT_LMASK);
146  x = mtiM ^ (x >> 1) ^ ((x & 1) ? matrix_a : 0);
147 
148  mt[iState] = x;
149  iState = iState1;
150 
151  //Tempering transformation
152  x ^= (x >> MT_SHIFT0);
153  x ^= (x << MT_SHIFTB) & mask_b;
154  x ^= (x << MT_SHIFTC) & mask_c;
155  x ^= (x >> MT_SHIFT1);
156 
157  //Convert to (0, 1] float and write to global memory
158  d_Random[tid + iOut * mt_rng_count] = ((float)x + 1.0f) / 4294967296.0f;
159  }
160  ds_MT[tid].iState = iState;
161 }

164 // Transform each of MT_RNG_COUNT lanes of nPerRng uniformly distributed
165 // random samples, produced by RandomGPU(), to normally distributed lanes
166 // using Cartesian form of Box-Muller transformation.
167 // nPerRng must be even.
169 #define PI 3.14159265358979f
170 __device__ inline void BoxMuller(float& u1, float& u2){
171  float r = sqrtf(-2.0f * logf(u1));
172  float phi = 2 * PI * u2;
173  u1 = r * __cosf(phi);
174  u2 = r * __sinf(phi);
175 }
