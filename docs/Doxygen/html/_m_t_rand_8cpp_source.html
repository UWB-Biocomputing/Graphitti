116uint32_t MTRand::randInt()
117{
118 // Pull a 32-bit integer from the generator state
119 // Every other access function simply transforms the numbers extracted here
120
121 if (left == 0)
122 reload();
123 --left;
124
125 uint32_t s1;
126 s1 = state[iNext++];
127 s1 ^= (s1 >> 11);
128 s1 ^= (s1 << 7) & 0x9d2c5680UL;
129 s1 ^= (s1 << 15) & 0xefc60000UL;
130 return (s1 ^ (s1 >> 18));
131}

133uint32_t MTRand::randInt(uint32_t n)
134{
135 // Find which bits are used in n
136 // Optimized by Magnus Jonsson (magnus@smartelectronix.com)
137 uint32_t used = n;
138 used |= used >> 1;
139 used |= used >> 2;
140 used |= used >> 4;
141 used |= used >> 8;
142 used |= used >> 16;
143
144 // Draw numbers until one is found in [0,n]
145 uint32_t i;
146 do {
147 i = randInt() & used; // toss unused bits to shorten search
148 } while (i > n);
149 return i;
150}

167BGFLOAT MTRand::randNorm(BGFLOAT mean, BGFLOAT variance)
168{
169 // Return a real number from a normal (Gaussian) distribution with given
170 // mean and variance by Box-Muller method
171 double r = sqrt(-2.0 * log(1.0 - randDblExc())) * variance;
172 double phi = 2.0 * 3.14159265358979323846264338328 * randExc();
173 return static_cast<BGFLOAT>(mean + r * cos(phi));
174}

185void MTRand::seed(uint32_t *const bigSeed, uint32_t seedLength)
186{
187 // Seed the generator with an array of uint32_t's
188 // There are 2^19937-1 possible initial states. This function allows
189 // all of those to be accessed by providing at least 19937 bits (with a
190 // default seed length of N = 624 uint32_t's). Any bits above the lower 32
191 // in each element are discarded.
192 // Just call seed() if you want to get array from /dev/urandom
193 initialize(19650218UL);
194 int i = 1;
195 uint32_t j = 0;
196 int k = (N > seedLength ? N : seedLength);
197 for (; k; --k) {
198 state[i] = state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1664525UL);
199 state[i] += (bigSeed[j] & 0xffffffffUL) + j;
200 state[i] &= 0xffffffffUL;
201 ++i;
202 ++j;
203 if (i >= N) {
204 state[0] = state[N - 1];
205 i = 1;
206 }
207 if (j >= seedLength)
208 j = 0;
209 }
210 for (k = N - 1; k; --k) {
211 state[i] = state[i] ^ ((state[i - 1] ^ (state[i - 1] >> 30)) * 1566083941UL);
212 state[i] -= i;
213 state[i] &= 0xffffffffUL;
214 ++i;
215 if (i >= N) {
216 state[0] = state[N - 1];
217 i = 1;
218 }
219 }
220 state[0] = 0x80000000UL; // MSB is 1, assuring non-zero initial array
221 reload();
222}

225void MTRand::seed()
226{
227 // Seed the generator with an array from /dev/urandom if available
228 // Otherwise use a hash of time() and clock() values
229
230 // First try getting an array from /dev/urandom
231 FILE *urandom = fopen("/dev/urandom", "rb");
232 if (urandom) {
233 uint32_t bigSeed[N];
234 uint32_t *s = bigSeed;
235 int i = N;
236 bool success = true;
237 while (success && i--)
238 success = fread(s++, sizeof(uint32_t), 1, urandom);
239 fclose(urandom);
240 if (success) {
241 seed(bigSeed, N);
242 return;
243 }
244 }
245
246 // Was not successful, so use time() and clock() instead
247 seed(hash(time(nullptr), clock()));
248}

295void MTRand::initialize(uint32_t seed)
296{
297 // Initialize generator state with seed
298 // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
299 // In previous versions, most significant bits (MSBs) of the seed affect
300 // only MSBs of the state array. Modified 9 Jan 2002 by Makoto Matsumoto.
301 uint32_t *s = state;
302 uint32_t *r = state;
303 int i = 1;
304 *s++ = seed & 0xffffffffUL;
305 for (; i < N; ++i) {
306 *s++ = (1812433253UL * (*r ^ (*r >> 30)) + i) & 0xffffffffUL;
307 r++;
308 }
309}

312void MTRand::reload()
313{
314 // Generate N new values in state
315 // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
316 uint32_t *p = state;
317 int i;
318 for (i = N - M; i--; ++p)
319 *p = twist(p[M], p[0], p[1]);
320 for (i = M; --i; ++p)
321 *p = twist(p[M - N], p[0], p[1]);
322 *p = twist(p[M - N], p[0], state[0]);
323
324 left = N, iNext = 0;
325}

328uint32_t MTRand::hash(time_t t, clock_t c)
329{
330 // Get a uint32_t from t and c
331 // Better than uint32_t(x) in case x is floating point in [0,1]
332 // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
333
334 static uint32_t differ = 0; // guarantee time-based seeds will change
335
336 uint32_t h1 = 0;
337 unsigned char *p = (unsigned char *)&t;
338 for (size_t i = 0; i < sizeof(t); ++i) {
339 h1 *= UCHAR_MAX + 2U;
340 h1 += p[i];
341 }
342 uint32_t h2 = 0;
343 p = (unsigned char *)&c;
344 for (size_t j = 0; j < sizeof(c); ++j) {
345 h2 *= UCHAR_MAX + 2U;
346 h2 += p[j];
347 }
348 return (h1 + differ++) ^ h2;
349}
