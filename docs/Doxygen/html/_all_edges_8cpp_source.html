154void AllEdges::createEdgeIndexMap(EdgeIndexMap &edgeIndexMap)
155{
156 Simulator &simulator = Simulator::getInstance();
157 int vertexCount = simulator.getTotalVertices();
158 int totalEdgeCount = 0;
159
160 // count the total edges
161 for (int i = 0; i < vertexCount; i++) {
162 assert(static_cast<int>(edgeCounts_[i]) < simulator.getMaxEdgesPerVertex());
163 totalEdgeCount += edgeCounts_[i];
164 }
165
166 LOG4CPLUS_TRACE(fileLogger_, endl
167 << "totalEdgeCount: in edgeIndexMap " << totalEdgeCount << endl);
168
169 // Create vector for edge forwarding map
170 vector<BGSIZE> rgEdgeEdgeIndexMap[vertexCount];
171
172 BGSIZE edg_i = 0;
173 int curr = 0;
174
175 LOG4CPLUS_TRACE(edgeLogger_,
176 "\nSize of edge Index Map " << vertexCount << "," << totalEdgeCount << endl);
177
178 for (int i = 0; i < vertexCount; i++) {
179 BGSIZE edge_count = 0;
180 edgeIndexMap.incomingEdgeBegin_[i] = curr;
181 for (int j = 0; j < simulator.getMaxEdgesPerVertex(); j++, edg_i++) {
182 if (inUse_[edg_i]) {
183 int idx = sourceVertexIndex_[edg_i];
184 rgEdgeEdgeIndexMap[idx].push_back(edg_i);
185
186 edgeIndexMap.incomingEdgeIndexMap_[curr] = edg_i;
187 curr++;
188 edge_count++;
189 }
190 }
191
192 if (edge_count != edgeCounts_[i]) {
193 LOG4CPLUS_FATAL(edgeLogger_,
194 "\nedge_count does not match edgeCounts_" << edge_count << endl);
195 throw runtime_error("createEdgeIndexMap: edge_count does not match edgeCounts_.");
196 }
197
198 edgeIndexMap.incomingEdgeCount_[i] = edge_count;
199 }
200
201 if (totalEdgeCount != curr) {
202 LOG4CPLUS_FATAL(edgeLogger_,
203 "Curr does not match the totalEdgeCount. curr are " << curr << endl);
204 throw runtime_error("createEdgeIndexMap: Curr does not match the totalEdgeCount.");
205 }
206 totalEdgeCount_ = totalEdgeCount;
207 LOG4CPLUS_DEBUG(edgeLogger_, endl << "totalEdgeCount: " << totalEdgeCount_ << endl);
208
209 edg_i = 0;
210 for (int i = 0; i < vertexCount; i++) {
211 edgeIndexMap.outgoingEdgeBegin_[i] = edg_i;
212 edgeIndexMap.outgoingEdgeCount_[i] = rgEdgeEdgeIndexMap[i].size();
213
214 for (BGSIZE j = 0; j < rgEdgeEdgeIndexMap[i].size(); j++, edg_i++) {
215 edgeIndexMap.outgoingEdgeIndexMap_[edg_i] = rgEdgeEdgeIndexMap[i][j];
216 }
217 }
218}

256BGSIZE AllEdges::addEdge(edgeType type, int srcVertex, int destVertex, BGFLOAT deltaT)
257{
258 BGSIZE iEdg;
259 if (edgeCounts_[destVertex] >= maxEdgesPerVertex_) {
260 LOG4CPLUS_FATAL(edgeLogger_, "Vertex : " << destVertex << " ran out of space for new edges.");
261 throw runtime_error("Vertex " + to_string(destVertex)
262 + " ran out of space for new edges, current edge = "
263 + to_string(edgeCounts_[destVertex]));
264 }
265
266 // add it to the list: find first edge location for vertex destVertex
267 // that isn't in use.
268 for (BGSIZE i = 0; i < maxEdgesPerVertex_; i++) {
269 iEdg = maxEdgesPerVertex_ * destVertex + i;
270 if (!inUse_[iEdg]) {
271 break;
272 }
273 }
274
275 edgeCounts_[destVertex]++;
276
277 // create an edge
278 createEdge(iEdg, srcVertex, destVertex, deltaT, type);
279 return iEdg;
280}
