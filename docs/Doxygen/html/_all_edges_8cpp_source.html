168void AllEdges::createEdgeIndexMap(EdgeIndexMap &edgeIndexMap)
169{
170 Simulator &simulator = Simulator::getInstance();
171 int vertexCount = simulator.getTotalVertices();
172 int totalEdgeCount = 0;
173
174 // count the total edges
175 for (int i = 0; i < vertexCount; i++) {
176 assert(static_cast<int>(edgeCounts_[i]) < simulator.getMaxEdgesPerVertex());
177 totalEdgeCount += edgeCounts_[i];
178 }
179
180 LOG4CPLUS_TRACE(fileLogger_, endl
181 << "totalEdgeCount: in edgeIndexMap " << totalEdgeCount << endl);
182
183 // Create vector for edge forwarding map
184 vector<BGSIZE> rgEdgeEdgeIndexMap[vertexCount];
185
186 BGSIZE edg_i = 0;
187 int curr = 0;
188
189 LOG4CPLUS_TRACE(edgeLogger_,
190 "\nSize of edge Index Map " << vertexCount << "," << totalEdgeCount << endl);
191
192 for (int i = 0; i < vertexCount; i++) {
193 BGSIZE edge_count = 0;
194 edgeIndexMap.incomingEdgeBegin_[i] = curr;
195 for (int j = 0; j < simulator.getMaxEdgesPerVertex(); j++, edg_i++) {
196 if (inUse_[edg_i]) {
197 int idx = sourceVertexIndex_[edg_i];
198 rgEdgeEdgeIndexMap[idx].push_back(edg_i);
199
200 edgeIndexMap.incomingEdgeIndexMap_[curr] = edg_i;
201 curr++;
202 edge_count++;
203 }
204 }
205
206 if (edge_count != edgeCounts_[i]) {
207 LOG4CPLUS_FATAL(edgeLogger_,
208 "\nedge_count does not match edgeCounts_" << edge_count << endl);
209 throw runtime_error("createEdgeIndexMap: edge_count does not match edgeCounts_.");
210 }
211
212 edgeIndexMap.incomingEdgeCount_[i] = edge_count;
213 }
214
215 if (totalEdgeCount != curr) {
216 LOG4CPLUS_FATAL(edgeLogger_,
217 "Curr does not match the totalEdgeCount. curr are " << curr << endl);
218 throw runtime_error("createEdgeIndexMap: Curr does not match the totalEdgeCount.");
219 }
220 totalEdgeCount_ = totalEdgeCount;
221 LOG4CPLUS_DEBUG(edgeLogger_, endl << "totalEdgeCount: " << totalEdgeCount_ << endl);
222
223 edg_i = 0;
224 for (int i = 0; i < vertexCount; i++) {
225 edgeIndexMap.outgoingEdgeBegin_[i] = edg_i;
226 edgeIndexMap.outgoingEdgeCount_[i] = rgEdgeEdgeIndexMap[i].size();
227
228 for (BGSIZE j = 0; j < rgEdgeEdgeIndexMap[i].size(); j++, edg_i++) {
229 edgeIndexMap.outgoingEdgeIndexMap_[edg_i] = rgEdgeEdgeIndexMap[i][j];
230 }
231 }
232}

272void AllEdges::addEdge(BGSIZE &iEdg, edgeType type, const int srcVertex, const int destVertex,
273 BGFLOAT *sumPoint, const BGFLOAT deltaT)
274{
275 if (edgeCounts_[destVertex] >= maxEdgesPerVertex_) {
276 LOG4CPLUS_FATAL(edgeLogger_, "Vertex : " << destVertex << " ran out of space for new edges.");
277 throw runtime_error("Vertex " + to_string(destVertex)
278 + " ran out of space for new edges, current edge = "
279 + to_string(edgeCounts_[destVertex]));
280 }
281
282 // add it to the list: find first edge location for vertex destVertex
283 // that isn't in use.
284 for (BGSIZE i = 0; i < maxEdgesPerVertex_; i++) {
285 iEdg = maxEdgesPerVertex_ * destVertex + i;
286 if (!inUse_[iEdg]) {
287 break;
288 }
289 }
290
291 edgeCounts_[destVertex]++;
292
293 // create an edge
294 createEdge(iEdg, srcVertex, destVertex, sumPoint, deltaT, type);
295}
