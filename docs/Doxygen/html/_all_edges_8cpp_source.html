176void AllEdges::createEdgeIndexMap(EdgeIndexMap &edgeIndexMap)
177{
178 Simulator &simulator = Simulator::getInstance();
179 int vertexCount = simulator.getTotalVertices();
180 int totalEdgeCount = 0;
181
182 // count the total edges
183 for (int i = 0; i < vertexCount; i++) {
184 assert(static_cast<int>(edgeCounts_[i]) < simulator.getMaxEdgesPerVertex());
185 totalEdgeCount += edgeCounts_[i];
186 }
187
188 LOG4CPLUS_TRACE(fileLogger_, endl
189 << "totalEdgeCount: in edgeIndexMap " << totalEdgeCount << endl);
190
191 // Create vector for edge forwarding map
192 vector<BGSIZE> rgEdgeEdgeIndexMap[vertexCount];
193
194 BGSIZE edg_i = 0;
195 int curr = 0;
196
197 LOG4CPLUS_TRACE(edgeLogger_,
198 "\nSize of edge Index Map " << vertexCount << "," << totalEdgeCount << endl);
199
200 for (int i = 0; i < vertexCount; i++) {
201 BGSIZE edge_count = 0;
202 edgeIndexMap.incomingEdgeBegin_[i] = curr;
203 for (int j = 0; j < simulator.getMaxEdgesPerVertex(); j++, edg_i++) {
204 if (inUse_[edg_i]) {
205 int idx = sourceVertexIndex_[edg_i];
206 rgEdgeEdgeIndexMap[idx].push_back(edg_i);
207
208 edgeIndexMap.incomingEdgeIndexMap_[curr] = edg_i;
209 curr++;
210 edge_count++;
211 }
212 }
213
214 if (edge_count != edgeCounts_[i]) {
215 LOG4CPLUS_FATAL(edgeLogger_,
216 "\nedge_count does not match edgeCounts_" << edge_count << endl);
217 throw runtime_error("createEdgeIndexMap: edge_count does not match edgeCounts_.");
218 }
219
220 edgeIndexMap.incomingEdgeCount_[i] = edge_count;
221 }
222
223 if (totalEdgeCount != curr) {
224 LOG4CPLUS_FATAL(edgeLogger_,
225 "Curr does not match the totalEdgeCount. curr are " << curr << endl);
226 throw runtime_error("createEdgeIndexMap: Curr does not match the totalEdgeCount.");
227 }
228 totalEdgeCount_ = totalEdgeCount;
229 LOG4CPLUS_DEBUG(edgeLogger_, endl << "totalEdgeCount: " << totalEdgeCount_ << endl);
230
231 edg_i = 0;
232 for (int i = 0; i < vertexCount; i++) {
233 edgeIndexMap.outgoingEdgeBegin_[i] = edg_i;
234 edgeIndexMap.outgoingEdgeCount_[i] = rgEdgeEdgeIndexMap[i].size();
235
236 for (BGSIZE j = 0; j < rgEdgeEdgeIndexMap[i].size(); j++, edg_i++) {
237 edgeIndexMap.outgoingEdgeIndexMap_[edg_i] = rgEdgeEdgeIndexMap[i][j];
238 }
239 }
240}

278BGSIZE AllEdges::addEdge(edgeType type, int srcVertex, int destVertex, BGFLOAT deltaT)
279{
280 BGSIZE iEdg;
281 if (edgeCounts_[destVertex] >= maxEdgesPerVertex_) {
282 LOG4CPLUS_FATAL(edgeLogger_, "Vertex : " << destVertex << " ran out of space for new edges.");
283 throw runtime_error("Vertex " + to_string(destVertex)
284 + " ran out of space for new edges, current edge = "
285 + to_string(edgeCounts_[destVertex]));
286 }
287
288 // add it to the list: find first edge location for vertex destVertex
289 // that isn't in use.
290 for (BGSIZE i = 0; i < maxEdgesPerVertex_; i++) {
291 iEdg = maxEdgesPerVertex_ * destVertex + i;
292 if (!inUse_[iEdg]) {
293 break;
294 }
295 }
296
297 edgeCounts_[destVertex]++;
298
299 // create an edge
300 createEdge(iEdg, srcVertex, destVertex, deltaT, type);
301 return iEdg;
302}
