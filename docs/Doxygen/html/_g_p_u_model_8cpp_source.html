85 //initialize Mersenne Twister

90 int rng_mt_rng_count = Simulator::getInstance().getTotalVertices()
91 / rng_nPerRng; //# of threads to generate for numVertices rand #s
92 int rng_threads = rng_mt_rng_count / rng_blocks; //# threads per block needed
93 initMTGPU(Simulator::getInstance().getNoiseRngSeed(), rng_blocks, rng_threads, rng_nPerRng,
94 rng_mt_rng_count);
95

104#endif // PERFORMANCE_METRICS
105
106 // allocates memories on CUDA device
107 allocDeviceStruct((void **)&allVerticesDevice_, (void **)&allEdgesDevice_);

167

170 cudaStartTimer();

176#ifdef PERFORMANCE_METRICS

266void GPUModel::copySynapseIndexMapHostToDevice(EdgeIndexMap &synapseIndexMapHost, int numVertices)
267{
268 AllEdges *synapses = connections_->getEdges();
269 int totalSynapseCount = dynamic_cast<AllEdges *>(synapses)->totalEdgeCount_;
270 if (totalSynapseCount == 0)
271 return;
272 EdgeIndexMapDevice synapseIMapDevice;
273 HANDLE_ERROR(cudaMemcpy(&synapseIMapDevice, synapseIndexMapDevice_, sizeof(EdgeIndexMapDevice),
274 cudaMemcpyDeviceToHost));
275 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.outgoingEdgeBegin_,
276 synapseIndexMapHost.outgoingEdgeBegin_.data(),
277 numVertices * sizeof(BGSIZE), cudaMemcpyHostToDevice));
278 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.outgoingEdgeCount_,
279 synapseIndexMapHost.outgoingEdgeCount_.data(),
280 numVertices * sizeof(BGSIZE), cudaMemcpyHostToDevice));
281 if (synapseIMapDevice.outgoingEdgeIndexMap_ != nullptr) {
282 HANDLE_ERROR(cudaFree(synapseIMapDevice.outgoingEdgeIndexMap_));
283 }
284 HANDLE_ERROR(cudaMalloc((void **)&synapseIMapDevice.outgoingEdgeIndexMap_,
285 totalSynapseCount * sizeof(BGSIZE)));
286 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.outgoingEdgeIndexMap_,
287 synapseIndexMapHost.outgoingEdgeIndexMap_.data(),
288 totalSynapseCount * sizeof(BGSIZE), cudaMemcpyHostToDevice));
289 // active synapse map
290 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.incomingEdgeBegin_,
291 synapseIndexMapHost.incomingEdgeBegin_.data(),
292 numVertices * sizeof(BGSIZE), cudaMemcpyHostToDevice));
293 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.incomingEdgeCount_,
294 synapseIndexMapHost.incomingEdgeCount_.data(),
295 numVertices * sizeof(BGSIZE), cudaMemcpyHostToDevice));
296 // the number of synapses may change, so we reallocate the memory
297 if (synapseIMapDevice.incomingEdgeIndexMap_ != nullptr) {
298 HANDLE_ERROR(cudaFree(synapseIMapDevice.incomingEdgeIndexMap_));
299 synapseIMapDevice.incomingEdgeIndexMap_ = nullptr;
300 }
301 HANDLE_ERROR(cudaMalloc((void **)&synapseIMapDevice.incomingEdgeIndexMap_,
302 totalSynapseCount * sizeof(BGSIZE)));
303 HANDLE_ERROR(cudaMemcpy(synapseIMapDevice.incomingEdgeIndexMap_,
304 synapseIndexMapHost.incomingEdgeIndexMap_.data(),
305 totalSynapseCount * sizeof(BGSIZE), cudaMemcpyHostToDevice));
306 HANDLE_ERROR(cudaMemcpy(synapseIndexMapDevice_, &synapseIMapDevice, sizeof(EdgeIndexMapDevice),
307 cudaMemcpyHostToDevice));
308}

326__global__ void
327 calcSummationMapDevice(int totalVertices,
328 AllSpikingNeuronsDeviceProperties *__restrict__ allVerticesDevice,
329 const EdgeIndexMapDevice *__restrict__ synapseIndexMapDevice_,
330 const AllSpikingSynapsesDeviceProperties *__restrict__ allEdgesDevice)
331{
332 // The usual thread ID calculation and guard against excess threads
333 // (beyond the number of vertices, in this case).
334 int idx = blockIdx.x * blockDim.x + threadIdx.x;
335 if (idx >= totalVertices)
336 return;
337
338 // Number of incoming synapses
339 const BGSIZE synCount = synapseIndexMapDevice_->incomingEdgeCount_[idx];
340 // Optimization: terminate thread if no incoming synapses
341 if (synCount != 0) {
342 // Index of start of this neuron's block of forward map entries
343 const int beginIndex = synapseIndexMapDevice_->incomingEdgeBegin_[idx];
344 // Address of the start of this neuron's block of forward map entries
345 const BGSIZE *activeMapBegin = &(synapseIndexMapDevice_->incomingEdgeIndexMap_[beginIndex]);
346 // Summed post-synaptic response (PSR)
347 BGFLOAT sum = 0.0;
348 // Index of the current incoming synapse
349 BGSIZE synIndex;
350 // Repeat for each incoming synapse
351 for (BGSIZE i = 0; i < synCount; i++) {
352 // Get index of current incoming synapse
353 synIndex = activeMapBegin[i];
354 // Fetch its PSR and add into sum
355 sum += allEdgesDevice->psr_[synIndex];
356 }
357 // Store summed PSR into this neuron's summation point
358 allVerticesDevice->summationMap_[idx] = sum;
359 }
360}
