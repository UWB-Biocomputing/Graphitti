134void AllIFNeurons::createNeuron(int i, Layout *layout)
135{
136 // set the neuron info for neurons
137 Iinject_[i] = initRNG.inRange(IinjectRange_[0], IinjectRange_[1]);
138 Inoise_[i] = initRNG.inRange(InoiseRange_[0], InoiseRange_[1]);
139 Vthresh_[i] = initRNG.inRange(VthreshRange_[0], VthreshRange_[1]);
140 Vrest_[i] = initRNG.inRange(VrestingRange_[0], VrestingRange_[1]);
141 Vreset_[i] = initRNG.inRange(VresetRange_[0], VresetRange_[1]);
142 Vinit_[i] = initRNG.inRange(VinitRange_[0], VinitRange_[1]);
143 Vm_[i] = Vinit_[i];
144
145 initNeuronConstsFromParamValues(i, Simulator::getInstance().getDeltaT());
146
147 switch (layout->vertexTypeMap_[i]) {
148 case INH:
149 LOG4CPLUS_DEBUG(vertexLogger_, "Setting inhibitory neuron: " << i);
150 // set inhibitory absolute refractory period
151 Trefract_[i] = DEFAULT_InhibTrefract; // TODO(derek): move defaults inside model.
152 break;
153
154 case EXC:
155 LOG4CPLUS_DEBUG(vertexLogger_, "Setting excitatory neuron: " << i);
156 // set excitatory absolute refractory period
157 Trefract_[i] = DEFAULT_ExcitTrefract;
158 break;
159
160 default:
161 LOG4CPLUS_DEBUG(vertexLogger_,
162 "ERROR: unknown neuron type: " << layout->vertexTypeMap_[i] << "@" << i);
163 assert(false);
164 break;
165 }
166 // endogenously_active_neuron_map -> Model State
167 if (layout->starterMap_[i]) {
168 // set endogenously active threshold voltage, reset voltage, and refractory period
169 Vthresh_[i] = initRNG.inRange(starterVthreshRange_[0], starterVthreshRange_[1]);
170 Vreset_[i] = initRNG.inRange(starterVresetRange_[0], starterVresetRange_[1]);
171 Trefract_[i] = DEFAULT_ExcitTrefract; // TODO(derek): move defaults inside model.
172 }
173
174 LOG4CPLUS_DEBUG(vertexLogger_, "\nCREATE NEURON[" << i << "] {" << endl
175 << "\tVm = " << Vm_[i] << endl
176 << "\tVthresh = " << Vthresh_[i] << endl
177 << "\tI0 = " << I0_[i] << endl
178 << "\tInoise = " << Inoise_[i] << " from : ("
179 << InoiseRange_[0] << "," << InoiseRange_[1]
180 << ")" << endl
181 << "\tC1 = " << C1_[i] << endl
182 << "\tC2 = " << C2_[i] << endl
183 << "}" << endl);
184}

207void AllIFNeurons::initNeuronConstsFromParamValues(int i, const BGFLOAT deltaT)
208{
209 BGFLOAT &Tau = this->Tau_[i];
210 BGFLOAT &C1 = this->C1_[i];
211 BGFLOAT &C2 = this->C2_[i];
212 BGFLOAT &Rm = this->Rm_[i];
213 BGFLOAT &I0 = this->I0_[i];
214 BGFLOAT &Iinject = this->Iinject_[i];
215 BGFLOAT &Vrest = this->Vrest_[i];
216
217 /* init consts C1,C2 for exponential Euler integration */
218 if (Tau > 0) {
219 C1 = exp(-deltaT / Tau);
220 C2 = Rm * (1 - C1);
221 } else {
222 C1 = 0.0;
223 C2 = Rm;
224 }
225 /* calculate const IO */
226 if (Rm > 0) {
227 I0 = Iinject + Vrest / Rm;
228 } else {
229 assert(false);
230 }
231}

250 ss << "nStepsInRefr: " << numStepsInRefractoryPeriod_[index]
