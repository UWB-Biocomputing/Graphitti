179 void AllIFNeurons::createNeuron(int neuronIndex, Layout *layout) {
180  // set the neuron info for neurons
181  Iinject_[neuronIndex] = rng.inRange(IinjectRange_[0], IinjectRange_[1]);
182  Inoise_[neuronIndex] = rng.inRange(InoiseRange_[0], InoiseRange_[1]);
183  Vthresh_[neuronIndex] = rng.inRange(VthreshRange_[0], VthreshRange_[1]);
184  Vrest_[neuronIndex] = rng.inRange(VrestingRange_[0], VrestingRange_[1]);
185  Vreset_[neuronIndex] = rng.inRange(VresetRange_[0], VresetRange_[1]);
186  Vinit_[neuronIndex] = rng.inRange(VinitRange_[0], VinitRange_[1]);
187  Vm_[neuronIndex] = Vinit_[neuronIndex];
188 
189  initNeuronConstsFromParamValues(neuronIndex, Simulator::getInstance().getDeltaT());
190 
191  int maxSpikes = (int) ((Simulator::getInstance().getEpochDuration() * Simulator::getInstance().getMaxFiringRate()));
192  spikeHistory_[neuronIndex] = new uint64_t[maxSpikes];
193  for (int j = 0; j < maxSpikes; ++j) {
194  spikeHistory_[neuronIndex][j] = ULONG_MAX;
195  }
196 
197  switch (layout->vertexTypeMap_[neuronIndex]) {
198  case INH:
199  LOG4CPLUS_DEBUG(vertexLogger_, "Setting inhibitory neuron: " << neuronIndex);
200  // set inhibitory absolute refractory period
201  Trefract_[neuronIndex] = DEFAULT_InhibTrefract;// TODO(derek): move defaults inside model.
202  break;
203 
204  case EXC:
205  LOG4CPLUS_DEBUG(vertexLogger_, "Setting excitatory neuron: " << neuronIndex);
206  // set excitatory absolute refractory period
207  Trefract_[neuronIndex] = DEFAULT_ExcitTrefract;
208  break;
209 
210  default:
211  LOG4CPLUS_DEBUG(vertexLogger_, "ERROR: unknown neuron type: "
212  << layout->vertexTypeMap_[neuronIndex] << "@" << neuronIndex);
213  assert(false);
214  break;
215  }
216  // endogenously_active_neuron_map -> Model State
217  if (layout->starterMap_[neuronIndex]) {
218  // set endogenously active threshold voltage, reset voltage, and refractory period
219  Vthresh_[neuronIndex] = rng.inRange(starterVthreshRange_[0], starterVthreshRange_[1]);
220  Vreset_[neuronIndex] = rng.inRange(starterVresetRange_[0], starterVresetRange_[1]);
221  Trefract_[neuronIndex] = DEFAULT_ExcitTrefract; // TODO(derek): move defaults inside model.
222  }
223 
224  LOG4CPLUS_DEBUG(vertexLogger_, "\nCREATE NEURON[" << neuronIndex << "] {" << endl
225  << "\tVm = " << Vm_[neuronIndex] << endl
226  << "\tVthresh = " << Vthresh_[neuronIndex] << endl
227  << "\tI0 = " << I0_[neuronIndex] << endl
228  << "\tInoise = " << Inoise_[neuronIndex] << " from : (" << InoiseRange_[0] << "," << InoiseRange_[1]
229  << ")"
230  << endl
231  << "\tC1 = " << C1_[neuronIndex] << endl
232  << "\tC2 = " << C2_[neuronIndex] << endl
233  << "}" << endl);
234 }

256 void AllIFNeurons::initNeuronConstsFromParamValues(int neuronIndex, const BGFLOAT deltaT) {
257  BGFLOAT &Tau = this->Tau_[neuronIndex];
258  BGFLOAT &C1 = this->C1_[neuronIndex];
259  BGFLOAT &C2 = this->C2_[neuronIndex];
260  BGFLOAT &Rm = this->Rm_[neuronIndex];
261  BGFLOAT &I0 = this->I0_[neuronIndex];
262  BGFLOAT &Iinject = this->Iinject_[neuronIndex];
263  BGFLOAT &Vrest = this->Vrest_[neuronIndex];
264 
265  /* init consts C1,C2 for exponential Euler integration */
266  if (Tau > 0) {
267  C1 = exp(-deltaT / Tau);
268  C2 = Rm * (1 - C1);
269  } else {
270  C1 = 0.0;
271  C2 = Rm;
272  }
273  /* calculate const IO */
274  if (Rm > 0) {
275  I0 = Iinject + Vrest / Rm;
276  } else {
277  assert(false);
278  }
279 }
