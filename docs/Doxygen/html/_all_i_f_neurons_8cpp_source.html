117void AllIFNeurons::createNeuron(int i, Layout &layout)
118{
119 // set the neuron info for neurons
120 Iinject_[i] = initRNG.inRange(IinjectRange_[0], IinjectRange_[1]);
121 Inoise_[i] = initRNG.inRange(InoiseRange_[0], InoiseRange_[1]);
122 Vthresh_[i] = initRNG.inRange(VthreshRange_[0], VthreshRange_[1]);
123 Vrest_[i] = initRNG.inRange(VrestingRange_[0], VrestingRange_[1]);
124 Vreset_[i] = initRNG.inRange(VresetRange_[0], VresetRange_[1]);
125 Vinit_[i] = initRNG.inRange(VinitRange_[0], VinitRange_[1]);
126 Vm_[i] = Vinit_[i];
127
128 initNeuronConstsFromParamValues(i, Simulator::getInstance().getDeltaT());
129
130 switch (layout.vertexTypeMap_[i]) {
131 case vertexType::INH:
132 LOG4CPLUS_DEBUG(vertexLogger_, "Setting inhibitory neuron: " << i);
133 // set inhibitory absolute refractory period
134 Trefract_[i] = DEFAULT_InhibTrefract; // TODO(derek): move defaults inside model.
135 break;
136
137 case vertexType::EXC:
138 LOG4CPLUS_DEBUG(vertexLogger_, "Setting excitatory neuron: " << i);
139 // set excitatory absolute refractory period
140 Trefract_[i] = DEFAULT_ExcitTrefract;
141 break;
142
143 default:
144 LOG4CPLUS_DEBUG(vertexLogger_,
145 "ERROR: unknown neuron type: " << layout.vertexTypeMap_[i] << "@" << i);
146 assert(false);
147 break;
148 }
149 // endogenously_active_neuron_map -> Model State
150 if (layout.starterMap_[i]) {
151 // set endogenously active threshold voltage, reset voltage, and refractory period
152 Vthresh_[i] = initRNG.inRange(starterVthreshRange_[0], starterVthreshRange_[1]);
153 Vreset_[i] = initRNG.inRange(starterVresetRange_[0], starterVresetRange_[1]);
154 Trefract_[i] = DEFAULT_ExcitTrefract; // TODO(derek): move defaults inside model.
155 }
156
157 LOG4CPLUS_DEBUG(vertexLogger_, "\nCREATE NEURON[" << i << "] {" << endl
158 << "\tVm = " << Vm_[i] << endl
159 << "\tVthresh = " << Vthresh_[i] << endl
160 << "\tI0 = " << I0_[i] << endl
161 << "\tInoise = " << Inoise_[i] << " from : ("
162 << InoiseRange_[0] << "," << InoiseRange_[1]
163 << ")" << endl
164 << "\tC1 = " << C1_[i] << endl
165 << "\tC2 = " << C2_[i] << endl
166 << "}" << endl);
167}

190void AllIFNeurons::initNeuronConstsFromParamValues(int i, BGFLOAT deltaT)
191{
192 BGFLOAT &Tau = this->Tau_[i];
193 BGFLOAT &C1 = this->C1_[i];
194 BGFLOAT &C2 = this->C2_[i];
195 BGFLOAT &Rm = this->Rm_[i];
196 BGFLOAT &I0 = this->I0_[i];
197 BGFLOAT &Iinject = this->Iinject_[i];
198 BGFLOAT &Vrest = this->Vrest_[i];
199
200 /* init consts C1,C2 for exponential Euler integration */
201 if (Tau > 0) {
202 C1 = exp(-deltaT / Tau);
203 C2 = Rm * (1 - C1);
204 } else {
205 C1 = 0.0;
206 C2 = Rm;
207 }
208 /* calculate const IO */
209 if (Rm > 0) {
210 I0 = Iinject + Vrest / Rm;
211 } else {
212 assert(false);
213 }
214}
