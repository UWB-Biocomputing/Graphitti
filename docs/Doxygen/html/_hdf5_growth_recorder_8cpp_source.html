137void Hdf5GrowthRecorder::writeRadiiRates()
138{
139 try {
140 // Write radii and rates histories information:
141 hsize_t offset[2], count[2];
142 hsize_t dimsm[2];
143 DataSpace *dataspace;
144 DataSpace *memspace;
145
146 // write radii history
147 offset[0] = Simulator::getInstance().getCurrentStep();
148 offset[1] = 0;
149 count[0] = 1;
150 count[1] = Simulator::getInstance().getTotalVertices();
151 dimsm[0] = 1;
152 dimsm[1] = Simulator::getInstance().getTotalVertices();
153 memspace = new DataSpace(2, dimsm, nullptr);
154 dataspace = new DataSpace(dataSetRadiiHist_->getSpace());
155 dataspace->selectHyperslab(H5S_SELECT_SET, count, offset);
156 dataSetRadiiHist_->write(radiiHistory_, H5_FLOAT, *memspace, *dataspace);
157 delete dataspace;
158 delete memspace;
159
160 // write rates history
161 offset[0] = Simulator::getInstance().getCurrentStep();
162 offset[1] = 0;
163 count[0] = 1;
164 count[1] = Simulator::getInstance().getTotalVertices();
165 dimsm[0] = 1;
166 dimsm[1] = Simulator::getInstance().getTotalVertices();
167 memspace = new DataSpace(2, dimsm, nullptr);
168 dataspace = new DataSpace(dataSetRadiiHist_->getSpace());
169 dataspace->selectHyperslab(H5S_SELECT_SET, count, offset);
170 dataSetRatesHist_->write(ratesHistory_, H5_FLOAT, *memspace, *dataspace);
171 delete dataspace;
172 delete memspace;
173 }
