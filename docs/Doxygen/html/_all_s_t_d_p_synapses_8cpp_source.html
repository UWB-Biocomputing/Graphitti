246void AllSTDPSynapses::advanceEdge(BGSIZE iEdg, AllVertices &neurons)
247{
248 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
249 //LOG4CPLUS_DEBUG(edgeLogger_, "iEdg : " << iEdg );
250
251 BGFLOAT &W = W_[iEdg];
252 /*
253 if (W <= 0.0) {
254 AllSpikingSynapses::advanceEdge(iEdg, neurons);
255 return;
256 }
257 */
258
259 BGFLOAT &decay = decay_[iEdg];
260 BGFLOAT &psr = psr_[iEdg];
261 int sumPointIndex = destVertexIndex_[iEdg];
262
263 // is an input in the queue?
264 bool fPre = isSpikeQueue(iEdg);
265 bool fPost = isSpikeQueuePost(iEdg);
266
267 if (fPre || fPost) {
268 const BGFLOAT taupos = taupos_[iEdg];
269 const BGFLOAT tauneg = tauneg_[iEdg];
270 const int total_delay = totalDelay_[iEdg];
271
272 BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
273 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(neurons);
274
275 // pre and post neurons index
276 int idxPre = sourceVertexIndex_[iEdg];
277 int idxPost = destVertexIndex_[iEdg];
278 uint64_t t_post, t_pre;
279 BGFLOAT delta;
280
281 // epre and epost are set to 1 for the independent model [Froemke and Dan (2002)]
282 BGFLOAT epre = 1.0;
283 BGFLOAT epost = 1.0;
284
285 if (fPre) { // preSpikeHit
286 // call the learning function stdpLearning() for each pair of pre-post spikes
287 int offIndex = -1; // last spike
288 while (true) {
289 // time of the offset post-spike hit
290 t_post = spNeurons.getSpikeHistory(idxPost, offIndex);
291
292 // break if no post-spike is found
293 if (t_post == numeric_limits<unsigned long>::max())
294 break;
295
296 // delta is the spike interval between pre-post spikes
297 // deltaT converts the timestep to seconds
298 delta = -static_cast<BGFLOAT>(g_simulationStep - t_post) * deltaT;
299
300 /*
301 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPre" << endl
302 << "\tiEdg: " << iEdg << endl
303 << "\tidxPre: " << idxPre << endl
304 << "\tidxPost: " << idxPost << endl
305 << "\tspikeHistory: " << spikeHistory << endl
306 << "\tepre: " << epre << endl
307 << "\tepost: " << epost << endl
308 << "\tg_simulationStep: " << g_simulationStep << endl
309 << "\tdelta: " << delta << endl << endl);
310 */
311
312 if (delta <= -3.0 * tauneg)
313 break;
314
315 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
316 --offIndex;
317 }
318
319 changePSR(iEdg, deltaT);
320 }
321
322 if (fPost) { // postSpikeHit
323 // call the learning function stdpLearning() for each pair of post-pre spikes
324 int offIndex = -1; // last spike
325 while (true) {
326 // time when the pre-spike left the source neuron
327 t_pre = spNeurons.getSpikeHistory(idxPre, offIndex);
328
329 // break if no pre-spike is found
330 if (t_pre == numeric_limits<unsigned long>::max())
331 break;
332
333 // delay accounts for the time it takes the spike to reach the destination neuron
334 if (t_pre + total_delay > g_simulationStep) {
335 --offIndex;
336 continue;
337 }
338
339 // delta is the spike interval between post-pre spikes
340 delta = static_cast<BGFLOAT>(g_simulationStep - t_pre - total_delay) * deltaT;
341
342 /*
343 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPost" << endl
344 << "\tiEdg: " << iEdg << endl
345 << "\tidxPre: " << idxPre << endl
346 << "\tidxPost: " << idxPost << endl
347 << "\tspikeHistory: " << spikeHistory << endl
348 << "\tg_simulationStep: " << g_simulationStep << endl
349 << "\tepre: " << epre << endl
350 << "\tepost: " << epost << endl
351 << "\tdelta: " << delta << endl << endl);
352 */
353
354 if (delta >= 3.0 * taupos)
355 break;
356
357 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
358 --offIndex;
359 }
360 }
361 }
362
363 // decay the post spike response
364 psr *= decay;
365}

367BGFLOAT AllSTDPSynapses::synapticWeightModification(BGSIZE iEdg, BGFLOAT synapticWeight,
368 double delta)
369{
370 BGFLOAT STDPgap = STDPgap_[iEdg];
371 BGFLOAT muneg = muneg_[iEdg];
372 BGFLOAT mupos = mupos_[iEdg];
373 BGFLOAT tauneg = tauneg_[iEdg];
374 BGFLOAT taupos = taupos_[iEdg];
375 BGFLOAT Aneg = Aneg_[iEdg];
376 BGFLOAT Apos = Apos_[iEdg];
377 BGFLOAT Wex = Wex_[iEdg];
378 BGFLOAT &W = W_[iEdg];
379 BGFLOAT dw = 0;
380
381 // muneg and mupos default to 0 for the basic multiplicative model
382 if (delta < -STDPgap) {
383 // depression
384 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
385 } else if (delta > STDPgap) {
386 // potentiation
387 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
388 }
389
390 return dw;
391}

403void AllSTDPSynapses::stdpLearning(BGSIZE iEdg, double delta, double epost, double epre,
404 int srcVertex, int destVertex)
405{
406 BGFLOAT STDPgap = STDPgap_[iEdg];
407 BGFLOAT Wex = Wex_[iEdg];
408 BGFLOAT &W = W_[iEdg];
409 edgeType type = type_[iEdg];
410 BGFLOAT oldW = W;
411
412 if (fabs(delta) <= STDPgap) {
413 return;
414 }
415
416 // dw is the fractional change in synaptic strength; add 1.0 to become the scaling ratio
417 // dw = 1.0 + dw * epre * epost; // used for the non-independent model
418 BGFLOAT dw = 1.0 + synapticWeightModification(iEdg, W, delta);
419
420 // if scaling ratio is less than zero, set it to zero so this synapse, its
421 // strength is always zero
422 // TODO: Where is the code for this?
423
424 // current weight multiplies dw (scaling ratio) to generate new weight
425 if (dw != 0.0)
426 W *= dw;
427
428 // if new weight is bigger than Wex (maximum allowed weight), then set it to Wex
429 if (fabs(W) > Wex) {
430 W = edgSign(type) * Wex;
431 }

449bool AllSTDPSynapses::isSpikeQueuePost(BGSIZE iEdg)
450{
451 uint32_t &delayQueue = delayQueuePost_[iEdg];
452 int &delayIdx = delayIndexPost_[iEdg];
453 int &ldelayQueue = delayQueuePostLength_[iEdg];
454
455 bool r = delayQueue & (0x1 << delayIdx);
456 delayQueue &= ~(0x1 << delayIdx);
457 if (++delayIdx >= ldelayQueue) {
458 delayIdx = 0;
459 }
460 return r;
461}

466void AllSTDPSynapses::postSpikeHit(BGSIZE iEdg)
467{
468 uint32_t &delay_queue = delayQueuePost_[iEdg];
469 int &delayIdx = delayIndexPost_[iEdg];
470 int &ldelayQueue = delayQueuePostLength_[iEdg];
471 int &total_delay = totalDelayPost_[iEdg];
472
473 // Add to spike queue
474
475 // calculate index where to insert the spike into delayQueue
476 int idx = delayIdx + total_delay;
477 if (idx >= ldelayQueue) {
478 idx -= ldelayQueue;
479 }
480
481 // set a spike
482 assert(!(delay_queue & (0x1 << idx)));
483 delay_queue |= (0x1 << idx);
484}
