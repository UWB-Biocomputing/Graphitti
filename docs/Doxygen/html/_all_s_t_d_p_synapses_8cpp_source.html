289void AllSTDPSynapses::advanceEdge(const BGSIZE iEdg, AllVertices *neurons)
290{
291 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
292 //LOG4CPLUS_DEBUG(edgeLogger_, "iEdg : " << iEdg );
293
294 BGFLOAT &W = W_[iEdg];
295 /*
296 if (W <= 0.0) {
297 AllSpikingSynapses::advanceEdge(iEdg, neurons);
298 return;
299 }
300 */
301
302 BGFLOAT &decay = decay_[iEdg];
303 BGFLOAT &psr = psr_[iEdg];
304 BGFLOAT &summationPoint = *(summationPoint_[iEdg]);
305
306 // is an input in the queue?
307 bool fPre = isSpikeQueue(iEdg);
308 bool fPost = isSpikeQueuePost(iEdg);
309
310 if (fPre || fPost) {
311 const BGFLOAT taupos = taupos_[iEdg];
312 const BGFLOAT tauneg = tauneg_[iEdg];
313 const int total_delay = totalDelay_[iEdg];
314
315 BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
316 AllSpikingNeurons *spNeurons = dynamic_cast<AllSpikingNeurons *>(neurons);
317
318 // pre and post neurons index
319 int idxPre = sourceVertexIndex_[iEdg];
320 int idxPost = destVertexIndex_[iEdg];
321 uint64_t spikeHistory, spikeHistory2;
322 BGFLOAT delta;
323 BGFLOAT epre, epost;
324
325 if (fPre) { // preSpikeHit
326 // spikeCount points to the next available position of spike_history,
327 // so the getSpikeHistory w/offset = -2 will return the spike time
328 // just one before the last spike.
329 spikeHistory = spNeurons->getSpikeHistory(idxPre, -2);
330
331 epre = 1.0;
332 epost = 1.0;
333 // call the learning function stdpLearning() for each pair of
334 // pre-post spikes
335 int offIndex = -1; // last spike
336 while (true) {
337 spikeHistory = spNeurons->getSpikeHistory(idxPost, offIndex);
338 if (spikeHistory == numeric_limits<unsigned long>::max())
339 break;
340 // delta is the spike interval between pre-post spikes
341 // (include pre-synaptic transmission delay)
342 delta = -static_cast<BGFLOAT>(g_simulationStep - spikeHistory) * deltaT;
343 /*
344 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPre" << endl
345 << "\tiEdg: " << iEdg << endl
346 << "\tidxPre: " << idxPre << endl
347 << "\tidxPost: " << idxPost << endl
348 << "\tspikeHistory: " << spikeHistory << endl
349 << "\tepre: " << epre << endl
350 << "\tepost: " << epost << endl
351 << "\tg_simulationStep: " << g_simulationStep << endl
352 << "\tdelta: " << delta << endl << endl);
353 */
354
355 if (delta <= -3.0 * tauneg)
356 break;
357
358 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
359 --offIndex;
360 }
361
362 changePSR(iEdg, deltaT);
363 }
364
365 if (fPost) { // postSpikeHit
366 // spikeCount points to the next available position of spike_history,
367 // so the getSpikeHistory w/offset = -2 will return the spike time
368 // just one before the last spike.
369 spikeHistory = spNeurons->getSpikeHistory(idxPost, -2);
370 epost = 1.0;
371 epre = 1;
372
373
374 // call the learning function stdpLearning() for each pair of
375 // post-pre spikes
376 int offIndex = -1; // last spike
377 while (true) {
378 spikeHistory = spNeurons->getSpikeHistory(idxPre, offIndex);
379 if (spikeHistory == numeric_limits<unsigned long>::max())
380 break;
381
382 if (spikeHistory + total_delay > g_simulationStep) {
383 --offIndex;
384 continue;
385 }
386 // delta is the spike interval between post-pre spikes
387 delta = static_cast<BGFLOAT>(g_simulationStep - spikeHistory - total_delay) * deltaT;
388 /*
389 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPost" << endl
390 << "\tiEdg: " << iEdg << endl
391 << "\tidxPre: " << idxPre << endl
392 << "\tidxPost: " << idxPost << endl
393 << "\tspikeHistory: " << spikeHistory << endl
394 << "\tg_simulationStep: " << g_simulationStep << endl
395 << "\tepre: " << epre << endl
396 << "\tepost: " << epost << endl
397 << "\tdelta: " << delta << endl << endl);
398 */
399 if (delta >= 3.0 * taupos)
400 break;
401
402 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
403 --offIndex;
404 }
405 }
406 }
407
408 // decay the post spike response
409 psr *= decay;
410 // and apply it to the summation point
411 #ifdef USE_OMP
412 #pragma omp atomic
413 #endif
414 summationPoint += psr;
415 #ifdef USE_OMP
416 //PAB: atomic above has implied flush (following statement generates error -- can't be member variable)
417 //#pragma omp flush (summationPoint)
418 #endif
419}

422BGFLOAT AllSTDPSynapses::synapticWeightModification(const BGSIZE iEdg, BGFLOAT synapticWeight,
423 double delta)
424{
425 BGFLOAT STDPgap = STDPgap_[iEdg];
426 BGFLOAT muneg = muneg_[iEdg];
427 BGFLOAT mupos = mupos_[iEdg];
428 BGFLOAT tauneg = tauneg_[iEdg];
429 BGFLOAT taupos = taupos_[iEdg];
430 BGFLOAT Aneg = Aneg_[iEdg];
431 BGFLOAT Apos = Apos_[iEdg];
432 BGFLOAT Wex = Wex_[iEdg];
433 BGFLOAT &W = W_[iEdg];
434 edgeType type = type_[iEdg];
435 BGFLOAT dw = 0;
436 BGFLOAT oldW = W;
437
438 // BGFLOAT modDelta = fabs(delta);
439
440 if (delta < -STDPgap) {
441 // depression
442
443 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
444 } else if (delta > STDPgap) {
445 // potentiation
446 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
447 }
448
449 return dw;
450}

462void AllSTDPSynapses::stdpLearning(const BGSIZE iEdg, double delta, double epost, double epre,
463 int srcVertex, int destVertex)
464{
465 BGFLOAT STDPgap = STDPgap_[iEdg];
466 BGFLOAT muneg = muneg_[iEdg];
467 BGFLOAT mupos = mupos_[iEdg];
468 BGFLOAT tauneg = tauneg_[iEdg];
469 BGFLOAT taupos = taupos_[iEdg];
470 BGFLOAT Aneg = Aneg_[iEdg];
471 BGFLOAT Apos = Apos_[iEdg];
472 BGFLOAT Wex = Wex_[iEdg];
473 BGFLOAT &W = W_[iEdg];
474 edgeType type = type_[iEdg];
475 BGFLOAT oldW = W;
476 // BGFLOAT modDelta = fabs(delta);
477
478 if (delta <= fabs(STDPgap)) {
479 return;
480 }
481
482 // dw is the fractional change in synaptic strength; add 1.0 to become the scaling ratio
483 //dw = 1.0 + dw * epre * epost;
484 BGFLOAT dw = 1.0 + synapticWeightModification(iEdg, W, delta);
485
486 // if scaling ratio is less than zero, set it to zero so this synapse, its
487 // strength is always zero
488 // TODO: Where is the code for this?
489
490 // current weight multiplies dw (scaling ratio) to generate new weight
491 if (dw != 0.0)
492 W *= dw;
493
494 // if new weight is bigger than Wex (maximum allowed weight), then set it to Wex
495 if (fabs(W) > Wex) {
496 W = edgSign(type) * Wex;
497 }
498 /*

515bool AllSTDPSynapses::isSpikeQueuePost(const BGSIZE iEdg)
516{
517 uint32_t &delayQueue = delayQueuePost_[iEdg];
518 int &delayIdx = delayIndexPost_[iEdg];
519 int &ldelayQueue = delayQueuePostLength_[iEdg];
520
521 bool r = delayQueue & (0x1 << delayIdx);
522 delayQueue &= ~(0x1 << delayIdx);
523 if (++delayIdx >= ldelayQueue) {
524 delayIdx = 0;
525 }
526 return r;
527}

532void AllSTDPSynapses::postSpikeHit(const BGSIZE iEdg)
533{
534 uint32_t &delay_queue = delayQueuePost_[iEdg];
535 int &delayIdx = delayIndexPost_[iEdg];
536 int &ldelayQueue = delayQueuePostLength_[iEdg];
537 int &total_delay = totalDelayPost_[iEdg];
538
539 // Add to spike queue
540
541 // calculate index where to insert the spike into delayQueue
542 int idx = delayIdx + total_delay;
543 if (idx >= ldelayQueue) {
544 idx -= ldelayQueue;
545 }
546
547 // set a spike
548 assert(!(delay_queue & (0x1 << idx)));
549 delay_queue |= (0x1 << idx);
550}
