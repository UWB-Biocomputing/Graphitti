261void AllSTDPSynapses::advanceEdge(BGSIZE iEdg, AllVertices &neurons)
262{
263 // If the synapse is inhibitory or its weight is zero, update synapse state using AllSpikingSynapses::advanceEdge method
264 //LOG4CPLUS_DEBUG(edgeLogger_, "iEdg : " << iEdg );
265
266 BGFLOAT &W = W_[iEdg];
267 /*
268 if (W <= 0.0) {
269 AllSpikingSynapses::advanceEdge(iEdg, neurons);
270 return;
271 }
272 */
273
274 BGFLOAT &decay = decay_[iEdg];
275 BGFLOAT &psr = psr_[iEdg];
276 int sumPointIndex = destVertexIndex_[iEdg];
277
278 // is an input in the queue?
279 bool fPre = isSpikeQueue(iEdg);
280 bool fPost = isSpikeQueuePost(iEdg);
281
282 if (fPre || fPost) {
283 const BGFLOAT taupos = taupos_[iEdg];
284 const BGFLOAT tauneg = tauneg_[iEdg];
285 const int total_delay = totalDelay_[iEdg];
286
287 BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
288 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(neurons);
289
290 // pre and post neurons index
291 int idxPre = sourceVertexIndex_[iEdg];
292 int idxPost = destVertexIndex_[iEdg];
293 uint64_t spikeHistory, spikeHistory2;
294 BGFLOAT delta;
295 BGFLOAT epre, epost;
296
297 if (fPre) { // preSpikeHit
298 // spikeCount points to the next available position of spike_history,
299 // so the getSpikeHistory w/offset = -2 will return the spike time
300 // just one before the last spike.
301 spikeHistory = spNeurons.getSpikeHistory(idxPre, -2);
302
303 epre = 1.0;
304 epost = 1.0;
305 // call the learning function stdpLearning() for each pair of
306 // pre-post spikes
307 int offIndex = -1; // last spike
308 while (true) {
309 spikeHistory = spNeurons.getSpikeHistory(idxPost, offIndex);
310 if (spikeHistory == numeric_limits<unsigned long>::max())
311 break;
312 // delta is the spike interval between pre-post spikes
313 // (include pre-synaptic transmission delay)
314 delta = -static_cast<BGFLOAT>(g_simulationStep - spikeHistory) * deltaT;
315 /*
316 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPre" << endl
317 << "\tiEdg: " << iEdg << endl
318 << "\tidxPre: " << idxPre << endl
319 << "\tidxPost: " << idxPost << endl
320 << "\tspikeHistory: " << spikeHistory << endl
321 << "\tepre: " << epre << endl
322 << "\tepost: " << epost << endl
323 << "\tg_simulationStep: " << g_simulationStep << endl
324 << "\tdelta: " << delta << endl << endl);
325 */
326
327 if (delta <= -3.0 * tauneg)
328 break;
329
330 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
331 --offIndex;
332 }
333
334 changePSR(iEdg, deltaT);
335 }
336
337 if (fPost) { // postSpikeHit
338 // spikeCount points to the next available position of spike_history,
339 // so the getSpikeHistory w/offset = -2 will return the spike time
340 // just one before the last spike.
341 spikeHistory = spNeurons.getSpikeHistory(idxPost, -2);
342 epost = 1.0;
343 epre = 1;
344
345
346 // call the learning function stdpLearning() for each pair of
347 // post-pre spikes
348 int offIndex = -1; // last spike
349 while (true) {
350 spikeHistory = spNeurons.getSpikeHistory(idxPre, offIndex);
351 if (spikeHistory == numeric_limits<unsigned long>::max())
352 break;
353
354 if (spikeHistory + total_delay > g_simulationStep) {
355 --offIndex;
356 continue;
357 }
358 // delta is the spike interval between post-pre spikes
359 delta = static_cast<BGFLOAT>(g_simulationStep - spikeHistory - total_delay) * deltaT;
360 /*
361 LOG4CPLUS_DEBUG(fileLogger_,"\nAllSTDPSynapses::advanceSynapse: fPost" << endl
362 << "\tiEdg: " << iEdg << endl
363 << "\tidxPre: " << idxPre << endl
364 << "\tidxPost: " << idxPost << endl
365 << "\tspikeHistory: " << spikeHistory << endl
366 << "\tg_simulationStep: " << g_simulationStep << endl
367 << "\tepre: " << epre << endl
368 << "\tepost: " << epost << endl
369 << "\tdelta: " << delta << endl << endl);
370 */
371 if (delta >= 3.0 * taupos)
372 break;
373
374 stdpLearning(iEdg, delta, epost, epre, idxPre, idxPost);
375 --offIndex;
376 }
377 }
378 }
379
380 // decay the post spike response
381 psr *= decay;
382 // and apply it to the summation point
383 #ifdef USE_OMP
384 #pragma omp atomic
385 #endif
386 neurons.summationPoints_[sumPointIndex] += psr;
387 #ifdef USE_OMP
388 //PAB: atomic above has implied flush (following statement generates error -- can't be member variable)
389 //#pragma omp flush (summationPoint)
390 #endif
391}

394BGFLOAT AllSTDPSynapses::synapticWeightModification(BGSIZE iEdg, BGFLOAT synapticWeight,
395 double delta)
396{
397 BGFLOAT STDPgap = STDPgap_[iEdg];
398 BGFLOAT muneg = muneg_[iEdg];
399 BGFLOAT mupos = mupos_[iEdg];
400 BGFLOAT tauneg = tauneg_[iEdg];
401 BGFLOAT taupos = taupos_[iEdg];
402 BGFLOAT Aneg = Aneg_[iEdg];
403 BGFLOAT Apos = Apos_[iEdg];
404 BGFLOAT Wex = Wex_[iEdg];
405 BGFLOAT &W = W_[iEdg];
406 edgeType type = type_[iEdg];
407 BGFLOAT dw = 0;
408 BGFLOAT oldW = W;
409
410 // BGFLOAT modDelta = fabs(delta);
411
412 if (delta < -STDPgap) {
413 // depression
414
415 dw = pow(fabs(W) / Wex, muneg) * Aneg * exp(delta / tauneg); // normalize
416 } else if (delta > STDPgap) {
417 // potentiation
418 dw = pow(fabs(Wex - fabs(W)) / Wex, mupos) * Apos * exp(-delta / taupos); // normalize
419 }
420
421 return dw;
422}

434void AllSTDPSynapses::stdpLearning(BGSIZE iEdg, double delta, double epost, double epre,
435 int srcVertex, int destVertex)
436{
437 BGFLOAT STDPgap = STDPgap_[iEdg];
438 BGFLOAT muneg = muneg_[iEdg];
439 BGFLOAT mupos = mupos_[iEdg];
440 BGFLOAT tauneg = tauneg_[iEdg];
441 BGFLOAT taupos = taupos_[iEdg];
442 BGFLOAT Aneg = Aneg_[iEdg];
443 BGFLOAT Apos = Apos_[iEdg];
444 BGFLOAT Wex = Wex_[iEdg];
445 BGFLOAT &W = W_[iEdg];
446 edgeType type = type_[iEdg];
447 BGFLOAT oldW = W;
448 // BGFLOAT modDelta = fabs(delta);
449
450 if (delta <= fabs(STDPgap)) {
451 return;
452 }
453
454 // dw is the fractional change in synaptic strength; add 1.0 to become the scaling ratio
455 //dw = 1.0 + dw * epre * epost;
456 BGFLOAT dw = 1.0 + synapticWeightModification(iEdg, W, delta);
457
458 // if scaling ratio is less than zero, set it to zero so this synapse, its
459 // strength is always zero
460 // TODO: Where is the code for this?
461
462 // current weight multiplies dw (scaling ratio) to generate new weight
463 if (dw != 0.0)
464 W *= dw;
465
466 // if new weight is bigger than Wex (maximum allowed weight), then set it to Wex
467 if (fabs(W) > Wex) {
468 W = edgSign(type) * Wex;
469 }
470 /*
471 LOG4CPLUS_DEBUG(edgeLogger_, endl <<
472 "iEdg value " << iEdg
473 << "; source:" << srcVertex
474 << "; dest:" << destVertex
475 << "; delta:" << delta
476 << "; oldW:" << oldW
477 << " ;W:" << W
478 << endl);
479
480 */

487bool AllSTDPSynapses::isSpikeQueuePost(BGSIZE iEdg)
488{
489 uint32_t &delayQueue = delayQueuePost_[iEdg];
490 int &delayIdx = delayIndexPost_[iEdg];
491 int &ldelayQueue = delayQueuePostLength_[iEdg];
492
493 bool r = delayQueue & (0x1 << delayIdx);
494 delayQueue &= ~(0x1 << delayIdx);
495 if (++delayIdx >= ldelayQueue) {
496 delayIdx = 0;
497 }
498 return r;
499}

504void AllSTDPSynapses::postSpikeHit(BGSIZE iEdg)
505{
506 uint32_t &delay_queue = delayQueuePost_[iEdg];
507 int &delayIdx = delayIndexPost_[iEdg];
508 int &ldelayQueue = delayQueuePostLength_[iEdg];
509 int &total_delay = totalDelayPost_[iEdg];
510
511 // Add to spike queue
512
513 // calculate index where to insert the spike into delayQueue
514 int idx = delayIdx + total_delay;
515 if (idx >= ldelayQueue) {
516 idx -= ldelayQueue;
517 }
518
519 // set a spike
520 assert(!(delay_queue & (0x1 << idx)));
521 delay_queue |= (0x1 << idx);
522}
