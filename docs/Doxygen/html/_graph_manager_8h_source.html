176template <typename VertexProperties> bool GraphManager<VertexProperties>::readGraph()
177{
178 // Load graphml file into a BGL graph
179 ifstream graph_file;
180
181 // If graphFilePath_ isn't already defined, get it from ParameterManager
182 if (graphFilePath_ == "") {
183 string path = "//graphmlFile/text()";
184 if (!ParameterManager::getInstance().getStringByXpath(path, graphFilePath_)) {
185 cerr << "Could not find XML path: " << path << ".\n";
186 return false;
187 };
188 }
189
190 graph_file.open(graphFilePath_.c_str());
191 if (!graph_file.is_open()) {
192 cerr << "Failed to open file: " << graphFilePath_ << ".\n";
193 return false;
194 }
195
196 boost::read_graphml(graph_file, graph_, dp_);
197 return true;
198}

270template <typename VertexProperties>
271const list<typename GraphManager<VertexProperties>::EdgeDescriptor>
272 GraphManager<VertexProperties>::edgesSortByTarget() const
273{
274 list<EdgeDescriptor> ei_list;
275 EdgeIterator ei, ei_end;
276 for (boost::tie(ei, ei_end) = edges(); ei != ei_end; ++ei) {
277 ei_list.push_back(*ei);
278 }
279
280 // Use a lambda function for sorting the list of edges
281 ei_list.sort([this](EdgeDescriptor const &a, EdgeDescriptor const &b) {
282 return this->target(a) < this->target(b);
283 });
284
285 return ei_list;
286}
