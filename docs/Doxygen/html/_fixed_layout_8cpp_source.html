19void FixedLayout::setup()
20{
21 // Base class allocates memory for: xLoc_, yLoc, dist2_, and dist_
22 // so we call its method first
23 Layout::setup();
24
25 // Initialize neuron locations memory, grab global info
26 initVerticesLocs();
27
28 // computing distance between each pair of vertices given each vertex's xy location
29 for (int n = 0; n < numVertices_ - 1; n++) {
30 for (int n2 = n + 1; n2 < numVertices_; n2++) {
31 // distance^2 between two points in point-slope form
32 dist2_(n, n2) = (xloc_[n] - xloc_[n2]) * (xloc_[n] - xloc_[n2])
33 + (yloc_[n] - yloc_[n2]) * (yloc_[n] - yloc_[n2]);
34
35 // both points are equidistant from each other
36 dist2_(n2, n) = dist2_(n, n2);
37 }
38 }
39
40 // take the square root to get actual distance (Pythagoras was right!)
41 dist_ = sqrt(dist2_);
42}

159void FixedLayout::initVerticesLocs()
160{
161 int numVertices = Simulator::getInstance().getTotalVertices();
162
163 // Initialize vertex locations
164 if (gridLayout_) {
165 // grid layout
166 for (int i = 0; i < numVertices; i++) {
167 xloc_[i] = i % Simulator::getInstance().getHeight();
168 yloc_[i] = i / Simulator::getInstance().getHeight();
169 }
170 } else {
171 // random layout
172 for (int i = 0; i < numVertices; i++) {
173 xloc_[i] = initRNG.inRange(0, Simulator::getInstance().getWidth());
174 yloc_[i] = initRNG.inRange(0, Simulator::getInstance().getHeight());
175 }
176 }
