20void FixedLayout::setup()
21{
22 // Base class allocates memory for: xLoc_, yLoc, dist2_, and dist_
23 // so we call its method first
24 Layout::setup();
25
26 // Initialize neuron locations memory, grab global info
27 initVerticesLocs();
28
29 // computing distance between each pair of vertices given each vertex's xy location
30 for (int n = 0; n < numVertices_ - 1; n++) {
31 for (int n2 = n + 1; n2 < numVertices_; n2++) {
32 // distance^2 between two points in point-slope form
33 dist2_(n, n2) = (xloc_[n] - xloc_[n2]) * (xloc_[n] - xloc_[n2])
34 + (yloc_[n] - yloc_[n2]) * (yloc_[n] - yloc_[n2]);
35
36 // both points are equidistant from each other
37 dist2_(n2, n) = dist2_(n, n2);
38 }
39 }
40
41 // take the square root to get actual distance (Pythagoras was right!)
42 dist_ = sqrt(dist2_);
43}

144edgeType FixedLayout::edgType(int srcVertex, int destVertex)
145{
146 if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == INH)

148 else if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == EXC)
149 return IE;

151 return EI;

159void FixedLayout::initVerticesLocs()
160{
161 int numVertices = Simulator::getInstance().getTotalVertices();
162
163 // Initialize vertex locations
164 if (gridLayout_) {
165 // grid layout
166 for (int i = 0; i < numVertices; i++) {
167 xloc_[i] = i % height_;
168 yloc_[i] = i / height_;
169 }
170 } else {
171 // random layout
172 for (int i = 0; i < numVertices; i++) {
173 xloc_[i] = initRNG.inRange(0, width_);
174 yloc_[i] = initRNG.inRange(0, height_);
175 }
176 }
177}
