121 bool Connections911::erasePSAP(IAllVertices &vertices, Layout *layout) {

123 
124  vector<int> psaps;
125  psaps.clear();
126 
127  // Find all psaps
128  for (int i = 0; i < numVertices; i++) {
129  if (layout->vertexTypeMap_[i] == PSAP) {
130  psaps.push_back(i);
131  }
132  }
133 
134  // Only 1 psap, do not delete me :(
135  if (psaps.size() < 2) { return false; }

137  // Pick random PSAP
138  int randVal = rng.inRange(0, psaps.size());
139  int randPSAP = psaps[randVal];
140  psaps.erase(psaps.begin() + randVal);
141 
142  BGSIZE maxTotalEdges = edges_->maxEdgesPerVertex_ * numVertices;
143  bool changesMade = false;

145  vector<int> respsToReroute;
146 
147  callersToReroute.clear();
148  respsToReroute.clear();
149 
150  // Iterate through all edges
151  for (int iEdg = 0; iEdg < maxTotalEdges; iEdg++) {
152  if (!edges_->inUse_[iEdg]) { continue; }
153  int srcVertex = edges_->sourceVertexIndex_[iEdg];
154  int destVertex = edges_->destVertexIndex_[iEdg];
155 
156  // Find PSAP edge
157  if (srcVertex == randPSAP || destVertex == randPSAP) {
158  // Record erased edge
159  ChangedEdge erasedEdge;

166  edges_->eraseEdge(destVertex, iEdg);
167

169  if (layout->vertexTypeMap_[srcVertex] == CALR) {
170  callersToReroute.push_back(srcVertex);
171  }
172 
173  // Identify all psap-less responders
174  if (layout->vertexTypeMap_[destVertex] == RESP) {
175  respsToReroute.push_back(destVertex);
176  }
177  }
178  }
179 
180  if (changesMade) {
181  // This is here so that we don't delete the vertex if we can't find any edges
182  verticesErased.push_back(randPSAP);
183  layout->vertexTypeMap_[randPSAP] = VTYPE_UNDEF;
184  }
185 
186  // Failsafe
187  if (psaps.size() < 1) { return false; }
188 
189  // For each psap-less caller, find closest match
190  for (int i = 0; i < callersToReroute.size(); i++) {
191  int srcVertex = callersToReroute[i];
192 
193  int closestPSAP = psaps[0];
194  BGFLOAT smallestDist = (*layout->dist_)(srcVertex, closestPSAP);
195 
196  // Find closest PSAP
197  for (int i = 0; i < psaps.size(); i++) {
198  BGFLOAT dist = (*layout->dist_)(srcVertex, psaps[i]);
199  if (dist < smallestDist) {
200  smallestDist = dist;
201  closestPSAP = psaps[i];
202  }
203  }
204 
205  // Insert Caller to PSAP edge
206  BGFLOAT *sumPoint = &(dynamic_cast<AllVertices *>(&vertices)->summationMap_[closestPSAP]);
207  BGSIZE iEdg;
208  edges_->addEdge(iEdg, CP, srcVertex, closestPSAP, sumPoint, Simulator::getInstance().getDeltaT());
209 
210  // Record added edge
211  ChangedEdge addedEdge;
212  addedEdge.srcV = srcVertex;
213  addedEdge.destV = closestPSAP;
214  addedEdge.eType = CP;
215  edgesAdded.push_back(addedEdge);
216  }
217 
218  // For each psap-less responder, find closest match
219  for (int i = 0; i < respsToReroute.size(); i++) {
220  int destVertex = respsToReroute[i];
221 
222  int closestPSAP = psaps[0];
223  BGFLOAT smallestDist = (*layout->dist_)(closestPSAP, destVertex);
224
