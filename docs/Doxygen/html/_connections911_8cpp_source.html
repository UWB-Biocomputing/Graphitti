16void Connections911::setup()
17{
18 int added = 0;
19 LOG4CPLUS_INFO(fileLogger_, "Initializing connections");
20
21 // we can obtain the Layout, which holds the vertices, from the Model
22 Layout &layout = *Simulator::getInstance().getModel()->getLayout();
23 AllVertices &vertices = *layout.getVertices();
24
25 // Get list of edges sorted by target in ascending order from GraphManager
26 GraphManager &gm = GraphManager::getInstance();
27 auto sorted_edge_list = gm.edgesSortByTarget();
28
29 // add sorted edges
30 for (auto it = sorted_edge_list.begin(); it != sorted_edge_list.end(); ++it) {
31 size_t srcV = gm.source(*it);
32 size_t destV = gm.target(*it);
33 edgeType type = layout.edgType(srcV, destV);
34 BGFLOAT *sumPoint = &vertices.summationMap_[destV];
35
36 BGFLOAT dist = layout.dist_(srcV, destV);
37 LOG4CPLUS_DEBUG(edgeLogger_, "Source: " << srcV << " Dest: " << destV << " Dist: " << dist);
38
39 BGSIZE iEdg;
40 edges_->addEdge(iEdg, type, srcV, destV, sumPoint, Simulator::getInstance().getDeltaT());
41 added++;
42 }
43
44 LOG4CPLUS_DEBUG(fileLogger_, "Added connections: " << added);
45}

102bool Connections911::erasePSAP(AllVertices &vertices, Layout &layout)
103{
104 int numVertices = Simulator::getInstance().getTotalVertices();
105
106 vector<int> psaps;
107 psaps.clear();
108
109 // Find all psaps
110 for (int i = 0; i < numVertices; i++) {
111 if (layout.vertexTypeMap_[i] == PSAP) {
112 psaps.push_back(i);
113 }
114 }
115
116 // Only 1 psap, do not delete me :(
117 if (psaps.size() < 2) {
118 return false;
119 }
120
121 // Pick random PSAP
122 int randVal = initRNG.inRange(0, psaps.size());
123 int randPSAP = psaps[randVal];
124 psaps.erase(psaps.begin() + randVal);
125
126 BGSIZE maxTotalEdges = edges_->maxEdgesPerVertex_ * numVertices;
127 bool changesMade = false;
128 vector<int> callersToReroute;
129 vector<int> respsToReroute;
130
131 callersToReroute.clear();
132 respsToReroute.clear();
133
134 // Iterate through all edges
135 for (int iEdg = 0; iEdg < maxTotalEdges; iEdg++) {
136 if (!edges_->inUse_[iEdg]) {
137 continue;
138 }
139 int srcVertex = edges_->sourceVertexIndex_[iEdg];
140 int destVertex = edges_->destVertexIndex_[iEdg];
141
142 // Find PSAP edge
143 if (srcVertex == randPSAP || destVertex == randPSAP) {
144 // Record erased edge
145 ChangedEdge erasedEdge;
146 erasedEdge.srcV = srcVertex;
147 erasedEdge.destV = destVertex;
148 erasedEdge.eType = layout.edgType(srcVertex, destVertex);
149 edgesErased.push_back(erasedEdge);
150
151 changesMade = true;
152 edges_->eraseEdge(destVertex, iEdg);
153
154 // Identify all psap-less callers
155 if (layout.vertexTypeMap_[srcVertex] == CALR) {
156 callersToReroute.push_back(srcVertex);
157 }
158
159 // Identify all psap-less responders
160 if (layout.vertexTypeMap_[destVertex] == RESP) {
161 respsToReroute.push_back(destVertex);
162 }
163 }
164 }
165
166 if (changesMade) {
167 // This is here so that we don't delete the vertex if we can't find any edges
168 verticesErased.push_back(randPSAP);
169 layout.vertexTypeMap_[randPSAP] = VTYPE_UNDEF;
170 }
171
172 // Failsafe
173 if (psaps.size() < 1) {
174 return false;
175 }
176
177 // For each psap-less caller, find closest match
178 for (int i = 0; i < callersToReroute.size(); i++) {
179 int srcVertex = callersToReroute[i];
180
181 int closestPSAP = psaps[0];
182 BGFLOAT smallestDist = layout.dist_(srcVertex, closestPSAP);
183
184 // Find closest PSAP
185 for (int i = 0; i < psaps.size(); i++) {
186 BGFLOAT dist = layout.dist_(srcVertex, psaps[i]);
187 if (dist < smallestDist) {
188 smallestDist = dist;
189 closestPSAP = psaps[i];
190 }
191 }
192
193 // Insert Caller to PSAP edge
194 BGFLOAT *sumPoint = &vertices.summationMap_[closestPSAP];
195 BGSIZE iEdg;
196 edges_->addEdge(iEdg, CP, srcVertex, closestPSAP, sumPoint,
197 Simulator::getInstance().getDeltaT());
198
199 // Record added edge
200 ChangedEdge addedEdge;
201 addedEdge.srcV = srcVertex;
202 addedEdge.destV = closestPSAP;
203 addedEdge.eType = CP;
204 edgesAdded.push_back(addedEdge);
205 }
206
207 // For each psap-less responder, find closest match
208 for (int i = 0; i < respsToReroute.size(); i++) {
209 int destVertex = respsToReroute[i];
210
211 int closestPSAP = psaps[0];
212 BGFLOAT smallestDist = layout.dist_(closestPSAP, destVertex);
213
214 // Find closest PSAP
215 for (int i = 0; i < psaps.size(); i++) {
216 BGFLOAT dist = layout.dist_(psaps[i], destVertex);
217 if (dist < smallestDist) {
218 smallestDist = dist;
219 closestPSAP = psaps[i];
220 }
221 }
222
223 // Insert PSAP to Responder edge
224 BGFLOAT *sumPoint = &vertices.summationMap_[destVertex];
225 BGSIZE iEdg;
226 edges_->addEdge(iEdg, PR, closestPSAP, destVertex, sumPoint,
227 Simulator::getInstance().getDeltaT());
228
229 // Record added edge
230 ChangedEdge addedEdge;
231 addedEdge.srcV = closestPSAP;
232 addedEdge.destV = destVertex;
233 addedEdge.eType = PR;
234 edgesAdded.push_back(addedEdge);
235 }
236
237 return changesMade;
238}
