15void Connections911::setup()
16{
17 int added = 0;
18 LOG4CPLUS_INFO(fileLogger_, "Initializing connections");
19
20 // we can obtain the Layout, which holds the vertices, from the Model
21 Layout &layout = Simulator::getInstance().getModel().getLayout();
22 AllVertices &vertices = layout.getVertices();
23
24 // Get list of edges sorted by target in ascending order from GraphManager
25 GraphManager &gm = GraphManager::getInstance();
26 auto sorted_edge_list = gm.edgesSortByTarget();
27
28 // add sorted edges
29 for (auto it = sorted_edge_list.begin(); it != sorted_edge_list.end(); ++it) {
30 size_t srcV = gm.source(*it);
31 size_t destV = gm.target(*it);
32 edgeType type = layout.edgType(srcV, destV);
33
34 BGFLOAT dist = layout.dist_(srcV, destV);
35 LOG4CPLUS_DEBUG(edgeLogger_, "Source: " << srcV << " Dest: " << destV << " Dist: " << dist);
36
37 BGSIZE iEdg = edges_->addEdge(type, srcV, destV, Simulator::getInstance().getDeltaT());
38 added++;
39 }
40
41 LOG4CPLUS_DEBUG(fileLogger_, "Added connections: " << added);
42}

94BGSIZE Connections911::getEdgeToClosestResponder(const Call &call, BGSIZE vertexIdx)
95{
96 All911Edges &edges911 = dynamic_cast<All911Edges &>(*edges_);
97
98 vertexType requiredType;
99 if (call.type == "Law")
100 requiredType = LAW;
101 else if (call.type == "EMS")
102 requiredType = EMS;
103 else if (call.type == "Fire")
104 requiredType = FIRE;
105
106 // loop over the outgoing edges looking for the responder with the shortest
107 // Euclidean distance to the call's location.
108 BGSIZE startOutEdg = synapseIndexMap_->outgoingEdgeBegin_[vertexIdx];
109 BGSIZE outEdgCount = synapseIndexMap_->outgoingEdgeCount_[vertexIdx];
110 Layout911 &layout911
111 = dynamic_cast<Layout911 &>(Simulator::getInstance().getModel().getLayout());
112
113 BGSIZE resp, respEdge;
114 double minDistance = numeric_limits<double>::max();
115 for (BGSIZE eIdxMap = startOutEdg; eIdxMap < startOutEdg + outEdgCount; ++eIdxMap) {
116 BGSIZE outEdg = synapseIndexMap_->outgoingEdgeIndexMap_[eIdxMap];
117 assert(edges911.inUse_[outEdg]); // Edge must be in use
118
119 BGSIZE dstVertex = edges911.destVertexIndex_[outEdg];
120 if (layout911.vertexTypeMap_[dstVertex] == requiredType) {
121 double distance = layout911.getDistance(dstVertex, call.x, call.y);
122
123 if (distance < minDistance) {
124 minDistance = distance;
125 resp = dstVertex;
126 respEdge = outEdg;
127 }
128 }
129 }
130
131 // We must have found the closest responder of the right type
132 assert(minDistance < numeric_limits<double>::max());
133 assert(layout911.vertexTypeMap_[resp] == requiredType);
134 return respEdge;
135}

139bool Connections911::erasePSAP(AllVertices &vertices, Layout &layout)
140{
141 int numVertices = Simulator::getInstance().getTotalVertices();
142
143 vector<int> psaps;
144 psaps.clear();
145
146 // Find all psaps
147 for (int i = 0; i < numVertices; i++) {
148 if (layout.vertexTypeMap_[i] == PSAP) {
149 psaps.push_back(i);
150 }
151 }
152
153 // Only 1 psap, do not delete me :(
154 if (psaps.size() < 2) {
155 return false;
156 }
157
158 // Pick random PSAP
159 int randVal = initRNG.inRange(0, psaps.size());
160 int randPSAP = psaps[randVal];
161 psaps.erase(psaps.begin() + randVal);
162
163 BGSIZE maxTotalEdges = edges_->maxEdgesPerVertex_ * numVertices;
164 bool changesMade = false;
165 vector<int> callersToReroute;
166 vector<int> respsToReroute;
167
168 callersToReroute.clear();
169 respsToReroute.clear();
170
171 // Iterate through all edges
172 for (int iEdg = 0; iEdg < maxTotalEdges; iEdg++) {
173 if (!edges_->inUse_[iEdg]) {
174 continue;
175 }
176 int srcVertex = edges_->sourceVertexIndex_[iEdg];
177 int destVertex = edges_->destVertexIndex_[iEdg];
178
179 // Find PSAP edge
180 if (srcVertex == randPSAP || destVertex == randPSAP) {
181 // Record erased edge
182 ChangedEdge erasedEdge;
183 erasedEdge.srcV = srcVertex;
184 erasedEdge.destV = destVertex;
185 erasedEdge.eType = layout.edgType(srcVertex, destVertex);
186 edgesErased.push_back(erasedEdge);
187
188 changesMade = true;
189 edges_->eraseEdge(destVertex, iEdg);
190
191 // Identify all psap-less callers
192 if (layout.vertexTypeMap_[srcVertex] == CALR) {
193 callersToReroute.push_back(srcVertex);
194 }
195
196 // Identify all psap-less responders
197 if (layout.vertexTypeMap_[destVertex] == LAW || layout.vertexTypeMap_[destVertex] == FIRE
198 || layout.vertexTypeMap_[destVertex] == EMS) {
199 respsToReroute.push_back(destVertex);
200 }
201 }
202 }
203
204 if (changesMade) {
205 // This is here so that we don't delete the vertex if we can't find any edges
206 verticesErased.push_back(randPSAP);
207 layout.vertexTypeMap_[randPSAP] = VTYPE_UNDEF;
208 }
209
210 // Failsafe
211 if (psaps.size() < 1) {
212 return false;
213 }
214
215 // For each psap-less caller, find closest match
216 for (int i = 0; i < callersToReroute.size(); i++) {
217 int srcVertex = callersToReroute[i];
218
219 int closestPSAP = psaps[0];
220 BGFLOAT smallestDist = layout.dist_(srcVertex, closestPSAP);
221
222 // Find closest PSAP
223 for (int i = 0; i < psaps.size(); i++) {
224 BGFLOAT dist = layout.dist_(srcVertex, psaps[i]);
225 if (dist < smallestDist) {
226 smallestDist = dist;
227 closestPSAP = psaps[i];
228 }
229 }
230
231 // Insert Caller to PSAP edge
232 BGSIZE iEdg
233 = edges_->addEdge(CP, srcVertex, closestPSAP, Simulator::getInstance().getDeltaT());
234
235 // Record added edge
236 ChangedEdge addedEdge;
237 addedEdge.srcV = srcVertex;
238 addedEdge.destV = closestPSAP;
239 addedEdge.eType = CP;
240 edgesAdded.push_back(addedEdge);
241 }
242
243 // For each psap-less responder, find closest match
244 for (int i = 0; i < respsToReroute.size(); i++) {
245 int destVertex = respsToReroute[i];
246
247 int closestPSAP = psaps[0];
248 BGFLOAT smallestDist = layout.dist_(closestPSAP, destVertex);
249
250 // Find closest PSAP
251 for (int i = 0; i < psaps.size(); i++) {
252 BGFLOAT dist = layout.dist_(psaps[i], destVertex);
253 if (dist < smallestDist) {
254 smallestDist = dist;
255 closestPSAP = psaps[i];
256 }
257 }
258
259 // Insert PSAP to Responder edge
260 BGSIZE iEdg
261 = edges_->addEdge(PR, closestPSAP, destVertex, Simulator::getInstance().getDeltaT());
262
263 // Record added edge
264 ChangedEdge addedEdge;
265 addedEdge.srcV = closestPSAP;
266 addedEdge.destV = destVertex;
267 addedEdge.eType = PR;
268 edgesAdded.push_back(addedEdge);
269 }
270
271 return changesMade;
272}
