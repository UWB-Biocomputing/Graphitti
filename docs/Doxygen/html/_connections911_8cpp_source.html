15void Connections911::setup()
16{
17 int added = 0;
18 LOG4CPLUS_INFO(fileLogger_, "Initializing connections");
19
20 // we can obtain the Layout, which holds the vertices, from the Model
21 Layout &layout = Simulator::getInstance().getModel().getLayout();
22 AllVertices &vertices = layout.getVertices();
23
24 // Get list of edges sorted by target in ascending order from GraphManager
25 GraphManager<NG911VertexProperties> &gm = GraphManager<NG911VertexProperties>::getInstance();
26 auto sorted_edge_list = gm.edgesSortByTarget();
27
28 // add sorted edges
29 for (auto it = sorted_edge_list.begin(); it != sorted_edge_list.end(); ++it) {
30 size_t srcV = gm.source(*it);
31 size_t destV = gm.target(*it);
32 edgeType type = layout.edgType(srcV, destV);
33
34 BGFLOAT dist = layout.dist_(srcV, destV);
35 LOG4CPLUS_DEBUG(edgeLogger_, "Source: " << srcV << " Dest: " << destV << " Dist: " << dist);
36
37 BGSIZE iEdg = edges_->addEdge(type, srcV, destV, Simulator::getInstance().getDeltaT());
38 added++;
39 }
40
41 LOG4CPLUS_DEBUG(fileLogger_, "Added connections: " << added);
42}

101BGSIZE Connections911::getEdgeToClosestResponder(const Call &call, BGSIZE vertexIdx)
102{
103 All911Edges &edges911 = dynamic_cast<All911Edges &>(*edges_);
104
105 vertexType requiredType;
106 if (call.type == "Law")
107 requiredType = vertexType::LAW;
108 else if (call.type == "EMS")
109 requiredType = vertexType::EMS;
110 else if (call.type == "Fire")
111 requiredType = vertexType::FIRE;
112
113 // loop over the outgoing edges looking for the responder with the shortest
114 // Euclidean distance to the call's location.
115 BGSIZE startOutEdg = synapseIndexMap_->outgoingEdgeBegin_[vertexIdx];
116 BGSIZE outEdgCount = synapseIndexMap_->outgoingEdgeCount_[vertexIdx];
117 Layout911 &layout911
118 = dynamic_cast<Layout911 &>(Simulator::getInstance().getModel().getLayout());
119
120 BGSIZE resp, respEdge;
121 double minDistance = numeric_limits<double>::max();
122 for (BGSIZE eIdxMap = startOutEdg; eIdxMap < startOutEdg + outEdgCount; ++eIdxMap) {
123 BGSIZE outEdg = synapseIndexMap_->outgoingEdgeIndexMap_[eIdxMap];
124 assert(edges911.inUse_[outEdg]); // Edge must be in use
125
126 BGSIZE dstVertex = edges911.destVertexIndex_[outEdg];
127 if (layout911.vertexTypeMap_[dstVertex] == requiredType) {
128 double distance = layout911.getDistance(dstVertex, call.x, call.y);
129
130 if (distance < minDistance) {
131 minDistance = distance;
132 resp = dstVertex;
133 respEdge = outEdg;
134 }
135 }
136 }
137
138 // We must have found the closest responder of the right type
139 assert(minDistance < numeric_limits<double>::max());
140 assert(layout911.vertexTypeMap_[resp] == requiredType);
141 return respEdge;
142}

146bool Connections911::erasePSAP(AllVertices &vertices, Layout &layout)
147{
148 int numVertices = Simulator::getInstance().getTotalVertices();
149
150 vector<int> psaps;
151 psaps.clear();
152
153 // Find all psaps
154 for (int i = 0; i < numVertices; i++) {
155 if (layout.vertexTypeMap_[i] == vertexType::PSAP) {
156 psaps.push_back(i);
157 }
158 }
159
160 // Only 1 psap, do not delete me :(
161 if (psaps.size() < 2) {
162 return false;
163 }
164
165 // Pick random PSAP
166 int randVal = initRNG.inRange(0, psaps.size());
167 int randPSAP = psaps[randVal];
168 psaps.erase(psaps.begin() + randVal);
169
170 BGSIZE maxTotalEdges = edges_->maxEdgesPerVertex_ * numVertices;
171 bool changesMade = false;
172 vector<int> callersToReroute;
173 vector<int> respsToReroute;
174
175 callersToReroute.clear();
176 respsToReroute.clear();
177
178 // Iterate through all edges
179 for (int iEdg = 0; iEdg < maxTotalEdges; iEdg++) {
180 if (!edges_->inUse_[iEdg]) {
181 continue;
182 }
183 int srcVertex = edges_->sourceVertexIndex_[iEdg];
184 int destVertex = edges_->destVertexIndex_[iEdg];
185
186 // Find PSAP edge
187 if (srcVertex == randPSAP || destVertex == randPSAP) {
188 // Record erased edge
189 ChangedEdge erasedEdge;
190 erasedEdge.srcV = srcVertex;
191 erasedEdge.destV = destVertex;
192 erasedEdge.eType = layout.edgType(srcVertex, destVertex);
193 edgesErased_.push_back(erasedEdge);
194
195 changesMade = true;
196 edges_->eraseEdge(destVertex, iEdg);
197
198 // Identify all psap-less callers
199 if (layout.vertexTypeMap_[srcVertex] == vertexType::CALR) {
200 callersToReroute.push_back(srcVertex);
201 }
202
203 // Identify all psap-less responders
204 if (layout.vertexTypeMap_[destVertex] == vertexType::LAW
205 || layout.vertexTypeMap_[destVertex] == vertexType::FIRE
206 || layout.vertexTypeMap_[destVertex] == vertexType::EMS) {
207 respsToReroute.push_back(destVertex);
208 }
209 }
210 }
211
212 if (changesMade) {
213 // This is here so that we don't delete the vertex if we can't find any edges
214 verticesErased_.push_back(randPSAP);
215 layout.vertexTypeMap_[randPSAP] = vertexType::VTYPE_UNDEF;
216 }
217
218 // Failsafe
219 if (psaps.size() < 1) {
220 return false;
221 }
222
223 // For each psap-less caller, find closest match
224 for (int i = 0; i < callersToReroute.size(); i++) {
225 int srcVertex = callersToReroute[i];
226
227 int closestPSAP = psaps[0];
228 BGFLOAT smallestDist = layout.dist_(srcVertex, closestPSAP);
229
230 // Find closest PSAP
231 for (int i = 0; i < psaps.size(); i++) {
232 BGFLOAT dist = layout.dist_(srcVertex, psaps[i]);
233 if (dist < smallestDist) {
234 smallestDist = dist;
235 closestPSAP = psaps[i];
236 }
237 }
238
239 // Insert Caller to PSAP edge
240 BGSIZE iEdg = edges_->addEdge(edgeType::CP, srcVertex, closestPSAP,
241 Simulator::getInstance().getDeltaT());
242
243 // Record added edge
244 ChangedEdge addedEdge;
245 addedEdge.srcV = srcVertex;
246 addedEdge.destV = closestPSAP;
247 addedEdge.eType = edgeType::CP;
248 edgesAdded_.push_back(addedEdge);
249 }
250
251 // For each psap-less responder, find closest match
252 for (int i = 0; i < respsToReroute.size(); i++) {
253 int destVertex = respsToReroute[i];
254
255 int closestPSAP = psaps[0];
256 BGFLOAT smallestDist = layout.dist_(closestPSAP, destVertex);
257
258 // Find closest PSAP
259 for (int i = 0; i < psaps.size(); i++) {
260 BGFLOAT dist = layout.dist_(psaps[i], destVertex);
261 if (dist < smallestDist) {
262 smallestDist = dist;
263 closestPSAP = psaps[i];
264 }
265 }
266
267 // Insert PSAP to Responder edge
268 BGSIZE iEdg = edges_->addEdge(edgeType::PR, closestPSAP, destVertex,
269 Simulator::getInstance().getDeltaT());
270
271 // Record added edge
272 ChangedEdge addedEdge;
273 addedEdge.srcV = closestPSAP;
274 addedEdge.destV = destVertex;
275 addedEdge.eType = edgeType::PR;
276 edgesAdded_.push_back(addedEdge);
277 }
278
279 return changesMade;
280}
