69 void Layout::setupLayout() {
70  int numVertices = Simulator::getInstance().getTotalVertices();
71 
72  // Allocate memory
73  xloc_ = new VectorMatrix(MATRIX_TYPE, MATRIX_INIT, 1, numVertices);
74  yloc_ = new VectorMatrix(MATRIX_TYPE, MATRIX_INIT, 1, numVertices);
75  dist2_ = new CompleteMatrix(MATRIX_TYPE, MATRIX_INIT, numVertices, numVertices);
76  dist_ = new CompleteMatrix(MATRIX_TYPE, MATRIX_INIT, numVertices, numVertices);
77 
78  // Initialize neuron locations memory, grab global info
79  initVerticesLocs();
80 
81  // computing distance between each pair of vertices given each vertex's xy location
82  for (int n = 0; n < numVertices - 1; n++) {
83  for (int n2 = n + 1; n2 < numVertices; n2++) {
84  // distance^2 between two points in point-slope form
85  (*dist2_)(n, n2) = ((*xloc_)[n] - (*xloc_)[n2]) * ((*xloc_)[n] - (*xloc_)[n2]) +
86  ((*yloc_)[n] - (*yloc_)[n2]) * ((*yloc_)[n] - (*yloc_)[n2]);
87 
88  // both points are equidistant from each other
89  (*dist2_)(n2, n) = (*dist2_)(n, n2);
90  }
91  }
92 
93  // take the square root to get actual distance (Pythagoras was right!)
94  (*dist_) = sqrt((*dist2_));
95 
96  // more allocation of internal memory
97  vertexTypeMap_ = new vertexType[numVertices]; // todo: make array into vector
98  starterMap_ = new bool[numVertices]; // todo: make array into vector
99 }
