55void EventBuffer::insertEvent(uint64_t timeStep)
56{
57 // If the buffer is full, then this is an error condition
58 assert(((queueEnd_ + 1) % eventTimeSteps_.size()) != queueFront_);
59
60 // Insert time step and increment the queue end index, mod the buffer size
61 eventTimeSteps_[queueEnd_] = timeStep;
62 queueEnd_ = (queueEnd_ + 1) % eventTimeSteps_.size();
63 numEventsInEpoch_ += 1;
64}

66uint64_t EventBuffer::getPastEvent(int offset) const
67{
68 // Quick checks: offset must be in past, and not larger than the buffer size
69 assert(((offset < 0)) && (offset > -(eventTimeSteps_.size() - 1)));
70
71 // The event is at queueEnd_ + offset (taking into account the
72 // buffer size, and the fact that offset is negative).
73 int index = queueEnd_ + offset;
74 if (index < 0)
75 index += eventTimeSteps_.size();
76
77 // Need to check that we're not asking for an item so long ago that it is
78 // not in the buffer. Note that there are three possibilities:
79 // 1. if queueEnd_ > queueFront_, then valid entries are within the range
80 // [queueFront_, queueEnd_)
81 // 2. if queueEnd_ < queueFront_, then the buffer wraps around the end of
82 // vector and valid entries are within the range [0, queueEnd_) or the
83 // range [queueFront_, size()).
84 // 3. if buffer is empty (queueFront_ == queueEnd_), then there are no events
85 //
86 // Note that this means that index at this point must always be less than
87 // queueEnd_ AND >= queueFront.
88 if ((index < queueEnd_) && (index >= queueFront_))
89 return eventTimeSteps_[index];
90 else
91 return numeric_limits<unsigned long>::max();
92}
