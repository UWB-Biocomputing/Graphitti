83void EventBuffer::insertEvent(uint64_t timeStep)
84{
85 // If the buffer is full, then this is an error condition
86 assert((numElementsInEpoch_ < dataSeries_.size()));
87
88 // Insert time step and increment the queue end index, mod the buffer size
89 dataSeries_[bufferEnd_] = timeStep;
90 bufferEnd_ = (bufferEnd_ + 1) % dataSeries_.size();

94uint64_t EventBuffer::getPastEvent(int offset) const
95{
96 // Quick checks: offset must be in past, and not larger than the buffer size
97 assert(((offset < 0)) && (offset > -(dataSeries_.size() - 1)));
98
99 // The event is at bufferEnd_ + offset (taking into account the
100 // buffer size, and the fact that offset is negative).
101 int index = bufferEnd_ + offset;
102 if (index < 0)
103 index += dataSeries_.size();
104
105 // Need to check that we're not asking for an item so long ago that it is
106 // not in the buffer. Note that there are three possibilities:
107 // 1. if bufferEnd_ > bufferFront_, then valid entries are within the range
108 // [bufferFront_, bufferEnd_)
109 // 2. if bufferEnd_ < bufferFront_, then the buffer wraps around the end of
110 // vector and valid entries are within the range [0, bufferEnd_) or the
111 // range [bufferFront_, size()).
112 // 3. if buffer is empty (bufferFront_ == bufferEnd_), then there are no events
113 //
114 // Note that this means that index at this point must always be less than
115 // bufferEnd_ AND >= queueFront.
116 if ((index < bufferEnd_) && (index >= bufferFront_))
117 return dataSeries_[index];
118 else
119 return numeric_limits<unsigned long>::max();
120}
