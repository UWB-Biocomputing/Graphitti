104 // Converts the data type of the variable to HDF5 data type
105 void convertType()
106 {
107 if (dataType_ == typeid(uint64_t).name()) {
108 hdf5Datatype_ = PredType::NATIVE_UINT64;
109 } else if (dataType_ == typeid(bool).name()) {
110 hdf5Datatype_ = PredType::NATIVE_INT;
111 } else if (dataType_ == typeid(int).name()) {
112 hdf5Datatype_ = PredType::NATIVE_INT;
113 } else if (dataType_ == typeid(float).name()) {
114 hdf5Datatype_ = PredType::NATIVE_FLOAT;
115 } else if (dataType_ == typeid(double).name()) {
116 hdf5Datatype_ = PredType::NATIVE_DOUBLE;
117 } else if (dataType_ == typeid(vertexType).name()) {
118 hdf5Datatype_ = PredType::NATIVE_INT;
119 } else if (dataType_ == typeid(unsigned char).name()) {
120 hdf5Datatype_ = PredType::NATIVE_UCHAR;
121 } else {
122 throw runtime_error("Unsupported data type");
123 }
124 }

126 // Method to capture and write data to the HDF5 dataset
127 void captureData()
128 {
129 // Ensure the dataset exists and data can be written
130 if (variableLocation_.getNumElements() > 0) {
131 // Prepare the data buffer based on the HDF5 data type
132 if (hdf5Datatype_ == PredType::NATIVE_FLOAT) {
133 vector<float> dataBuffer(variableLocation_.getNumElements());
134 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
135 dataBuffer[i] = get<float>(variableLocation_.getElement(i));
136 }
137 // Write the data to the dataset
138 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
139
140 } else if (hdf5Datatype_ == PredType::NATIVE_INT) {
141 vector<int> dataBuffer(variableLocation_.getNumElements());
142 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
143 // For type int, a distinction needs to be made between vertexType and regular integers
144 // Since vertexType is an enum class, it needs to first be converted to an integer before storing
145
146 // 'decltype' determines the type at compile time
147 // 'decay_t' removes any references/const from the type
148 // This simplifies comparisons through 'is_same_v' to ensure type matches vertexType
149 if (std::is_same_v<vertexType, std::decay_t<decltype(get<vertexType>(
150 variableLocation_.getElement(i)))>>) {
151 // If type matches vertexType, convert to int before storing
152 dataBuffer[i]
153 = static_cast<int>(get<vertexType>(variableLocation_.getElement(i)));
154 } else {
155 // Otherwise, store as a regular integer
156 dataBuffer[i] = get<int>(variableLocation_.getElement(i));
157 }
158 }
159 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
160
161 } else if (hdf5Datatype_ == PredType::NATIVE_UINT64) {
162 vector<uint64_t> dataBuffer(variableLocation_.getNumElements());
163 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
164 dataBuffer[i] = get<uint64_t>(variableLocation_.getElement(i));
165 }
166 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
167
168 } else if (hdf5Datatype_ == PredType::NATIVE_DOUBLE) {
169 vector<double> dataBuffer(variableLocation_.getNumElements());
170 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
171 dataBuffer[i] = get<double>(variableLocation_.getElement(i));
172 }
173 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
174
175 } else if (hdf5Datatype_ == PredType::NATIVE_UCHAR) {
176 vector<unsigned char> dataBuffer(variableLocation_.getNumElements());
177 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
178 dataBuffer[i] = get<unsigned char>(variableLocation_.getElement(i));
179 }
180 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
181 } else {
182 // Throw an error if the data type is unsupported
183 throw runtime_error("Unsupported data type");
184 }
185 }
186 }
