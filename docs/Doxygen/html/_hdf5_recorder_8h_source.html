105 void convertType()
106 {
107 if (dataType_ == typeid(uint64_t).name()) {
108 hdf5Datatype_ = PredType::NATIVE_UINT64;
109 } else if (dataType_ == typeid(bool).name()) {
110 hdf5Datatype_ = PredType::NATIVE_INT;
111 } else if (dataType_ == typeid(int).name()) {
112 hdf5Datatype_ = PredType::NATIVE_INT;
113 } else if (dataType_ == typeid(float).name()) {
114 hdf5Datatype_ = PredType::NATIVE_FLOAT;
115 } else if (dataType_ == typeid(double).name()) {
116 hdf5Datatype_ = PredType::NATIVE_DOUBLE;
117 } else {
118 throw runtime_error("Unsupported data type");

122 // Method to capture and write data to the HDF5 dataset
123 void captureData()
124 {
125 // Ensure the dataset exists and data can be written
126 if (variableLocation_.getNumElements() > 0) {
127 // Prepare the data buffer based on the HDF5 data type
128 if (hdf5Datatype_ == PredType::NATIVE_FLOAT) {
129 vector<float> dataBuffer(variableLocation_.getNumElements());
130 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
131 dataBuffer[i] = get<float>(variableLocation_.getElement(i));
132 }
133 // Write the data to the dataset
134 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
135
136 } else if (hdf5Datatype_ == PredType::NATIVE_INT) {
137 vector<int> dataBuffer(variableLocation_.getNumElements());
138 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
139 dataBuffer[i] = get<int>(variableLocation_.getElement(i));
140 }
141 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);

143 } else if (hdf5Datatype_ == PredType::NATIVE_UINT64) {
144 vector<uint64_t> dataBuffer(variableLocation_.getNumElements());
145 for (int i = 0; i < variableLocation_.getNumElements(); ++i) {
146 dataBuffer[i] = get<uint64_t>(variableLocation_.getElement(i));
147 }
148 hdf5DataSet_.write(dataBuffer.data(), hdf5Datatype_);
