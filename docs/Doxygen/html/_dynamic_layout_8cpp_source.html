148edgeType DynamicLayout::edgType(const int srcVertex, const int destVertex)
149{
150 if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == INH)
151 return II;
152 else if (vertexTypeMap_[srcVertex] == INH && vertexTypeMap_[destVertex] == EXC)
153 return IE;
154 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == INH)
155 return EI;
156 else if (vertexTypeMap_[srcVertex] == EXC && vertexTypeMap_[destVertex] == EXC)
157 return EE;
158
159 return ETYPE_UNDEF;
