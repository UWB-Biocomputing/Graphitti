58void Layout911::setup()
59{
60 // Base class allocates memory for: xLoc_, yLoc, dist2_, and dist_
61 // so we call its method first
62 Layout::setup();
63
64 // Loop over all vertices and set thir x and y locations
65 GraphManager::VertexIterator vi, vi_end;
66 GraphManager &gm = GraphManager::getInstance();
67 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
68 assert(*vi < numVertices_);
69 xloc_[*vi] = gm[*vi].x;
70 yloc_[*vi] = gm[*vi].y;
71 }
72
73 // Now we cache the between each pair of vertices distances^2 into a matrix
74 for (int n = 0; n < numVertices_ - 1; n++) {
75 for (int n2 = n + 1; n2 < numVertices_; n2++) {
76 // distance^2 between two points in point-slope form
77 dist2_(n, n2) = (xloc_[n] - xloc_[n2]) * (xloc_[n] - xloc_[n2])
78 + (yloc_[n] - yloc_[n2]) * (yloc_[n] - yloc_[n2]);
79
80 // both points are equidistant from each other
81 dist2_(n2, n) = dist2_(n, n2);
82 }
83 }
84
85 // Finally take the square root to get the distances
86 dist_ = sqrt(dist2_);
87}

95void Layout911::generateVertexTypeMap(int numVertices)
96{
97 DEBUG(cout << "\nInitializing vertex type map" << endl;);
98
99 // Map vertex type string to vertexType
100 // In the GraphML file Responders are divided in LAW, FIRE, and EMS.
101 // Perhaps, we need to expand the vertex types?
102 map<string, vertexType> vTypeMap = {{"CALR", vertexType::CALR},
103 {"LAW", vertexType::RESP},
104 {"FIRE", vertexType::RESP},
105 {"EMS", vertexType::RESP},
106 {"PSAP", vertexType::PSAP}};
107 // Count map for debugging
108 map<string, int> vTypeCount;
109
110 // Add all vertices
111 GraphManager::VertexIterator vi, vi_end;
112 GraphManager &gm = GraphManager::getInstance();
113 LOG4CPLUS_DEBUG(fileLogger_, "\nvertices in graph: " << gm.numVertices());
114 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
115 assert(*vi < numVertices_);
116 vertexTypeMap_[*vi] = vTypeMap[gm[*vi].type];
117 vTypeCount[gm[*vi].type] += 1;
118 }
119
120 LOG4CPLUS_DEBUG(fileLogger_, "\nVERTEX TYPE MAP"

123 << "\tCaller vertices: " << vTypeCount["CALR"] << endl
124 << "\tPSAP vertices: " << vTypeCount["PSAP"] << endl
125 << "\tResponder vertices: " << vTypeCount["RESP"] << endl);

127 LOG4CPLUS_INFO(fileLogger_, "Finished initializing vertex type map");

148edgeType Layout911::edgType(const int srcVertex, const int destVertex)
149{
150 if (vertexTypeMap_[srcVertex] == CALR && vertexTypeMap_[destVertex] == PSAP)
151 return CP;
152 else if (vertexTypeMap_[srcVertex] == PSAP && vertexTypeMap_[destVertex] == RESP)
153 return PR;

156 else if (vertexTypeMap_[srcVertex] == PSAP && vertexTypeMap_[destVertex] == PSAP)
157 return PP;
158 else if (vertexTypeMap_[srcVertex] == RESP && vertexTypeMap_[destVertex] == PSAP)
159 return RP;
160 else
161 return ETYPE_UNDEF;
162}
