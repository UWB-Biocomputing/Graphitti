42// Setup the internal structure of the class.
43void Layout911::setup()
44{
45 // Base class allocates memory for: xLoc_, yLoc, dist2_, and dist_
46 // so we call its method first
47 Layout::setup();
48
49 // Loop over all vertices and set their x and y locations
50 GraphManager<NG911VertexProperties>::VertexIterator vi, vi_end;
51 GraphManager<NG911VertexProperties> &gm = GraphManager<NG911VertexProperties>::getInstance();
52 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
53 assert(*vi < numVertices_);
54 xloc_[*vi] = gm[*vi].x;
55 yloc_[*vi] = gm[*vi].y;
56 }
57
58 // Now we cache the between each pair of vertices distances^2 into a matrix
59 for (int n = 0; n < numVertices_ - 1; n++) {
60 for (int n2 = n + 1; n2 < numVertices_; n2++) {
61 // distance^2 between two points in point-slope form
62 dist2_(n, n2) = (xloc_[n] - xloc_[n2]) * (xloc_[n] - xloc_[n2])
63 + (yloc_[n] - yloc_[n2]) * (yloc_[n] - yloc_[n2]);
64
65 // both points are equidistant from each other
66 dist2_(n2, n) = dist2_(n, n2);
67 }
68 }
69
70 // Finally take the square root to get the distances
71 dist_ = sqrt(dist2_);
72}

119// Returns the type of synapse at the given coordinates
120edgeType Layout911::edgType(int srcVertex, int destVertex)
121{
122 if (vertexTypeMap_[srcVertex] == vertexType::CALR
123 && vertexTypeMap_[destVertex] == vertexType::PSAP)
124 return edgeType::CP;
125 else if (vertexTypeMap_[srcVertex] == vertexType::PSAP
126 && (vertexTypeMap_[destVertex] == vertexType::LAW
127 || vertexTypeMap_[destVertex] == vertexType::FIRE
128 || vertexTypeMap_[destVertex] == vertexType::EMS))
129 return edgeType::PR;
130 else if (vertexTypeMap_[srcVertex] == vertexType::PSAP
131 && vertexTypeMap_[destVertex] == vertexType::CALR)
132 return edgeType::PC;
133 else if (vertexTypeMap_[srcVertex] == vertexType::PSAP
134 && vertexTypeMap_[destVertex] == vertexType::PSAP)
135 return edgeType::PP;
136 else if ((vertexTypeMap_[srcVertex] == vertexType::LAW
137 || vertexTypeMap_[destVertex] == vertexType::FIRE
138 || vertexTypeMap_[destVertex] == vertexType::EMS)
139 && vertexTypeMap_[destVertex] == vertexType::PSAP)
140 return edgeType::RP;
141 else if ((vertexTypeMap_[srcVertex] == vertexType::LAW
142 || vertexTypeMap_[destVertex] == vertexType::FIRE
143 || vertexTypeMap_[destVertex] == vertexType::EMS)
144 && vertexTypeMap_[destVertex] == vertexType::CALR)
145 return edgeType::RC;
146 else
147 return edgeType::ETYPE_UNDEF;
148}
