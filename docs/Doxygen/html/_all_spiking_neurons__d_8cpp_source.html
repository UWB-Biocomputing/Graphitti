66void AllSpikingNeurons::copyFromDevice(void *deviceAddress)
67{
68 int numVertices = Simulator::getInstance().getTotalVertices();
69
70 AllSpikingNeuronsDeviceProperties allVerticesDevice;
71 HANDLE_ERROR(cudaMemcpy(&allVerticesDevice, deviceAddress,
72 sizeof(AllSpikingNeuronsDeviceProperties), cudaMemcpyDeviceToHost));
73
74 bool cpu_has_fired[numVertices];
75 HANDLE_ERROR(cudaMemcpy(cpu_has_fired, allVerticesDevice.hasFired_, numVertices * sizeof(bool),
76 cudaMemcpyDeviceToHost));
77 for (int i = 0; i < numVertices; i++) {
78 hasFired_[i] = cpu_has_fired[i];
79 }
80
81 // We have to copy the whole state of the event buffer from GPU memory because
82 // we reset it in CPU code and then copy the new state back to the GPU.
83 int cpu_spike_count[numVertices];
84 HANDLE_ERROR(cudaMemcpy(cpu_spike_count, allVerticesDevice.numEventsInEpoch_,
85 numVertices * sizeof(int), cudaMemcpyDeviceToHost));
86 for (int i = 0; i < numVertices; i++) {
87 vertexEvents_[i].numEventsInEpoch_ = cpu_spike_count[i];
88 }
89
90 int queue_front[numVertices];
91 HANDLE_ERROR(cudaMemcpy(queue_front, allVerticesDevice.queueFront_, numVertices * sizeof(int),
92 cudaMemcpyDeviceToHost));
93 for (int i = 0; i < numVertices; i++) {
94 vertexEvents_[i].queueFront_ = queue_front[i];
95 }

97 int queue_end[numVertices];
98 HANDLE_ERROR(cudaMemcpy(queue_end, allVerticesDevice.queueEnd_, numVertices * sizeof(int),
99 cudaMemcpyDeviceToHost));
100 for (int i = 0; i < numVertices; i++) {
101 vertexEvents_[i].queueEnd_ = queue_end[i];
102 }
103
104 int epoch_start[numVertices];
105 HANDLE_ERROR(cudaMemcpy(epoch_start, allVerticesDevice.epochStart_, numVertices * sizeof(int),
106 cudaMemcpyDeviceToHost));

108 vertexEvents_[i].epochStart_ = epoch_start[i];
109 }

111 uint64_t *pSpikeHistory[numVertices];

113 numVertices * sizeof(uint64_t), cudaMemcpyDeviceToHost));
114
115 // All EventBuffers are of the same size,

117 int maxSpikes = vertexEvents_[0].eventTimeSteps_.size() - 1;
