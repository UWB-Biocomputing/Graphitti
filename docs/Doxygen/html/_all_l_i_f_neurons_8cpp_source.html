24void AllLIFNeurons::advanceNeuron(int index)
25{
26 BGFLOAT &Vm = this->Vm_[index];
27 BGFLOAT &Vthresh = this->Vthresh_[index];
28 BGFLOAT &summationPoint = this->summationPoints_[index];
29 BGFLOAT &I0 = this->I0_[index];
30 BGFLOAT &Inoise = this->Inoise_[index];
31 BGFLOAT &C1 = this->C1_[index];
32 BGFLOAT &C2 = this->C2_[index];
33 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
34
35 BGFLOAT noise;
36 if (nStepsInRefr > 0) {
37 // is neuron refractory?
38 #ifdef VALIDATION_MODE
39 noise = (*noiseRNG)();
40 //LOG4CPLUS_DEBUG(vertexLogger_, "neuron refractory LIF[" << index << "] :: Noise = " << noise);
41 #endif
42 --nStepsInRefr;
43 } else if (Vm >= Vthresh) {
44 // should it fire?
45 #ifdef VALIDATION_MODE
46 noise = (*noiseRNG)();
47 //LOG4CPLUS_DEBUG(vertexLogger_, "FIRE NEURON LIF[" << index << "] :: Noise = " << noise);
48 #endif
49 fire(index);
50 } else {
51 summationPoint += I0; // add IO
52 // add noise
53 noise = (*noiseRNG)();
54 #ifdef VALIDATION_MODE
55 //LOG4CPLUS_DEBUG(vertexLogger_, "ADVANCE NEURON LIF[" << index << "] :: Noise = " << noise);
56 #endif
57 summationPoint += noise * Inoise; // add noise
58 Vm = C1 * Vm + C2 * summationPoint; // decay Vm and add inputs
59 }
60
61 // Causes a huge slowdown since it's printed so frequently
62 #ifdef VALIDATION_MODE
63 LOG4CPLUS_DEBUG(vertexLogger_, endl
64 << "Advance Index[" << index << "] :: Noise = " << noise
65 << "\tVm: " << Vm << endl
66 << "\tsp = " << summationPoint << endl
67 << "\tInoise = " << Inoise << endl);
68 // LOG4CPLUS_DEBUG(vertexLogger_, "Index: " << index << " Vm: " << Vm);
69 // LOG4CPLUS_DEBUG(vertexLogger_, "NEURON[" << index << "] {" << endl
70 // << "\tVm = " << Vm << endl
71 // << "\tVthresh = " << Vthresh << endl
72 // << "\tsummationPoint = " << summationPoint << endl
73 // << "\tI0 = " << I0 << endl
74 // << "\tInoise = " << Inoise << endl
75 // << "\tC1 = " << C1 << endl
76 // << "\tC2 = " << C2 << endl
77 // << "}" << endl);
78 #endif
79
80 // clear synaptic input for next time step
81 summationPoint = 0;
82}
