109void Connections::createSynapsesFromWeights()
110{
111 int numVertices = Simulator::getInstance().getTotalVertices();
112 Layout &layout = *Simulator::getInstance().getModel()->getLayout();
113 AllVertices &vertices = *layout.getVertices();
114
115 // for each neuron
116 for (int i = 0; i < numVertices; i++) {
117 // for each synapse in the vertex
118 for (BGSIZE synapseIndex = 0; synapseIndex < Simulator::getInstance().getMaxEdgesPerVertex();
119 synapseIndex++) {
120 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * i + synapseIndex;
121 // if the synapse weight is not zero (which means there is a connection), create the synapse
122 if (edges_->W_[iEdg] != 0.0) {
123 BGFLOAT theW = edges_->W_[iEdg];
124 BGFLOAT *sumPoint = &(vertices.summationMap_[i]);
125 int srcVertex = edges_->sourceVertexIndex_[iEdg];
126 int destVertex = edges_->destVertexIndex_[iEdg];
127 edgeType type = layout.edgType(srcVertex, destVertex);
128 edges_->edgeCounts_[i]++;
129 edges_->createEdge(iEdg, srcVertex, destVertex, sumPoint,
130 Simulator::getInstance().getDeltaT(), type);
131 edges_->W_[iEdg] = theW;
132 }
133 }
134 }
