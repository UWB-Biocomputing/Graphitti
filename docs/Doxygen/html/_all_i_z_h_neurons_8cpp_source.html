218void AllIZHNeurons::advanceNeuron(const int index)
219{
220 BGFLOAT &Vm = this->Vm_[index];
221 BGFLOAT &Vthresh = this->Vthresh_[index];
222 BGFLOAT &summationPoint = this->summationMap_[index];
223 BGFLOAT &I0 = this->I0_[index];
224 BGFLOAT &Inoise = this->Inoise_[index];
225 BGFLOAT &C1 = this->C1_[index];
226 BGFLOAT &C2 = this->C2_[index];
227 BGFLOAT &C3 = this->C3_[index];
228 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
229
230 BGFLOAT &a = Aconst_[index];
231 BGFLOAT &b = Bconst_[index];
232 BGFLOAT &u = this->u_[index];
233
234 if (nStepsInRefr > 0) {
235 // is neuron refractory?
236 --nStepsInRefr;
237 } else if (Vm >= Vthresh) {
238 // should it fire?
239 fire(index);
240 } else {
241 summationPoint += I0; // add IO
242 // add noise
243 BGFLOAT noise = (*noiseRNG)();
244 // Happens really often, causes drastic slow down
245 // DEBUG_MID(cout << "ADVANCE NEURON[" << index << "] :: noise = " << noise << endl;)
246 summationPoint += noise * Inoise; // add noise
247
248 BGFLOAT Vint = Vm * 1000;
249
250 // Izhikevich model integration step
251 BGFLOAT Vb = Vint + C3 * (0.04 * Vint * Vint + 5 * Vint + 140 - u);
252 u = u + C3 * a * (b * Vint - u);
253
254 Vm = Vb * 0.001 + C2 * summationPoint; // add inputs
255 }
256
257 // Happens really often, causes drastic slow down
258 // DEBUG_MID(cout << index << " " << Vm << endl;)
259 // DEBUG_MID(cout << "NEURON[" << index << "] {" << endl
260 // << "\tVm = " << Vm << endl
261 // << "\ta = " << a << endl
262 // << "\tb = " << b << endl
263 // << "\tc = " << Cconst_[index] << endl
264 // << "\td = " << Dconst_[index] << endl
265 // << "\tu = " << u << endl
266 // << "\tVthresh = " << Vthresh << endl
267 // << "\tsummationPoint = " << summationPoint << endl
268 // << "\tI0 = " << I0 << endl
269 // << "\tInoise = " << Inoise << endl
270 // << "\tC1 = " << C1 << endl
271 // << "\tC2 = " << C2 << endl
272 // << "\tC3 = " << C3 << endl
273 // << "}" << endl;)
274
275 // clear synaptic input for next time step
276 summationPoint = 0;
277}

282void AllIZHNeurons::fire(const int index)
283{
284 const BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
285 AllSpikingNeurons::fire(index);
286
287 // calculate the number of steps in the absolute refractory period
288 BGFLOAT &Vm = this->Vm_[index];
289 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
290 BGFLOAT &Trefract = this->Trefract_[index];
291
292 BGFLOAT &c = Cconst_[index];
293 BGFLOAT &d = Dconst_[index];
294 BGFLOAT &u = this->u_[index];
295
296 nStepsInRefr = static_cast<int>(Trefract / deltaT + 0.5);
297
298 // reset to 'Vreset'
299 Vm = c * 0.001;
300 u = u + d;
301}
