210void AllIZHNeurons::advanceNeuron(int index)
211{
212 BGFLOAT &Vm = this->Vm_[index];
213 BGFLOAT &Vthresh = this->Vthresh_[index];
214 BGFLOAT &summationPoint = this->summationPoints_[index];
215 BGFLOAT &I0 = this->I0_[index];
216 BGFLOAT &Inoise = this->Inoise_[index];
217 BGFLOAT &C1 = this->C1_[index];
218 BGFLOAT &C2 = this->C2_[index];
219 BGFLOAT &C3 = this->C3_[index];
220 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
221
222 BGFLOAT &a = Aconst_[index];
223 BGFLOAT &b = Bconst_[index];
224 BGFLOAT &u = this->u_[index];
225
226 if (nStepsInRefr > 0) {
227 // is neuron refractory?
228 #ifdef VALIDATION_MODE
229 BGFLOAT noise = (*noiseRNG)();
230 LOG4CPLUS_DEBUG(vertexLogger_, "REFRACTORY NEURON IZH[" << index << "] :: Noise = " << noise);
231 #endif
232 --nStepsInRefr;
233 } else if (Vm >= Vthresh) {
234 // should it fire?
235 #ifdef VALIDATION_MODE
236 BGFLOAT noise = (*noiseRNG)();
237 LOG4CPLUS_DEBUG(vertexLogger_, "FIRE NEURON IZH[" << index << "] :: Noise = " << noise);
238 #endif
239 fire(index);
240 } else {
241 summationPoint += I0; // add IO
242 // add noise
243 BGFLOAT noise = (*noiseRNG)();
244 // Happens really often, causes drastic slow down
245 #ifdef VALIDATION_MODE
246 LOG4CPLUS_DEBUG(vertexLogger_, "ADVANCE NEURON IZH[" << index << "] :: Noise = " << noise);
247 #endif
248 summationPoint += noise * Inoise; // add noise
249
250 BGFLOAT Vint = Vm * 1000;
251
252 // Izhikevich model integration step
253 BGFLOAT Vb = Vint + C3 * (0.04 * Vint * Vint + 5 * Vint + 140 - u);
254 u = u + C3 * a * (b * Vint - u);
255
256 Vm = Vb * 0.001 + C2 * summationPoint; // add inputs
257 }
258
259 // Happens really often, causes drastic slow down
260 // DEBUG_MID(cout << index << " " << Vm << endl;)
261 // DEBUG_MID(cout << "NEURON[" << index << "] {" << endl
262 // << "\tVm = " << Vm << endl
263 // << "\ta = " << a << endl
264 // << "\tb = " << b << endl
265 // << "\tc = " << Cconst_[index] << endl
266 // << "\td = " << Dconst_[index] << endl
267 // << "\tu = " << u << endl
268 // << "\tVthresh = " << Vthresh << endl
269 // << "\tsummationPoint = " << summationPoint << endl
270 // << "\tI0 = " << I0 << endl
271 // << "\tInoise = " << Inoise << endl
272 // << "\tC1 = " << C1 << endl
273 // << "\tC2 = " << C2 << endl
274 // << "\tC3 = " << C3 << endl
275 // << "}" << endl;)
276
277 // clear synaptic input for next time step
278 summationPoint = 0;
279}

284void AllIZHNeurons::fire(int index)
285{
286 const BGFLOAT deltaT = Simulator::getInstance().getDeltaT();
287 AllSpikingNeurons::fire(index);
288
289 // calculate the number of steps in the absolute refractory period
290 BGFLOAT &Vm = this->Vm_[index];
291 int &nStepsInRefr = this->numStepsInRefractoryPeriod_[index];
292 BGFLOAT &Trefract = this->Trefract_[index];
293
294 BGFLOAT &c = Cconst_[index];
295 BGFLOAT &d = Dconst_[index];
296 BGFLOAT &u = this->u_[index];
297
298 nStepsInRefr = static_cast<int>(Trefract / deltaT + 0.5);
299
300 // reset to 'Vreset'
301 Vm = c * 0.001;
302 u = u + d;
303}
