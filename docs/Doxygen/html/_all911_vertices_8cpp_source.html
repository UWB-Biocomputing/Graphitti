30void All911Vertices::createAllVertices(Layout *layout)
31{
32 vector<int> psapList;
33 vector<int> respList;
34 psapList.clear();
35 respList.clear();
36
37 int callersPerZone[] = {0, 0, 0, 0};
38 int respPerZone[] = {0, 0, 0, 0};
39
40 Layout911 *layout911 = dynamic_cast<Layout911 *>(layout);
41
42 for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
43 // Create all callers
44 if (layout->vertexTypeMap_[i] == CALR) {
45 callNum_[i] = initRNG.inRange(callNumRange_[0], callNumRange_[1]);
46 callersPerZone[layout911->zone(i)] += callNum_[i];
47 }
48
49 // Find all PSAPs
50 if (layout->vertexTypeMap_[i] == PSAP) {
51 psapList.push_back(i);
52 }
53
54 // Find all resps
55 if (layout->vertexTypeMap_[i] == RESP) {
56 respList.push_back(i);
57 respPerZone[layout911->zone(i)] += 1;
58 }
59 }
60
61 // Create all psaps
62 // Dispatchers in a psap = [callers in the zone * k] + some randomness
63 for (int i = 0; i < psapList.size(); i++) {
64 int psapQ = layout911->zone(i);
65 int dispCount = (callersPerZone[psapQ] * dispNumScale_) + initRNG.inRange(-5, 5);
66 if (dispCount < 1) {
67 dispCount = 1;
68 }
69 dispNum_[psapList[i]] = dispCount;
70 }
71
72 // Create all responders
73 // Responders in a node = [callers in the zone * k]/[number of responder nodes] + some randomness
74 for (int i = 0; i < respList.size(); i++) {
75 int respQ = layout911->zone(respList[i]);
76 int respCount
77 = (callersPerZone[respQ] * respNumScale_) / respPerZone[respQ] + initRNG.inRange(-5, 5);
78 if (respCount < 1) {
79 respCount = 1;
80 }
81 respNum_[respList[i]] = respCount;
82 }
83}
