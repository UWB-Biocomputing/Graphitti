53// Creates all the Vertices and assigns initial data for them.
54void All911Vertices::createAllVertices(Layout &layout)
55{
56 // Calcualte the total number of time-steps for the data structures that
57 // will record per-step histories
58 Simulator &simulator = Simulator::getInstance();
59 uint64_t stepsPerEpoch = simulator.getEpochDuration() / simulator.getDeltaT();
60 uint64_t totalTimeSteps = stepsPerEpoch * simulator.getNumEpochs();
61 BGFLOAT epochDuration = simulator.getEpochDuration();
62 BGFLOAT deltaT = simulator.getDeltaT();
63
64 // Loop over all vertices and set the number of servers and trunks, and
65 // determine the size of the waiting queue.
66 // We get the information needed from the GraphManager.
67 GraphManager<NG911VertexProperties>::VertexIterator vi, vi_end;
68 GraphManager<NG911VertexProperties> &gm = GraphManager<NG911VertexProperties>::getInstance();
69 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
70 assert(*vi < size_);
71
72 if (gm[*vi].type == "CALR") {
73 vertexQueues_[*vi].resize(stepsPerEpoch);
74 } else {
75 numServers_[*vi] = gm[*vi].servers;
76 numTrunks_[*vi] = gm[*vi].trunks;
77 // We should not have more servers than trunks
78 assert(numServers_[*vi] <= numTrunks_[*vi]);
79
80 // The waiting queue is of size # trunks. We keep track of the # of busy servers
81 // to know when there are no more trunks available.
82 vertexQueues_[*vi].resize(numTrunks_[*vi]);
83
84 // Initialize the data structures for agent availability
85 servingCall_[*vi].resize(gm[*vi].servers);
86 answerTime_[*vi].resize(gm[*vi].servers);
87 serverCountdown_[*vi].assign(gm[*vi].servers, 0);
88
89 // Initialize the data structures for system metrics
90 queueLengthHistory_[*vi].assign(totalTimeSteps, 0);
91 utilizationHistory_[*vi].assign(totalTimeSteps, 0);
92 }
93 }
94
95 // Read Input Events using the InputManager
96 inputManager_.readInputs();
97}

204// Take calls from the edges and transfer them to the vertex if it's queue is not full
205void All911Vertices::integrateVertexInputs(AllEdges &edges, EdgeIndexMap &edgeIndexMap)
206{
207 Simulator &simulator = Simulator::getInstance();
208 All911Edges &all911Edges = dynamic_cast<All911Edges &>(edges);
209
210 for (int vertex = 0; vertex < simulator.getTotalVertices(); ++vertex) {
211 int start = edgeIndexMap.incomingEdgeBegin_[vertex];
212 int count = edgeIndexMap.incomingEdgeCount_[vertex];
213
214 if (simulator.getModel().getLayout().vertexTypeMap_[vertex] == vertexType::CALR) {
215 continue; // TODO911: Caller Regions will have different behaviour
216 }
217
218 // Loop over all the edges and pull the data in
219 for (int edge = start; edge < start + count; ++edge) {
220 int edgeIdx = edgeIndexMap.incomingEdgeIndexMap_[edge];
221
222 if (!all911Edges.inUse_[edgeIdx]) {
223 continue;
224 } // Edge isn't in use
225 if (all911Edges.isAvailable_[edgeIdx]) {
226 continue;
227 } // Edge doesn't have a call
228
229 int dst = all911Edges.destVertexIndex_[edgeIdx];
230 // The destination vertex should be the one pulling the information
231 assert(dst == vertex);
232
233 CircularBuffer<Call> &dstQueue = getQueue(dst);
234 if (dstQueue.size() >= (dstQueue.capacity() - busyServers(dst))) {
235 // Call is dropped because there is no space in the waiting queue
236 if (!all911Edges.isRedial_[edgeIdx]) {
237 // Only count the dropped call if it's not a redial
238 droppedCalls(dst)++;
239 // Record that we received a call
240 receivedCalls(dst)++;
241 LOG4CPLUS_DEBUG(vertexLogger_,
242 "Call dropped: " << droppedCalls(dst)
243 << ", time: " << all911Edges.call_[edgeIdx].time
244 << ", vertex: " << dst
245 << ", queue size: " << dstQueue.size());
246 }
247 } else {
248 // Transfer call to destination
249 dstQueue.put(all911Edges.call_[edgeIdx]);
250 // Record that we received a call
251 receivedCalls(dst)++;
252 all911Edges.isAvailable_[edgeIdx] = true;
253 all911Edges.isRedial_[edgeIdx] = false;
254 }
255 }
256 }

259// Update internal state of the indexed vertex (called by every simulation step).
260void All911Vertices::advanceVertices(AllEdges &edges, const EdgeIndexMap &edgeIndexMap)
261{
262 Simulator &simulator = Simulator::getInstance();
263 Layout &layout = simulator.getModel().getLayout();
264 uint64_t endEpochStep
265 = g_simulationStep
266 + static_cast<uint64_t>(simulator.getEpochDuration() / simulator.getDeltaT());
267
268 All911Edges &edges911 = dynamic_cast<All911Edges &>(edges);
269
270 // Advance vertices
271 for (int vertex = 0; vertex < simulator.getTotalVertices(); ++vertex) {
272 if (layout.vertexTypeMap_[vertex] == vertexType::CALR) {
273 advanceCALR(vertex, edges911, edgeIndexMap);
274 } else if (layout.vertexTypeMap_[vertex] == vertexType::PSAP) {
275 advancePSAP(vertex, edges911, edgeIndexMap);
276 } else if (layout.vertexTypeMap_[vertex] == vertexType::EMS
277 || layout.vertexTypeMap_[vertex] == vertexType::FIRE
278 || layout.vertexTypeMap_[vertex] == vertexType::LAW) {
279 advanceRESP(vertex, edges911, edgeIndexMap);
280 }
281 }

285// Advance a CALR vertex. Send calls to the appropriate PSAP
286void All911Vertices::advanceCALR(BGSIZE vertexIdx, All911Edges &edges911,
287 const EdgeIndexMap &edgeIndexMap)
288{
289 // There is only one outgoing edge from CALR to a PSAP
290 BGSIZE start = edgeIndexMap.outgoingEdgeBegin_[vertexIdx];
291 BGSIZE edgeIdx = edgeIndexMap.outgoingEdgeIndexMap_[start];
292
293 // Check for dropped calls, indicated by the edge not being available
294 if (!edges911.isAvailable_[edgeIdx]) {
295 // If the call is still there, it means that there was no space in the PSAP's waiting
296 // queue. Therefore, this is a dropped call.
297 // If readialing, we assume that it happens immediately and the caller tries until
298 // getting through.
299 if (!edges911.isRedial_[edgeIdx] && initRNG.randDblExc() >= redialP_) {
300 // We only make the edge available if no readialing occurs.
301 edges911.isAvailable_[edgeIdx] = true;
302 LOG4CPLUS_DEBUG(vertexLogger_, "Did not redial at time: " << edges911.call_[edgeIdx].time);
303 } else {
304 // Keep the edge unavailable but mark it as a redial
305 edges911.isRedial_[edgeIdx] = true;
306 }
307 }
308
309 // peek at the next call in the queue
310 optional<Call> nextCall = vertexQueues_[vertexIdx].peek();
311 if (edges911.isAvailable_[edgeIdx] && nextCall && nextCall->time <= g_simulationStep) {
312 // Calls that start at the same time are process in the order they appear.
313 // The call starts at the current time step so we need to pop it and process it
314 vertexQueues_[vertexIdx].get(); // pop from the queue
315
316 // Place new call in the edge going to the PSAP
317 assert(edges911.isAvailable_[edgeIdx]);
318 edges911.call_[edgeIdx] = nextCall.value();
319 edges911.isAvailable_[edgeIdx] = false;
320 LOG4CPLUS_DEBUG(vertexLogger_, "Calling PSAP at time: " << nextCall->time);
321 }
322}

325// Advance a PSAP vertex. Controls the redirection and handling of calls.
326void All911Vertices::advancePSAP(BGSIZE vertexIdx, All911Edges &edges911,
327 const EdgeIndexMap &edgeIndexMap)
328{
329 // Loop over all servers and free the ones finishing serving calls
330 vector<int> availableServers;
331 for (size_t server = 0; server < serverCountdown_[vertexIdx].size(); ++server) {
332 if (serverCountdown_[vertexIdx][server] == 0) {
333 // Server is available to take calls. This check is needed because calls
334 // could have duration of zero or server has not been assigned a call yet
335 availableServers.push_back(server);
336 } else if (--serverCountdown_[vertexIdx][server] == 0) {
337 // Server becomes free to take calls
338 // TODO: What about wrap-up time?
339 Call &endingCall = servingCall_[vertexIdx][server];
340
341 //Store call metrics
342 wasAbandonedHistory_[vertexIdx].push_back(false);
343 beginTimeHistory_[vertexIdx].push_back(endingCall.time);
344 answerTimeHistory_[vertexIdx].push_back(answerTime_[vertexIdx][server]);
345 endTimeHistory_[vertexIdx].push_back(g_simulationStep);
346 LOG4CPLUS_DEBUG(vertexLogger_,
347 "Finishing call, begin time: "
348 << endingCall.time << ", end time: " << g_simulationStep
349 << ", waited: " << answerTime_[vertexIdx][server] - endingCall.time);
350
351 // Dispatch the Responder closest to the emergency location.
352 Connections911 &conn911
353 = dynamic_cast<Connections911 &>(Simulator::getInstance().getModel().getConnections());
354 BGSIZE respEdge = conn911.getEdgeToClosestResponder(endingCall, vertexIdx);
355 BGSIZE responder = edges911.destVertexIndex_[respEdge];
356 LOG4CPLUS_DEBUG(vertexLogger_, "Dispatching Responder: " << responder);
357
358 // Place the call in the edge going to the responder
359 // Call becomes a dispatch order at this time
360 endingCall.time = g_simulationStep;
361 edges911.call_[respEdge] = endingCall;
362 edges911.isAvailable_[respEdge] = false;
363
364 // This assumes that the caller doesn't stay in the line until the responder
365 // arrives on scene. This not true in all instances.
366 availableServers.push_back(server);
367 }
368 }
369
370 // Assign calls to servers until either no servers are available or
371 // there are no more calls in the waiting queue
372 size_t serverId = 0;
373 while (serverId < availableServers.size() && !vertexQueues_[vertexIdx].isEmpty()) {
374 // TODO: calls with duration of zero are being added but because countdown will be zero
375 // they don't show up in the logs
376 optional<Call> call = vertexQueues_[vertexIdx].get();
377 assert(call);
378
379 if (call->patience < (g_simulationStep - call->time)) {
380 // If the patience time is less than the waiting time, the call is abandoned
381 wasAbandonedHistory_[vertexIdx].push_back(true);
382 beginTimeHistory_[vertexIdx].push_back(call->time);
383 // Answer time and end time get zero as sentinel for non-valid values
384 answerTimeHistory_[vertexIdx].push_back(0);
385 endTimeHistory_[vertexIdx].push_back(0);
386 LOG4CPLUS_DEBUG(vertexLogger_, "Call was abandoned, Patience: "
387 << call->patience
388 << " Ring Time: " << g_simulationStep - call->time);
389 } else {
390 // The available server starts serving the call
391 int availServer = availableServers[serverId];
392 servingCall_[vertexIdx][availServer] = call.value();
393 answerTime_[vertexIdx][availServer] = g_simulationStep;
394 serverCountdown_[vertexIdx][availServer] = call.value().duration;
395 LOG4CPLUS_DEBUG(vertexLogger_, "Serving Call starting at time: "
396 << call->time << ", sim-step: " << g_simulationStep);
397 // Next server
398 ++serverId;
399 }
400 }
401
402 // Update number of busy servers. This is used to check if there is space in the queue
403 busyServers_[vertexIdx] = numServers_[vertexIdx] - availableServers.size();
404
405 // Update queueLength and utilization histories
406 queueLengthHistory_[vertexIdx].resize(g_simulationStep + 1);
407 queueLengthHistory_[vertexIdx][g_simulationStep] = vertexQueues_[vertexIdx].size();
408 utilizationHistory_[vertexIdx].resize(g_simulationStep + 1);
409 utilizationHistory_[vertexIdx][g_simulationStep]
410 = static_cast<double>(busyServers_[vertexIdx]) / numServers_[vertexIdx];
411}

414// Advance a RESP vertex. Receives call from PSAP and responds to the emergency events
415void All911Vertices::advanceRESP(BGSIZE vertexIdx, All911Edges &edges911,
416 const EdgeIndexMap &edgeIndexMap)
417{
418 Layout &layout = Simulator::getInstance().getModel().getLayout();
419
420 // Free the units finishing up with emergency responses
421 vector<int> availableUnits;
422 for (size_t unit = 0; unit < serverCountdown_[vertexIdx].size(); ++unit) {
423 if (serverCountdown_[vertexIdx][unit] == 0) {
424 // Unit is available
425 availableUnits.push_back(unit);
426 } else if (--serverCountdown_[vertexIdx][unit] == 0) {
427 // Unit becomes available to responde to new incidents
428 Call &endingIncident = servingCall_[vertexIdx][unit];
429
430 //Store incident response metrics
431 wasAbandonedHistory_[vertexIdx].push_back(false);
432 beginTimeHistory_[vertexIdx].push_back(endingIncident.time);
433 answerTimeHistory_[vertexIdx].push_back(answerTime_[vertexIdx][unit]);
434 endTimeHistory_[vertexIdx].push_back(g_simulationStep);
435 LOG4CPLUS_DEBUG(vertexLogger_,
436 "Finishing response, begin time: "
437 << endingIncident.time << ", end time: " << g_simulationStep
438 << ", waited: " << answerTime_[vertexIdx][unit] - endingIncident.time);
439
440 // Unit is added to available units
441 availableUnits.push_back(unit);
442 }
443 }
444
445
446 // Assign reponse dispatches until no units are available or there are no more
447 // incidents in the waiting queue
448 for (size_t unit = 0; unit < availableUnits.size() && !vertexQueues_[vertexIdx].isEmpty();
449 ++unit) {
450 optional<Call> incident = vertexQueues_[vertexIdx].get();
451 assert(incident); // Safety check for valid incidents
452
453 // The available unit starts serving the call
454 int availUnit = availableUnits[unit];
455 servingCall_[vertexIdx][availUnit] = incident.value();
456 answerTime_[vertexIdx][availUnit] = g_simulationStep;
457
458 // We need to calculate the distance in miles but the x and y coordinates
459 // represent, respectively, degrees of longitude and latitude.
460 // One degree of latitude is aproximately 69 miles regardles of the location. However,
461 // a degree of longitude varies, being 69.172 miles at the equator and gradually shrinking
462 // to zero at the poles.
463 // One degree of longitude can be converted to miles using the following formula:
464 // 1 degree of longitude = cos(latitude) * 69.172
465 double lngDegreeLength = cos(layout.yloc_[vertexIdx] * (pi / 180)) * 69.172;
466 double latDegreeLength = 69.0;
467 double deltaLng = incident->x - layout.xloc_[vertexIdx];
468 double deltaLat = incident->y - layout.yloc_[vertexIdx];
469 double dist2incident
470 = sqrt(pow(deltaLng * lngDegreeLength, 2) + pow(deltaLat * latDegreeLength, 2));
471
472 // Calculate the driving time to the incident in seconds
473 double driveTime = (dist2incident / avgDrivingSpeed_) * 3600;
474 serverCountdown_[vertexIdx][availUnit] = driveTime + incident->onSiteTime;
475
476 serverCountdown_[vertexIdx][availUnit] = incident.value().duration;
477 LOG4CPLUS_DEBUG(vertexLogger_, "Response, driving time: " << driveTime << ", On-site time: "
478 << incident->onSiteTime);
479 }
480
481 // Update number of busy servers. This is used to check if there is space in the queue
482 busyServers_[vertexIdx] = numServers_[vertexIdx] - availableUnits.size();
483
484 // Update queueLength and utilization histories
485 queueLengthHistory_[vertexIdx].resize(g_simulationStep + 1);
486 queueLengthHistory_[vertexIdx][g_simulationStep] = vertexQueues_[vertexIdx].size();
487 utilizationHistory_[vertexIdx].resize(g_simulationStep + 1);
488 utilizationHistory_[vertexIdx][g_simulationStep]
489 = static_cast<double>(busyServers_[vertexIdx]) / numServers_[vertexIdx];
490}
