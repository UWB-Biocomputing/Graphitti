58  for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {

65  // Find all PSAPs

67  psapList.push_back(i);
68  }
69 
70  // Find all resps
71  if(layout->vertexTypeMap_[i] == RESP) {
72  respList.push_back(i);
73  respPerZone[layout911->zone(i)] += 1;
74  }
75  }
76 
77  // Create all psaps
78  // Dispatchers in a psap = [callers in the zone * k] + some randomness
79  for (int i = 0; i < psapList.size(); i++) {
80  int psapQ = layout911->zone(i);
81  int dispCount = (callersPerZone[psapQ] * dispNumScale_) + rng.inRange(-5, 5);
82  if (dispCount < 1) { dispCount = 1; }
83  dispNum_[psapList[i]] = dispCount;
84  }
85 
86  // Create all responders

88  for (int i = 0; i < respList.size(); i++) {
89  int respQ = layout911->zone(respList[i]);
90  int respCount = (callersPerZone[respQ] * respNumScale_)/respPerZone[respQ] + rng.inRange(-5, 5);
91  if (respCount < 1) { respCount = 1; }
92  respNum_[respList[i]] = respCount;
93  }
