53// Creates all the Vertices and assigns initial data for them.
54void All911Vertices::createAllVertices(Layout &layout)
55{
56 // Calcualte the total number of time-steps for the data structures that
57 // will record per-step histories
58 Simulator &simulator = Simulator::getInstance();
59 uint64_t stepsPerEpoch = simulator.getEpochDuration() / simulator.getDeltaT();
60 uint64_t totalTimeSteps = stepsPerEpoch * simulator.getNumEpochs();
61 BGFLOAT epochDuration = simulator.getEpochDuration();
62 BGFLOAT deltaT = simulator.getDeltaT();
63
64 // Loop over all vertices and set the number of servers and trunks, and
65 // determine the size of the waiting queue.
66 // We get the information needed from the GraphManager.
67 GraphManager::VertexIterator vi, vi_end;
68 GraphManager &gm = GraphManager::getInstance();
69 for (boost::tie(vi, vi_end) = gm.vertices(); vi != vi_end; ++vi) {
70 assert(*vi < size_);
71
72 if (gm[*vi].type == "CALR") {
73 vertexQueues_[*vi].resize(stepsPerEpoch);
74 } else {
75 numServers_[*vi] = gm[*vi].servers;
76 numTrunks_[*vi] = gm[*vi].trunks;
77 // We should not have more servers than trunks
78 assert(numServers_[*vi] <= numTrunks_[*vi]);
79
80 // The waiting queue is of size # trunks. We keep track of the # of busy servers
81 // to know when there are no more trunks available.
82 vertexQueues_[*vi].resize(numTrunks_[*vi]);
83
84 // Initialize the data structures for agent availability
85 servingCall_[*vi].resize(gm[*vi].servers);
86 answerTime_[*vi].resize(gm[*vi].servers);
87 serverCountdown_[*vi].assign(gm[*vi].servers, 0);
88
89 // Initialize the data structures for system metrics
90 queueLengthHistory_[*vi].assign(totalTimeSteps, 0);
91 utilizationHistory_[*vi].assign(totalTimeSteps, 0);
92 }
93 }
94
95 // Read Input Events using the InputManager
96 inputManager_.readInputs();
97}

164// Update internal state of the indexed vertex (called by every simulation step).
165void All911Vertices::advanceVertices(AllEdges &edges, const EdgeIndexMap &edgeIndexMap)
166{
167 Simulator &simulator = Simulator::getInstance();
168 Layout &layout = simulator.getModel().getLayout();
169 uint64_t endEpochStep
170 = g_simulationStep
171 + static_cast<uint64_t>(simulator.getEpochDuration() / simulator.getDeltaT());
172
173 All911Edges &edges911 = dynamic_cast<All911Edges &>(edges);
174
175 // Advance vertices
176 for (int vertex = 0; vertex < simulator.getTotalVertices(); ++vertex) {
177 if (layout.vertexTypeMap_[vertex] == CALR) {
178 advanceCALR(vertex, edges911, edgeIndexMap);
179 } else if (layout.vertexTypeMap_[vertex] == PSAP) {
180 advancePSAP(vertex, edges911, edgeIndexMap);
181 } else if (layout.vertexTypeMap_[vertex] == EMS || layout.vertexTypeMap_[vertex] == FIRE
182 || layout.vertexTypeMap_[vertex] == LAW) {
183 advanceRESP(vertex, edges911, edgeIndexMap);
184 }
185 }
186}

189// Advance a CALR vertex. Send calls to the appropriate PSAP
190void All911Vertices::advanceCALR(BGSIZE vertexIdx, All911Edges &edges911,
191 const EdgeIndexMap &edgeIndexMap)
192{
193 // There is only one outgoing edge from CALR to a PSAP
194 BGSIZE start = edgeIndexMap.outgoingEdgeBegin_[vertexIdx];
195 BGSIZE edgeIdx = edgeIndexMap.outgoingEdgeIndexMap_[start];
196
197 // Check for dropped calls, indicated by the edge not being available
198 if (!edges911.isAvailable_[edgeIdx]) {
199 // If the call is still there, it means that there was no space in the PSAP's waiting
200 // queue. Therefore, this is a dropped call.
201 // If readialing, we assume that it happens immediately and the caller tries until
202 // getting through.
203 if (!edges911.isRedial_[edgeIdx] && initRNG.randDblExc() >= redialP_) {
204 // We only make the edge available if no readialing occurs.
205 edges911.isAvailable_[edgeIdx] = true;
206 LOG4CPLUS_DEBUG(vertexLogger_, "Did not redial at time: " << edges911.call_[edgeIdx].time);
207 } else {
208 // Keep the edge unavailable but mark it as a redial
209 edges911.isRedial_[edgeIdx] = true;
210 }
211 }
212
213 // peek at the next call in the queue
214 optional<Call> nextCall = vertexQueues_[vertexIdx].peek();
215 if (edges911.isAvailable_[edgeIdx] && nextCall && nextCall->time <= g_simulationStep) {
216 // Calls that start at the same time are process in the order they appear.
217 // The call starts at the current time step so we need to pop it and process it
218 vertexQueues_[vertexIdx].get(); // pop from the queue
219
220 // Place new call in the edge going to the PSAP
221 assert(edges911.isAvailable_[edgeIdx]);
222 edges911.call_[edgeIdx] = nextCall.value();
223 edges911.isAvailable_[edgeIdx] = false;
224 LOG4CPLUS_DEBUG(vertexLogger_, "Calling PSAP at time: " << nextCall->time);
225 }
226}

229// Advance a PSAP vertex. Controls the redirection and handling of calls.
230void All911Vertices::advancePSAP(BGSIZE vertexIdx, All911Edges &edges911,
231 const EdgeIndexMap &edgeIndexMap)
232{
233 // Loop over all servers and free the ones finishing serving calls
234 vector<int> availableServers;
235 for (size_t server = 0; server < serverCountdown_[vertexIdx].size(); ++server) {
236 if (serverCountdown_[vertexIdx][server] == 0) {
237 // Server is available to take calls. This check is needed because calls
238 // could have duration of zero or server has not been assigned a call yet
239 availableServers.push_back(server);
240 } else if (--serverCountdown_[vertexIdx][server] == 0) {
241 // Server becomes free to take calls
242 // TODO: What about wrap-up time?
243 Call &endingCall = servingCall_[vertexIdx][server];
244
245 //Store call metrics
246 wasAbandonedHistory_[vertexIdx].push_back(false);
247 beginTimeHistory_[vertexIdx].push_back(endingCall.time);
248 answerTimeHistory_[vertexIdx].push_back(answerTime_[vertexIdx][server]);
249 endTimeHistory_[vertexIdx].push_back(g_simulationStep);
250 LOG4CPLUS_DEBUG(vertexLogger_,
251 "Finishing call, begin time: "
252 << endingCall.time << ", end time: " << g_simulationStep
253 << ", waited: " << answerTime_[vertexIdx][server] - endingCall.time);
254
255 // Dispatch the Responder closest to the emergency location.
256 Connections911 &conn911
257 = dynamic_cast<Connections911 &>(Simulator::getInstance().getModel().getConnections());
258 BGSIZE respEdge = conn911.getEdgeToClosestResponder(endingCall, vertexIdx);
259 BGSIZE responder = edges911.destVertexIndex_[respEdge];
260 LOG4CPLUS_DEBUG(vertexLogger_, "Dispatching Responder: " << responder);
261
262 // Place the call in the edge going to the responder
263 // Call becomes a dispatch order at this time
264 endingCall.time = g_simulationStep;
265 edges911.call_[respEdge] = endingCall;
266 edges911.isAvailable_[respEdge] = false;
267
268 // This assumes that the caller doesn't stay in the line until the responder
269 // arrives on scene. This not true in all instances.
270 availableServers.push_back(server);
271 }
272 }
273
274 // Assign calls to servers until either no servers are available or
275 // there are no more calls in the waiting queue
276 size_t serverId = 0;
277 while (serverId < availableServers.size() && !vertexQueues_[vertexIdx].isEmpty()) {
278 // TODO: calls with duration of zero are being added but because countdown will be zero
279 // they don't show up in the logs
280 optional<Call> call = vertexQueues_[vertexIdx].get();
281 assert(call);
282
283 if (call->patience < (g_simulationStep - call->time)) {
284 // If the patience time is less than the waiting time, the call is abandoned
285 wasAbandonedHistory_[vertexIdx].push_back(true);
286 beginTimeHistory_[vertexIdx].push_back(call->time);
287 // Answer time and end time get zero as sentinel for non-valid values
288 answerTimeHistory_[vertexIdx].push_back(0);
289 endTimeHistory_[vertexIdx].push_back(0);
290 LOG4CPLUS_DEBUG(vertexLogger_, "Call was abandoned, Patience: "
291 << call->patience
292 << " Ring Time: " << g_simulationStep - call->time);
293 } else {
294 // The available server starts serving the call
295 int availServer = availableServers[serverId];
296 servingCall_[vertexIdx][availServer] = call.value();
297 answerTime_[vertexIdx][availServer] = g_simulationStep;
298 serverCountdown_[vertexIdx][availServer] = call.value().duration;
299 LOG4CPLUS_DEBUG(vertexLogger_, "Serving Call starting at time: "
300 << call->time << ", sim-step: " << g_simulationStep);
301 // Next server
302 ++serverId;
303 }
304 }
305
306 // Update number of busy servers. This is used to check if there is space in the queue
307 busyServers_[vertexIdx] = numServers_[vertexIdx] - availableServers.size();
308
309 // Update queueLength and utilization histories
310 queueLengthHistory_[vertexIdx][g_simulationStep] = vertexQueues_[vertexIdx].size();
311 utilizationHistory_[vertexIdx][g_simulationStep]
312 = static_cast<double>(busyServers_[vertexIdx]) / numServers_[vertexIdx];
313}

316// Advance a RESP vertex. Receives call from PSAP and responds to the emergency events
317void All911Vertices::advanceRESP(BGSIZE vertexIdx, All911Edges &edges911,
318 const EdgeIndexMap &edgeIndexMap)
319{
320 Layout &layout = Simulator::getInstance().getModel().getLayout();
321
322 // Free the units finishing up with emergency responses
323 vector<int> availableUnits;
324 for (size_t unit = 0; unit < serverCountdown_[vertexIdx].size(); ++unit) {
325 if (serverCountdown_[vertexIdx][unit] == 0) {
326 // Unit is available
327 availableUnits.push_back(unit);
328 } else if (--serverCountdown_[vertexIdx][unit] == 0) {
329 // Unit becomes available to responde to new incidents
330 Call &endingIncident = servingCall_[vertexIdx][unit];
331
332 //Store incident response metrics
333 wasAbandonedHistory_[vertexIdx].push_back(false);
334 beginTimeHistory_[vertexIdx].push_back(endingIncident.time);
335 answerTimeHistory_[vertexIdx].push_back(answerTime_[vertexIdx][unit]);
336 endTimeHistory_[vertexIdx].push_back(g_simulationStep);
337 LOG4CPLUS_DEBUG(vertexLogger_,
338 "Finishing response, begin time: "
339 << endingIncident.time << ", end time: " << g_simulationStep
340 << ", waited: " << answerTime_[vertexIdx][unit] - endingIncident.time);
341
342 // Unit is added to available units
343 availableUnits.push_back(unit);
344 }
345 }
346
347
348 // Assign reponse dispatches until no units are available or there are no more
349 // incidents in the waiting queue
350 for (size_t unit = 0; unit < availableUnits.size() && !vertexQueues_[vertexIdx].isEmpty();
351 ++unit) {
352 optional<Call> incident = vertexQueues_[vertexIdx].get();
353 assert(incident); // Safety check for valid incidents
354
355 // The available unit starts serving the call
356 int availUnit = availableUnits[unit];
357 servingCall_[vertexIdx][availUnit] = incident.value();
358 answerTime_[vertexIdx][availUnit] = g_simulationStep;
359
360 // We need to calculate the distance in miles but the x and y coordinates
361 // represent, respectively, degrees of longitude and latitude.
362 // One degree of latitude is aproximately 69 miles regardles of the location. However,
363 // a degree of longitude varies, being 69.172 miles at the equator and gradually shrinking
364 // to zero at the poles.
365 // One degree of longitude can be converted to miles using the following formula:
366 // 1 degree of longitude = cos(latitude) * 69.172
367 double lngDegreeLength = cos(layout.yloc_[vertexIdx] * (pi / 180)) * 69.172;
368 double latDegreeLength = 69.0;
369 double deltaLng = incident->x - layout.xloc_[vertexIdx];
370 double deltaLat = incident->y - layout.yloc_[vertexIdx];
371 double dist2incident
372 = sqrt(pow(deltaLng * lngDegreeLength, 2) + pow(deltaLat * latDegreeLength, 2));
373
374 // Calculate the driving time to the incident in seconds
375 double driveTime = (dist2incident / avgDrivingSpeed_) * 3600;
376 serverCountdown_[vertexIdx][availUnit] = driveTime + incident->onSiteTime;
377
378 serverCountdown_[vertexIdx][availUnit] = incident.value().duration;
379 LOG4CPLUS_DEBUG(vertexLogger_, "Response, driving time: " << driveTime << ", On-site time: "
380 << incident->onSiteTime);
381 }
382
383 // Update number of busy servers. This is used to check if there is space in the queue
384 busyServers_[vertexIdx] = numServers_[vertexIdx] - availableUnits.size();
385
386 // Update queueLength and utilization histories
387 queueLengthHistory_[vertexIdx][g_simulationStep] = vertexQueues_[vertexIdx].size();
388 utilizationHistory_[vertexIdx][g_simulationStep]
389 = static_cast<double>(busyServers_[vertexIdx]) / numServers_[vertexIdx];
390}
