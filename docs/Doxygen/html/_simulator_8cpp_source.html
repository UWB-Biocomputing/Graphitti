148void Simulator::simulate()
149{
150 // Main simulation loop - execute maxGrowthSteps
151 for (int currentEpoch = 1; currentEpoch <= numEpochs_; currentEpoch++) {
152 LOG4CPLUS_TRACE(consoleLogger_, "Performing epoch number: " << currentEpoch);
153 LOG4CPLUS_TRACE(fileLogger_, "Begin network state:");
154 currentEpoch_ = currentEpoch;
155#ifdef PERFORMANCE_METRICS
156 // Start timer for advance
157 short_timer.start();
158#endif
159 // Advance simulation to next growth cycle
160 advanceEpoch(currentEpoch);
161#ifdef PERFORMANCE_METRICS
162 // Time to advance
163 t_host_advance += short_timer.lap() / 1000000.0;
164#endif
165 LOG4CPLUS_TRACE(consoleLogger_,
166 "done with epoch cycle " << currentEpoch_ << ", beginning growth update");
167 LOG4CPLUS_TRACE(edgeLogger_, "Epoch: " << currentEpoch_);
168
169#ifdef PERFORMANCE_METRICS
170 // Start timer for connection update
171 short_timer.start();
172#endif
173 // Update the neuron network
174 model_->updateConnections();
175 model_->updateHistory();
176
177#ifdef PERFORMANCE_METRICS
178 // Times converted from microseconds to seconds
179 // Time to update synapses
180 t_host_adjustEdges += short_timer.lap() / 1000000.0;
181 // Time since start of simulation
182 double total_time = timer.lap() / 1000000.0;
183
184 cout << "\ntotal_time: " << total_time << " seconds" << endl;
185 printPerformanceMetrics(total_time, currentEpoch);
186 cout << endl;
187#endif
188 }
189 LOG4CPLUS_TRACE(workbenchLogger_, "Complete");
190}

195void Simulator::advanceEpoch(const int &currentEpoch) const
196{
197 uint64_t count = 0;
198 // Compute step number at end of this simulation epoch
199 uint64_t endStep = g_simulationStep + static_cast<uint64_t>(epochDuration_ / deltaT_);
200 // DEBUG_MID(model->logSimStep();) // Generic model debug call
201 while (g_simulationStep < endStep) {
202 // Output status once every 10,000 steps
203 if (count % 10000 == 0) {
204 LOG4CPLUS_TRACE(consoleLogger_,
205 "Epoch: " << currentEpoch << "/" << numEpochs_
206 << " simulating time: " << g_simulationStep * deltaT_ << "/"
207 << (epochDuration_ * numEpochs_) - 1);
208 LOG4CPLUS_TRACE(workbenchLogger_,
209 "Epoch: " << currentEpoch << "/" << numEpochs_
210 << " simulating time: " << g_simulationStep * deltaT_ << "/"
211 << (epochDuration_ * numEpochs_) - 1);
212 count = 0;
213 }
214 count++;
215 // input stimulus
216 /***** S_INPUT NOT IN REPO YET *******/
217 // if (pInput != nullptr)
218 // pInput->inputStimulus();
219 // Advance the Network one time step
220 model_->advance();
221 g_simulationStep++;
222 }
