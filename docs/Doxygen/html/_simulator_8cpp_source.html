139void Simulator::simulate()
140{
141 // Main simulation loop - execute maxGrowthSteps
142 for (currentEpoch_ = 1; currentEpoch_ <= numEpochs_; currentEpoch_++) {
143 LOG4CPLUS_TRACE(consoleLogger_, "Performing epoch number: " << currentEpoch_);
144 LOG4CPLUS_TRACE(fileLogger_, "Begin network state:");
145
146#ifdef PERFORMANCE_METRICS
147 // Start timer for advance
148 short_timer.start();
149#endif
150 // Advance simulation to next growth cycle
151 advanceEpoch(currentEpoch_);
152#ifdef PERFORMANCE_METRICS
153 // Time to advance
154 t_host_advance += short_timer.lap() / 1000000.0;
155#endif
156 LOG4CPLUS_TRACE(consoleLogger_,
157 "done with epoch cycle " << currentEpoch_ << ", beginning growth update");
158 LOG4CPLUS_TRACE(edgeLogger_, "Epoch: " << currentEpoch_);
159
160#ifdef PERFORMANCE_METRICS
161 // Start timer for connection update
162 short_timer.start();
163#endif
164 // Update the neuron network
165 model_->updateConnections();
166 model_->updateHistory();
167
168#ifdef PERFORMANCE_METRICS
169 // Times converted from microseconds to seconds
170 // Time to update synapses
171 t_host_adjustEdges += short_timer.lap() / 1000000.0;
172 // Time since start of simulation
173 double total_time = timer.lap() / 1000000.0;
174
175 cout << "\ntotal_time: " << total_time << " seconds" << endl;
176 printPerformanceMetrics(total_time, currentEpoch);
177 cout << endl;
178#endif
179 }
180 LOG4CPLUS_TRACE(workbenchLogger_, "Complete");
181}

186void Simulator::advanceEpoch(int currentEpoch) const
187{
188 uint64_t count = 0;
189 // Compute step number at end of this simulation epoch
190 uint64_t endStep = g_simulationStep + static_cast<uint64_t>(epochDuration_ / deltaT_);
191 model_->getLayout().getVertices().loadEpochInputs(g_simulationStep, endStep);
192 // DEBUG_MID(model->logSimStep();) // Generic model debug call
193 uint64_t onePercent = (epochDuration_ / deltaT_) * numEpochs_ * 0.01;
194 while (g_simulationStep < endStep) {
195 // Output status once every 1% of total simulation time
196 if (count % onePercent == 0) {
197 // Simulation time is calculated independently to handle deserialization scenarios.
198 // If g_simulationStep is set to a value greater than totalDuration after deserialization,
199 // a modulo operation is applied to ensure the displayed simulation time is accurate.
200 // If totalDuration is zero, the entire simulation time is determined by g_simulationStep * deltaT_.
201 uint64_t totalDuration = (epochDuration_ * numEpochs_) - 1;
202 auto simulatedTime
203 = (totalDuration == 0 ? (g_simulationStep * deltaT_)
204 : (uint64_t)(g_simulationStep * deltaT_) % totalDuration);
205
206 LOG4CPLUS_TRACE(consoleLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
207 << " simulating time: " << simulatedTime << "/"
208 << totalDuration);
209
210 LOG4CPLUS_TRACE(workbenchLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
211 << " simulating time: " << simulatedTime << "/"
212 << totalDuration);
213
214 count = 0;
215 }
216
217 count++;
218 // input stimulus
219 /***** S_INPUT NOT IN REPO YET *******/
220 // if (pInput != nullptr)
221 // pInput->inputStimulus();
222 // Advance the Network one time step
223 model_->advance();
224 g_simulationStep++;
225 }
226}
