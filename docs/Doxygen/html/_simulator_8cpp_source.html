138void Simulator::simulate()
139{
140 // Main simulation loop - execute maxGrowthSteps
141 for (currentEpoch_ = 1; currentEpoch_ <= numEpochs_; currentEpoch_++) {
142 LOG4CPLUS_TRACE(consoleLogger_, "Performing epoch number: " << currentEpoch_);
143 LOG4CPLUS_TRACE(fileLogger_, "Begin network state:");
144
145#ifdef PERFORMANCE_METRICS
146 // Start timer for advance
147 short_timer.start();
148#endif
149 // Advance simulation to next growth cycle
150 advanceEpoch(currentEpoch_);
151#ifdef PERFORMANCE_METRICS
152 // Time to advance
153 t_host_advance += short_timer.lap() / 1000000.0;
154#endif
155 LOG4CPLUS_TRACE(consoleLogger_,
156 "done with epoch cycle " << currentEpoch_ << ", beginning growth update");
157 LOG4CPLUS_TRACE(edgeLogger_, "Epoch: " << currentEpoch_);
158
159#ifdef PERFORMANCE_METRICS
160 // Start timer for connection update
161 short_timer.start();
162#endif
163 // Update the neuron network
164 model_->updateConnections();
165 model_->updateHistory();
166
167#ifdef PERFORMANCE_METRICS
168 // Times converted from microseconds to seconds
169 // Time to update synapses
170 t_host_adjustEdges += short_timer.lap() / 1000000.0;
171 // Time since start of simulation
172 double total_time = timer.lap() / 1000000.0;
173
174 cout << "\ntotal_time: " << total_time << " seconds" << endl;
175 printPerformanceMetrics(total_time, currentEpoch);
176 cout << endl;
177#endif
178 }
179 LOG4CPLUS_TRACE(workbenchLogger_, "Complete");
180}

185void Simulator::advanceEpoch(int currentEpoch) const
186{
187 uint64_t count = 0;
188 // Compute step number at end of this simulation epoch
189 uint64_t endStep = g_simulationStep + static_cast<uint64_t>(epochDuration_ / deltaT_);
190 model_->getLayout().getVertices().loadEpochInputs(g_simulationStep, endStep);
191 // DEBUG_MID(model->logSimStep();) // Generic model debug call
192 uint64_t onePercent = (epochDuration_ / deltaT_) * numEpochs_ * 0.01;
193 while (g_simulationStep < endStep) {
194 // Output status once every 1% of total simulation time
195 if (count % onePercent == 0) {
196 // Simulation time is calculated independently to handle deserialization scenarios.
197 // If g_simulationStep is set to a value greater than totalDuration after deserialization,
198 // a modulo operation is applied to ensure the displayed simulation time is accurate.
199 // If totalDuration is zero, the entire simulation time is determined by g_simulationStep * deltaT_.
200 uint64_t totalDuration = (epochDuration_ * numEpochs_) - 1;
201 auto simulatedTime
202 = (totalDuration == 0 ? (g_simulationStep * deltaT_)
203 : (uint64_t)(g_simulationStep * deltaT_) % totalDuration);
204
205 LOG4CPLUS_TRACE(consoleLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
206 << " simulating time: " << simulatedTime << "/"
207 << totalDuration);
208
209 LOG4CPLUS_TRACE(workbenchLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
210 << " simulating time: " << simulatedTime << "/"
211 << totalDuration);
212
213 count = 0;
214 }
215
216 count++;
217 // input stimulus
218 /***** S_INPUT NOT IN REPO YET *******/
219 // if (pInput != nullptr)
220 // pInput->inputStimulus();
221 // Advance the Network one time step
222 model_->advance();
223 g_simulationStep++;
224 }
225}
