139 void Simulator::simulate() {
140  // Main simulation loop - execute maxGrowthSteps
141  for (int currentEpoch = 1; currentEpoch <= numEpochs_; currentEpoch++) {
142  LOG4CPLUS_TRACE(consoleLogger_, "Performing epoch number: " << currentEpoch);
143  LOG4CPLUS_TRACE(fileLogger_, "Begin network state:");
144  currentEpoch_ = currentEpoch;
145 #ifdef PERFORMANCE_METRICS
146  // Start timer for advance
147  short_timer.start();
148 #endif
149  // Advance simulation to next growth cycle
150  advanceEpoch(currentEpoch);
151 #ifdef PERFORMANCE_METRICS
152  // Time to advance
153  t_host_advance += short_timer.lap() / 1000000.0;
154 #endif
155  LOG4CPLUS_TRACE(consoleLogger_, "done with epoch cycle " << currentEpoch_ << ", beginning growth update");
156  LOG4CPLUS_TRACE(edgeLogger_, "Epoch: " << currentEpoch_);
157 
158 #ifdef PERFORMANCE_METRICS
159  // Start timer for connection update
160  short_timer.start();
161 #endif
162  // Update the neuron network
163  model_->updateConnections();
164  model_->updateHistory();
165 
166 #ifdef PERFORMANCE_METRICS
167  // Times converted from microseconds to seconds
168  // Time to update synapses
169  t_host_adjustEdges += short_timer.lap() / 1000000.0;
170  // Time since start of simulation
171  double total_time = timer.lap() / 1000000.0;
172 
173  cout << "\ntotal_time: " << total_time << " seconds" << endl;
174  printPerformanceMetrics(total_time, currentEpoch);
175  cout << endl;
176 #endif
177  }

183 void Simulator::advanceEpoch(const int &currentEpoch) const {
184  uint64_t count = 0;
185  // Compute step number at end of this simulation epoch
186  uint64_t endStep = g_simulationStep
187  + static_cast<uint64_t>(epochDuration_ / deltaT_);
188  // DEBUG_MID(model->logSimStep();) // Generic model debug call
189  while (g_simulationStep < endStep) {
190  // Output status once every 10,000 steps
191  if (count % 10000 == 0) {
192  LOG4CPLUS_TRACE(consoleLogger_, "Epoch: " << currentEpoch << "/" << numEpochs_
193  << " simulating time: "
194  << g_simulationStep * deltaT_ << "/"
195  << (epochDuration_ * numEpochs_) - 1);
196  count = 0;
197  }
198  count++;
199  // input stimulus
200  /***** S_INPUT NOT IN REPO YET *******/
201 // if (pInput != nullptr)
202 // pInput->inputStimulus();
203  // Advance the Network one time step
204  model_->advance();
205  g_simulationStep++;
206  }
