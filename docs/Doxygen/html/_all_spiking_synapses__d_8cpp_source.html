448__global__ void advanceSpikingSynapsesDevice(int totalSynapseCount,
449 EdgeIndexMapDevice *edgeIndexMapDevice,
450 uint64_t simulationStep, const BGFLOAT deltaT,
451 AllSpikingSynapsesDeviceProperties *allEdgesDevice)
452{
453 int idx = blockIdx.x * blockDim.x + threadIdx.x;
454 if (idx >= totalSynapseCount)
455 return;
456 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
457 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
458 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
459
460 // Checks if there is an input spike in the queue.
461 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
462
463 // is an input in the queue?
464 if (isFired) {
465 switch (classSynapses_d) {
466 case classAllSpikingSynapses:
467 changeSpikingSynapsesPSRDevice(
468 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
469 simulationStep, deltaT);
470 break;
471 case classAllDSSynapses:
472 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice),
473 iEdg, simulationStep, deltaT);
474 break;
475 default:
476 assert(false);
477 }
478 }
479 // decay the post spike response
480 psr *= decay;
481}

491CUDA_CALLABLE bool
492 isSpikingSynapsesSpikeQueueDevice(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
493 BGSIZE iEdg)
494{
495 uint32_t &delayQueue = allEdgesDevice->delayQueue_[iEdg];
496 int &delayIdx = allEdgesDevice->delayIndex_[iEdg];
497 int delayQueueLength = allEdgesDevice->delayQueueLength_[iEdg];
498 uint32_t delayMask = (0x1 << delayIdx);
499 bool isFired = delayQueue & (delayMask);
500 delayQueue &= ~(delayMask);
501 if (++delayIdx >= delayQueueLength) {
502 delayIdx = 0;
503 }
504 return isFired;
505}
