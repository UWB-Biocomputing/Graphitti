473__global__ void advanceSpikingSynapsesDevice(int totalSynapseCount,
474 EdgeIndexMapDevice *edgeIndexMapDevice,
475 uint64_t simulationStep, BGFLOAT deltaT,
476 AllSpikingSynapsesDeviceProperties *allEdgesDevice)
477{
478 int idx = blockIdx.x * blockDim.x + threadIdx.x;
479 if (idx >= totalSynapseCount)
480 return;
481 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
482 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
483 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
484
485 // Checks if there is an input spike in the queue.
486 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
487
488 // is an input in the queue?
489 if (isFired) {
490 switch (classSynapses_d) {
491 case enumClassSynapses::classAllSpikingSynapses:
492 changeSpikingSynapsesPSRDevice(
493 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
494 simulationStep, deltaT);
495 break;
496 case enumClassSynapses::classAllDSSynapses:
497 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice),
498 iEdg, simulationStep, deltaT);
499 break;
500 default:
501 assert(false);
502 }
503 }
504 // decay the post spike response
505 psr *= decay;
506}

516CUDA_CALLABLE bool
517 isSpikingSynapsesSpikeQueueDevice(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
518 BGSIZE iEdg)
519{
520 uint32_t &delayQueue = allEdgesDevice->delayQueue_[iEdg];
521 int &delayIdx = allEdgesDevice->delayIndex_[iEdg];
522 int delayQueueLength = allEdgesDevice->delayQueueLength_[iEdg];
523 uint32_t delayMask = (0x1 << delayIdx);
524 bool isFired = delayQueue & (delayMask);
525 delayQueue &= ~(delayMask);
526 if (++delayIdx >= delayQueueLength) {
527 delayIdx = 0;
528 }
529 return isFired;
530}
