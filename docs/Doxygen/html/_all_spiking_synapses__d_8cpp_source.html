451__global__ void advanceSpikingSynapsesDevice(int totalSynapseCount,
452 EdgeIndexMapDevice *edgeIndexMapDevice,
453 uint64_t simulationStep, BGFLOAT deltaT,
454 AllSpikingSynapsesDeviceProperties *allEdgesDevice)
455{
456 int idx = blockIdx.x * blockDim.x + threadIdx.x;
457 if (idx >= totalSynapseCount)
458 return;
459 BGSIZE iEdg = edgeIndexMapDevice->incomingEdgeIndexMap_[idx];
460 BGFLOAT &psr = allEdgesDevice->psr_[iEdg];
461 BGFLOAT decay = allEdgesDevice->decay_[iEdg];
462
463 // Checks if there is an input spike in the queue.
464 bool isFired = isSpikingSynapsesSpikeQueueDevice(allEdgesDevice, iEdg);
465
466 // is an input in the queue?
467 if (isFired) {
468 switch (classSynapses_d) {
469 case classAllSpikingSynapses:
470 changeSpikingSynapsesPSRDevice(
471 static_cast<AllSpikingSynapsesDeviceProperties *>(allEdgesDevice), iEdg,
472 simulationStep, deltaT);
473 break;
474 case classAllDSSynapses:
475 changeDSSynapsePSRDevice(static_cast<AllDSSynapsesDeviceProperties *>(allEdgesDevice),
476 iEdg, simulationStep, deltaT);
477 break;
478 default:
479 assert(false);
480 }
481 }
482 // decay the post spike response
483 psr *= decay;
484}

494CUDA_CALLABLE bool
495 isSpikingSynapsesSpikeQueueDevice(AllSpikingSynapsesDeviceProperties *allEdgesDevice,
496 BGSIZE iEdg)
497{
498 uint32_t &delayQueue = allEdgesDevice->delayQueue_[iEdg];
499 int &delayIdx = allEdgesDevice->delayIndex_[iEdg];
500 int delayQueueLength = allEdgesDevice->delayQueueLength_[iEdg];
501 uint32_t delayMask = (0x1 << delayIdx);
502 bool isFired = delayQueue & (delayMask);
503 delayQueue &= ~(delayMask);
504 if (++delayIdx >= delayQueueLength) {
505 delayIdx = 0;
506 }
507 return isFired;
508}
