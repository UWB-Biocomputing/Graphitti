56 // based on the order in which the type appears in the list, in this case:

91 void readInputs()
92 {
93 ptree pt;
94 ifstream inputFile;
95
96 if (inputFilePath_.empty()) {
97 // If no inputFilePath_ has been assigned get it from the ParameterManager
98 string xpath = "//inputFile/text()";
99 if (!ParameterManager::getInstance().getStringByXpath(xpath, inputFilePath_)) {
100 LOG4CPLUS_FATAL(consoleLogger_, "Could not find XML Path: " << xpath);
101 exit(EXIT_FAILURE);
102 }
103 }
104
105 assert(!inputFilePath_.empty());
106
107 inputFile.open(inputFilePath_.c_str());
108 if (!inputFile.is_open()) {
109 LOG4CPLUS_FATAL(consoleLogger_, "Faile to open file: " << inputFilePath_);
110 exit(EXIT_FAILURE);
111 }
112
113 // Load all events from the Input File
114 boost::property_tree::xml_parser::read_xml(inputFile, pt);
115
116 // Get clock tick size
117 const ptree &dataNode = pt.get_child("simulator_inputs").get_child("data");
118 const ptree &dataAttr = dataNode.get_child("<xmlattr>");
119 clockTickSize_ = dataAttr.get<int>("clock_tick_size");
120 clockTickUnit_ = dataAttr.get<string>("clock_tick_unit");
121
122 BOOST_FOREACH (ptree::value_type const &v, dataNode) {
123 if (v.first == "vertex") {
124 int vertex_id = v.second.get_child("<xmlattr>").get<int>("id");
125 // loop over list of events that belong to this vertex
126 BOOST_FOREACH (ptree::value_type const &evnt, v.second) {
127 if (evnt.first == "event") {
128 try {
129 const ptree &attributes = evnt.second.get_child("<xmlattr>");
130 T event;
131 // Loop over the set of attributes and add them to the event object.
132 // We only read registered properties, all others are ignored.
133 for (auto propPair : registeredPropMap_) {
134 getProperty(event, propPair.first, propPair.second, attributes);
135 }
136 // Add the event object to the event map. Map's operator[] creates
137 // and empty queue if it doesn't yet contain this vertex_id.
138 eventsMap_[vertex_id].push(event);
139 } catch (boost::property_tree::ptree_bad_data e) {
140 LOG4CPLUS_FATAL(consoleLogger_,
141 "InputManager failed to read event node: " << e.what());
142 exit(EXIT_FAILURE);
143 } catch (boost::bad_get e) {
144 LOG4CPLUS_FATAL(consoleLogger_, "Failed to read event property: " << e.what());
145 exit(EXIT_FAILURE);
146 }
147 }
148 }
149 }
150 }
151 LOG4CPLUS_DEBUG(fileLogger_, "Input file loaded successfully");
152 }

162 CircularBuffer<T> &getEvents(const VertexId_t &vertexId, uint64_t firstStep, uint64_t lastStep,
163 CircularBuffer<T> &buffer)
164 {
165 queue<T> &eventQueue = eventsMap_[vertexId]; // Get a reference to the event queue
166
167 while (!eventQueue.empty() && eventQueue.front().time < lastStep) {
168 // We shouldn't have previous epoch events in the queue
169 assert(eventQueue.front().time >= firstStep);
170 buffer.put(eventQueue.front());
171 eventQueue.pop();
172 }
173
174 return buffer;
175 }

267 bool getProperty(T &event, string propName, EventMemberPtr &eventMbrPtr,
268 const boost::property_tree::ptree &pTree)
269 {
270 switch (static_cast<PropertyType>(eventMbrPtr.which())) {
271 // variant.which() returns a value between 0 and number of types - 1
272 case PropertyType::INTEGER: {
273 int T::*propPtr = get<int T::*>(eventMbrPtr);
274 event.*propPtr = pTree.get<int>(propName);
275 } break;
276 case PropertyType::LONG: {
277 long T::*propPtr = get<long T::*>(eventMbrPtr);
278 event.*propPtr = pTree.get<long>(propName);
279 } break;
280 case PropertyType::UINT64: {
281 uint64_t T::*propPtr = get<uint64_t T::*>(eventMbrPtr);
282 event.*propPtr = pTree.get<uint64_t>(propName);
283 } break;
284 case PropertyType::FLOAT: {
285 float T::*propPtr = get<float T::*>(eventMbrPtr);
286 event.*propPtr = pTree.get<float>(propName);
287 } break;
288 case PropertyType::DOUBLE: {
289 double T::*propPtr = get<double T::*>(eventMbrPtr);
290 event.*propPtr = pTree.get<double>(propName);
291 } break;
292 case PropertyType::STRING: {
293 string T::*propPtr = get<string T::*>(eventMbrPtr);
294 event.*propPtr = pTree.get<string>(propName);
295 } break;
296 default:
297 LOG4CPLUS_DEBUG(fileLogger_, "Property not supported: " << propName);
298 return false;
299 }
300 return true;
301 }
