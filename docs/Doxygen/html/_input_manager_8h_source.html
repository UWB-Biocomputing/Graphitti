90 void readInputs()
91 {
92 ptree pt;
93 ifstream inputFile;
94
95 if (inputFilePath_.empty()) {
96 // If no inputFilePath_ has been assigned get it from the ParameterManager
97 string xpath = "//inputFile/text()";
98 if (!ParameterManager::getInstance().getStringByXpath(xpath, inputFilePath_)) {
99 LOG4CPLUS_FATAL(consoleLogger_, "Could not find XML Path: " << xpath);
100 exit(EXIT_FAILURE);
101 }
102 }
103
104 assert(!inputFilePath_.empty());
105
106 inputFile.open(inputFilePath_.c_str());
107 if (!inputFile.is_open()) {
108 LOG4CPLUS_FATAL(consoleLogger_, "Faile to open file: " << inputFilePath_);
109 exit(EXIT_FAILURE);
110 }
111
112 // Load all events from the Input File
113 boost::property_tree::xml_parser::read_xml(inputFile, pt);
114
115 // Get clock tick size
116 const ptree &dataNode = pt.get_child("simulator_inputs").get_child("data");
117 const ptree &dataAttr = dataNode.get_child("<xmlattr>");
118 clockTickSize_ = dataAttr.get<int>("clock_tick_size");
119 clockTickUnit_ = dataAttr.get<string>("clock_tick_unit");
120
121 BOOST_FOREACH (ptree::value_type const &v, dataNode) {
122 if (v.first == "vertex") {
123 int vertex_id = v.second.get_child("<xmlattr>").get<int>("id");
124 // loop over list of events that belong to this vertex
125 BOOST_FOREACH (ptree::value_type const &evnt, v.second) {
126 if (evnt.first == "event") {
127 try {
128 const ptree &attributes = evnt.second.get_child("<xmlattr>");
129 T event;
130 // Loop over the set of attributes and add them to the event object.
131 // We only read registered properties, all others are ignored.
132 for (auto propPair : registeredPropMap_) {
133 getProperty(event, propPair.first, propPair.second, attributes);
134 }
135 // Add the event object to the event map. Map's operator[] creates
136 // and empty queue if it doesn't yet contain this vertex_id.
137 eventsMap_[vertex_id].push(event);
138 } catch (boost::property_tree::ptree_bad_data e) {
139 LOG4CPLUS_FATAL(consoleLogger_,
140 "InputManager failed to read event node: " << e.what());
141 exit(EXIT_FAILURE);
142 } catch (boost::bad_get e) {
143 LOG4CPLUS_FATAL(consoleLogger_, "Failed to read event property: " << e.what());
144 exit(EXIT_FAILURE);
145 }
146 }
147 }
148 }
149 }
150 LOG4CPLUS_DEBUG(fileLogger_, "Input file loaded successfully");
151 }

159 vector<T> getEvents(const VertexId_t &vertexId, uint64_t firstStep, uint64_t lastStep)
160 {
161 vector<T> result = vector<T>(); // Will hold the list of events
162 queue<T> &eventQueue = eventsMap_[vertexId]; // Get a reference to the event queue
163
164 while (!eventQueue.empty() && eventQueue.front().time < lastStep) {
165 // We shouldn't have previous epoch events in the queue
166 assert(eventQueue.front().time >= firstStep);
167 result.push_back(eventQueue.front());
168 eventQueue.pop();
169 }
170
171 return result;
172 }

264 bool getProperty(T &event, string propName, EventMemberPtr &eventMbrPtr,
265 const boost::property_tree::ptree &pTree)
266 {
267 switch (eventMbrPtr.which()) {
268 // variant.which() returns a value between 0 and number of types - 1
269 case PropertyType::INTEGER: {
270 int T::*propPtr = get<int T::*>(eventMbrPtr);
271 event.*propPtr = pTree.get<int>(propName);
272 } break;
273 case PropertyType::LONG: {
274 long T::*propPtr = get<long T::*>(eventMbrPtr);
275 event.*propPtr = pTree.get<long>(propName);
276 } break;
277 case PropertyType::UINT64: {
278 uint64_t T::*propPtr = get<uint64_t T::*>(eventMbrPtr);
279 event.*propPtr = pTree.get<uint64_t>(propName);
280 } break;
281 case PropertyType::FLOAT: {
282 float T::*propPtr = get<float T::*>(eventMbrPtr);
283 event.*propPtr = pTree.get<float>(propName);
284 } break;
285 case PropertyType::DOUBLE: {
286 double T::*propPtr = get<double T::*>(eventMbrPtr);
287 event.*propPtr = pTree.get<double>(propName);
288 } break;
289 case PropertyType::STRING: {
290 string T::*propPtr = get<string T::*>(eventMbrPtr);
291 event.*propPtr = pTree.get<string>(propName);
292 } break;
293 default:
294 LOG4CPLUS_DEBUG(fileLogger_, "Property not supported: " << propName);
295 return false;
296 }
297 return true;
298 }
