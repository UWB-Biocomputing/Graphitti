104void XmlRecorder::saveSimData(const AllVertices &vertices)
105{
106 Simulator &simulator = Simulator::getInstance();
107 // create Neuron Types matrix
108 VectorMatrix neuronTypes(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), EXC);
109 for (int i = 0; i < simulator.getTotalVertices(); i++) {
110 neuronTypes[i] = simulator.getModel()->getLayout()->vertexTypeMap_[i];
111 }
112 // create neuron threshold matrix
113 VectorMatrix neuronThresh(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), 0);
114 for (int i = 0; i < simulator.getTotalVertices(); i++) {
115 neuronThresh[i] = dynamic_cast<const AllIFNeurons &>(vertices).Vthresh_[i];
116 }
117
118 // Write XML header information:
119 resultOut_ << "<?xml version=\"1.0\" standalone=\"no\"?>\n"
120 << "<!-- State output file for the DCT growth modeling-->\n";
121 // stateOut << version; TODO: version
122 auto layout = simulator.getModel()->getLayout();
123
124 // Write the core state information:
125 resultOut_ << "<SimState>\n";
126 resultOut_ << " " << burstinessHist_.toXML("burstinessHist") << endl;
127 resultOut_ << " " << spikesHistory_.toXML("spikesHistory") << endl;
128 resultOut_ << " " << layout->xloc_.toXML("xloc") << endl;
129 resultOut_ << " " << layout->yloc_.toXML("yloc") << endl;
130 resultOut_ << " " << neuronTypes.toXML("neuronTypes") << endl;
131
132 // create starter neurons matrix
133 int num_starter_neurons = static_cast<int>(layout->numEndogenouslyActiveNeurons_);
134 if (num_starter_neurons > 0) {
135 VectorMatrix starterNeurons(MATRIX_TYPE, MATRIX_INIT, 1, num_starter_neurons);
136 getStarterNeuronMatrix(starterNeurons, layout->starterMap_);
137 resultOut_ << " " << starterNeurons.toXML("starterNeurons") << endl;
138 }
139
140 // Write neuron threshold
141 resultOut_ << " " << neuronThresh.toXML("neuronThresh") << endl;
142
143 // write epoch duration
144 resultOut_
145 << " <Matrix name=\"Tsim\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
146 << endl;
147 resultOut_ << " " << simulator.getEpochDuration() << endl;
148 resultOut_ << "</Matrix>" << endl;
149
150 // write simulation end time
151 resultOut_
152 << " <Matrix name=\"simulationEndTime\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
153 << endl;
154 resultOut_ << " " << g_simulationStep * simulator.getDeltaT() << endl;
155 resultOut_ << "</Matrix>" << endl;
156 resultOut_ << "</SimState>" << endl;
157}
