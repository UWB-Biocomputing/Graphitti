95string XmlRecorder::toXML(const string &name, vector<multipleTypes> &singleBuffer_,
96 const string &basicType) const
97{
98 stringstream os;
99
100 // output file header
101 os << "<Matrix ";
102 if (!name.empty())
103 os << "name=\"" << name << "\" ";
104 os << "type=\"complete\" rows=\"" << 1 << "\" columns=\"" << singleBuffer_.size()
105 << "\" multiplier=\"1.0\">" << endl;
106 os << " ";
107
108 // Retrives value from variant
109 for (const multipleTypes &element : singleBuffer_) {
110 if (basicType == typeid(uint64_t).name()) {
111 os << get<uint64_t>(element) << " ";
112 } else if (basicType == typeid(bool).name()) {
113 os << get<bool>(element) << " ";
114 } else if (basicType == typeid(int).name()) {
115 os << get<int>(element) << " ";
116 } else if (basicType == typeid(BGFLOAT).name()) {
117 os << get<BGFLOAT>(element) << " ";
118 } else {
119 perror("Error recording Recordable object");
120 exit(EXIT_FAILURE);
121 }
122 // Add more conditions if there are additional supported data types
123 }
124
125 os << endl;
126 os << "</Matrix>";
127
128 return os.str();
129}
