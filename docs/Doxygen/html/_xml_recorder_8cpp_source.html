68 void XmlRecorder::compileHistories(IAllVertices &vertices) {
69  AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
70  Simulator& simulator = Simulator::getInstance();
71  int maxSpikes = static_cast<int>(simulator.getEpochDuration() * simulator.getMaxFiringRate());
72 
73  // output spikes
74  for (int iNeuron = 0; iNeuron < simulator.getTotalVertices(); iNeuron++) {
75  uint64_t *pSpikes = spNeurons.spikeHistory_[iNeuron];
76 
77  const int spikeCount = spNeurons.spikeCount_[iNeuron];
78  const int offset = spNeurons.spikeCountOffset_[iNeuron];
79  for (int i = 0, idxSp = offset; i < spikeCount; i++, idxSp++) {
80  // Single precision (float) gives you 23 bits of significand, 8 bits of exponent,
81  // and 1 sign bit. Double precision (double) gives you 52 bits of significand,
82  // 11 bits of exponent, and 1 sign bit.
83  // Therefore, single precision can only handle 2^23 = 8,388,608 simulation steps
84  // or 8 epochs (1 epoch = 100s, 1 simulation step = 0.1ms).
85 
86  if (idxSp >= maxSpikes) idxSp = 0;
87  // compile network wide burstiness index data in 1s bins
88  int idx1 = static_cast<int>(static_cast<double>(pSpikes[idxSp]) * simulator.getDeltaT());
89  burstinessHist_[idx1] = burstinessHist_[idx1] + 1.0;
90 
91  // compile network wide spike count in 10ms bins
92  int idx2 = static_cast<int>( static_cast<double>( pSpikes[idxSp] ) * Simulator::getInstance().getDeltaT() *
93  100);
94  spikesHistory_[idx2] = spikesHistory_[idx2] + 1.0;
95  }
96  }
97 
98  // clear spike count
99  spNeurons.clearSpikeCounts();
100 }
