136void ConnGrowth::updateConns(AllVertices &vertices)
137{
138 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
139
140 // Calculate growth cycle firing rate for previous period
141 int maxSpikes = static_cast<int>(Simulator::getInstance().getEpochDuration()
142 * Simulator::getInstance().getMaxFiringRate());
143 for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
144 // Calculate firing rate
145 assert(spNeurons.vertexEvents_[i].getNumEventsInEpoch() < maxSpikes);
146 rates_[i] = spNeurons.vertexEvents_[i].getNumEventsInEpoch()
147 / Simulator::getInstance().getEpochDuration();
148 }
149
150 // compute vertex radii change and assign new values
151 outgrowth_ = 1.0
152 - 2.0
153 / (1.0
154 + exp((growthParams_.epsilon - rates_ / growthParams_.maxRate)
155 / growthParams_.beta));
156 deltaR_ = Simulator::getInstance().getEpochDuration() * growthParams_.rho * outgrowth_;
157 radii_ += deltaR_;
158}

164void ConnGrowth::updateOverlap()
165{
166 int numVertices = Simulator::getInstance().getTotalVertices();
167 Layout &layout = *Simulator::getInstance().getModel()->getLayout();
168
169 LOG4CPLUS_INFO(fileLogger_, "Computing areas of overlap");
170
171 // Compute areas of overlap; this is only done for overlapping units
172 for (int i = 0; i < numVertices - 1; i++) {
173 for (int j = i + 1; j < numVertices; j++) {
174 area_(i, j) = 0.0;
175
176 // Calculate the distance between neuron frontiers
177 BGFLOAT frontierDelta = layout.dist_(j, i) - (radii_[j] + radii_[i]);
178
179 if (frontierDelta < 0) {
180 BGFLOAT lenAB = layout.dist_(i, j);
181 BGFLOAT r1 = radii_[i];
182 BGFLOAT r2 = radii_[j];
183
184 if (lenAB + min(r1, r2) <= max(r1, r2)) {
185 area_(i, j) = pi * min(r1, r2) * min(r1, r2); // Completely overlapping unit
186
187 LOG4CPLUS_DEBUG(fileLogger_, "Completely overlapping (i, j, r1, r2, area): "
188 << i << ", " << j << ", " << r1 << ", " << r2 << ", "
189 << area_(i, j) << endl);
190 } else {
191 // Partially overlapping unit
192 BGFLOAT lenAB2 = layout.dist2_(i, j);
193 BGFLOAT r12 = r1 * r1;
194 BGFLOAT r22 = r2 * r2;
195
196 BGFLOAT cosCBA = (r22 + lenAB2 - r12) / (2.0 * r2 * lenAB);
197 BGFLOAT cosCAB = (r12 + lenAB2 - r22) / (2.0 * r1 * lenAB);
198
199 if (fabs(cosCBA) >= 1.0 || fabs(cosCAB) >= 1.0) {
200 area_(i, j) = 0.0;
201 } else {
202 BGFLOAT angCBA = acos(cosCBA);
203 BGFLOAT angCBD = 2.0 * angCBA;
204
205 BGFLOAT angCAB = acos(cosCAB);
206 BGFLOAT angCAD = 2.0 * angCAB;
207
208 area_(i, j) = 0.5 * (r22 * (angCBD - sin(angCBD)) + r12 * (angCAD - sin(angCAD)));
209 }
210 }
211 }
212 }
213 }
214}

222void ConnGrowth::updateSynapsesWeights()
223{
224 int numVertices = Simulator::getInstance().getTotalVertices();
225 AllNeuroEdges &synapses = dynamic_cast<AllNeuroEdges &>(*edges_);
226 Layout &layout = *Simulator::getInstance().getModel()->getLayout();
227 AllVertices &vertices = *layout.getVertices();
228
229 // For now, we just set the weights to equal the areas. We will later
230 // scale it and set its sign (when we index and get its sign).
231 W_ = area_;
232 int adjusted = 0;
233 int couldBeRemoved = 0; // TODO: use this value
234 int removed = 0;
235 int added = 0;
236
237 LOG4CPLUS_INFO(fileLogger_, "Adjusting Synapse weights");
238
239 // Scale and add sign to the areas
240 // visit each neuron 'a'
241 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
242 // and each destination neuron 'b'
243 for (int destVertex = 0; destVertex < numVertices; destVertex++) {
244 // visit each synapse at (xa,ya)
245 bool connected = false;
246 edgeType type = layout.edgType(srcVertex, destVertex);
247
248 // for each existing synapse
249 BGSIZE synapseCounts = synapses.edgeCounts_[destVertex];
250 BGSIZE synapse_adjusted = 0;
251 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * destVertex;
252 for (BGSIZE synapseIndex = 0; synapse_adjusted < synapseCounts; synapseIndex++, iEdg++) {
253 if (synapses.inUse_[iEdg] == true) {
254 // if there is a synapse between a and b
255 if (synapses.sourceVertexIndex_[iEdg] == srcVertex) {
256 connected = true;
257 adjusted++;
258 // adjust the strength of the synapse or remove
259 // it from the synapse map if it has gone below
260 // zero.
261 if (W_(srcVertex, destVertex) <= 0) {
262 removed++;
263 synapses.eraseEdge(destVertex, iEdg);
264 } else {
265 // adjust
266 // SYNAPSE_STRENGTH_ADJUSTMENT is 1.0e-8;
267 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
268 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
269
270 LOG4CPLUS_DEBUG(fileLogger_, "Weight of rgSynapseMap"
271 << "[" << synapseIndex
272 << "]: " << synapses.W_[iEdg]);
273 }
274 }
275 synapse_adjusted++;
276 }
277 }
278
279 // if not connected and weight(a,b) > 0, add a new synapse from a to b
280 if (!connected && (W_(srcVertex, destVertex) > 0)) {
281 // locate summation point
282 BGFLOAT *sumPoint = &(vertices.summationMap_[destVertex]);
283 added++;
284
285 BGSIZE iEdg;
286 synapses.addEdge(iEdg, type, srcVertex, destVertex, sumPoint,
287 Simulator::getInstance().getDeltaT());
288 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
289 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
290 }
291 }
292 }
293
294 LOG4CPLUS_INFO(fileLogger_,
295 "\nAdjusted: " << adjusted << "\nCould have been removed (TODO: calculate this): "
296 << couldBeRemoved << "\nRemoved: " << removed
297 << "\nAdded: " << added);
298}
