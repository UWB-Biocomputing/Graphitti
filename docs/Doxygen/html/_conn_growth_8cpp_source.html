146 void ConnGrowth::updateConns(IAllVertices &vertices) {
147  AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
148 
149  // Calculate growth cycle firing rate for previous period
150  int maxSpikes = static_cast<int> (Simulator::getInstance().getEpochDuration() *
151  Simulator::getInstance().getMaxFiringRate());
152  for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
153  // Calculate firing rate
154  assert(spNeurons.spikeCount_[i] < maxSpikes);
155  (*rates_)[i] = spNeurons.spikeCount_[i] / Simulator::getInstance().getEpochDuration();
156  }
157 
158  // compute vertex radii change and assign new values
159  (*outgrowth_) =
160  1.0 - 2.0 / (1.0 + exp((growthParams_.epsilon - *rates_ / growthParams_.maxRate) / growthParams_.beta));
161  (*deltaR_) = Simulator::getInstance().getEpochDuration() * growthParams_.rho * *outgrowth_;
162  (*radii_) += (*deltaR_);
163 }

184 void ConnGrowth::updateOverlap(BGFLOAT numVertices, Layout *layout) {
185  LOG4CPLUS_INFO(fileLogger_, "Computing areas of overlap");
186 
187  // Compute areas of overlap; this is only done for overlapping units
188  for (int i = 0; i < numVertices; i++) {
189  for (int j = 0; j < numVertices; j++) {
190  (*area_)(i, j) = 0.0;
191 
192  if ((*delta_)(i, j) < 0) {
193  BGFLOAT lenAB = (*layout->dist_)(i, j);
194  BGFLOAT r1 = (*radii_)[i];
195  BGFLOAT r2 = (*radii_)[j];
196 
197  if (lenAB + min(r1, r2) <= max(r1, r2)) {
198  (*area_)(i, j) = pi * min(r1, r2) * min(r1, r2); // Completely overlapping unit
199 
200  LOG4CPLUS_DEBUG(fileLogger_, "Completely overlapping (i, j, r1, r2, area): "
201  << i << ", " << j << ", " << r1 << ", " << r2 << ", " << *area_ << endl);
202  } else {
203  // Partially overlapping unit
204  BGFLOAT lenAB2 = (*layout->dist2_)(i, j);
205  BGFLOAT r12 = r1 * r1;
206  BGFLOAT r22 = r2 * r2;
207 
208  BGFLOAT cosCBA = (r22 + lenAB2 - r12) / (2.0 * r2 * lenAB);
209  BGFLOAT cosCAB = (r12 + lenAB2 - r22) / (2.0 * r1 * lenAB);
210 
211  if (fabs(cosCBA) >= 1.0 || fabs(cosCAB) >= 1.0) {
212  (*area_)(i, j) = 0.0;
213  } else {
214 
215  BGFLOAT angCBA = acos(cosCBA);
216  BGFLOAT angCBD = 2.0 * angCBA;
217 
218  BGFLOAT angCAB = acos(cosCAB);
219  BGFLOAT angCAD = 2.0 * angCAB;
220 
221  (*area_)(i, j) = 0.5 * (r22 * (angCBD - sin(angCBD)) + r12 * (angCAD - sin(angCAD)));
222  }
223  }
224  }
225  }
226  }
227 }

240 void ConnGrowth::updateSynapsesWeights(const int numVertices, IAllVertices &ivertices, AllEdges &iedges,
241  Layout *layout) {
242  AllVertices &vertices = dynamic_cast<AllVertices &>(ivertices);
243  AllNeuroEdges &synapses = dynamic_cast<AllNeuroEdges &>(iedges);
244 
245  // For now, we just set the weights to equal the areas. We will later
246  // scale it and set its sign (when we index and get its sign).
247  (*W_) = (*area_);
248 
249  int adjusted = 0;
250  int couldBeRemoved = 0; // TODO: use this value
251  int removed = 0;
252  int added = 0;
253 
254  LOG4CPLUS_INFO(fileLogger_, "Adjusting Synapse weights");
255 
256  // Scale and add sign to the areas
257  // visit each neuron 'a'
258  for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
259  // and each destination neuron 'b'
260  for (int destVertex = 0; destVertex < numVertices; destVertex++) {
261  // visit each synapse at (xa,ya)
262  bool connected = false;
263  edgeType type = layout->edgType(srcVertex, destVertex);
264 
265  // for each existing synapse
266  BGSIZE synapseCounts = synapses.edgeCounts_[destVertex];
267  BGSIZE synapse_adjusted = 0;
268  BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * destVertex;
269  for (BGSIZE synapseIndex = 0; synapse_adjusted < synapseCounts; synapseIndex++, iEdg++) {
270  if (synapses.inUse_[iEdg] == true) {
271  // if there is a synapse between a and b
272  if (synapses.sourceVertexIndex_[iEdg] == srcVertex) {
273  connected = true;
274  adjusted++;
275  // adjust the strength of the synapse or remove
276  // it from the synapse map if it has gone below
277  // zero.
278  if ((*W_)(srcVertex, destVertex) <= 0) {
279  removed++;
280  synapses.eraseEdge(destVertex, iEdg);
281  } else {
282  // adjust
283  // SYNAPSE_STRENGTH_ADJUSTMENT is 1.0e-8;
284  synapses.W_[iEdg] = (*W_)(srcVertex, destVertex) *
285  synapses.edgSign(type) * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
286 
287  LOG4CPLUS_DEBUG(fileLogger_, "Weight of rgSynapseMap" <<
288  "[" << synapseIndex << "]: " <<
289  synapses.W_[iEdg]);
290  }
291  }
292  synapse_adjusted++;
293 
294  }
295  }
296 
297  // if not connected and weight(a,b) > 0, add a new synapse from a to b
298  if (!connected && ((*W_)(srcVertex, destVertex) > 0)) {
299 
300  // locate summation point
301  BGFLOAT *sumPoint = &(vertices.summationMap_[destVertex]);
302  added++;
303 
304  BGSIZE iEdg;
305  synapses.addEdge(iEdg, type, srcVertex, destVertex, sumPoint,
306  Simulator::getInstance().getDeltaT());
307  synapses.W_[iEdg] =
308  (*W_)(srcVertex, destVertex) * synapses.edgSign(type) *
309  AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
310 
311  }
312  }
313  }
314 
315  LOG4CPLUS_INFO(fileLogger_, "\nAdjusted: " << adjusted
316  << "\nCould have been removed (TODO: calculate this): " << couldBeRemoved
317  << "\nRemoved: " << removed
318  << "\nAdded: " << added);
319 }
