150void ConnGrowth::updateConns(AllVertices &vertices)
151{
152 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
153
154 // Calculate growth cycle firing rate for previous period
155 int maxSpikes = static_cast<int>(Simulator::getInstance().getEpochDuration()
156 * Simulator::getInstance().getMaxFiringRate());
157 for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
158 // Calculate firing rate
159 assert(spNeurons.vertexEvents_[i].getNumElementsInEpoch() < maxSpikes);
160 rates_[i] = spNeurons.vertexEvents_[i].getNumElementsInEpoch()
161 / Simulator::getInstance().getEpochDuration();
162 }
163
164 // compute vertex radii change and assign new values
165 outgrowth_ = 1.0
166 - 2.0
167 / (1.0
168 + exp((growthParams_.epsilon - rates_ / growthParams_.maxRate)
169 / growthParams_.beta));
170 deltaR_ = Simulator::getInstance().getEpochDuration() * growthParams_.rho * outgrowth_;
171 radii_ += deltaR_;
172}

176{

178 // Update distance between frontiers
179 Layout &layout = Simulator::getInstance().getModel().getLayout();
180 int numVertices = Simulator::getInstance().getTotalVertices();
181 for (int unit = 0; unit < numVertices - 1; unit++) {
182 for (int i = unit + 1; i < numVertices; i++) {
183 delta_(unit, i) = layout.dist_(unit, i) - (radii_[unit] + radii_[i]);
184 delta_(i, unit) = delta_(unit, i);

190void ConnGrowth::updateOverlap()
191{
192 int numVertices = Simulator::getInstance().getTotalVertices();
193 Layout &layout = Simulator::getInstance().getModel().getLayout();
194
195 LOG4CPLUS_INFO(fileLogger_, "Computing areas of overlap");
196
197 // Compute areas of overlap; this is only done for overlapping units
198 for (int i = 0; i < numVertices; i++) {
199 for (int j = 0; j < numVertices; j++) {
200 area_(i, j) = 0.0;
201
202 if (delta_(i, j) < 0) {
203 BGFLOAT lenAB = layout.dist_(i, j);
204 BGFLOAT r1 = radii_[i];
205 BGFLOAT r2 = radii_[j];
206
207 if (lenAB + min(r1, r2) <= max(r1, r2)) {
208 area_(i, j) = pi * min(r1, r2) * min(r1, r2); // Completely overlapping unit
209
210 LOG4CPLUS_DEBUG(fileLogger_, "Completely overlapping (i, j, r1, r2, area): "
211 << i << ", " << j << ", " << r1 << ", " << r2 << ", "
212 << area_(i, j) << endl);
213 } else {
214 // Partially overlapping unit
215 BGFLOAT lenAB2 = layout.dist2_(i, j);
216 BGFLOAT r12 = r1 * r1;
217 BGFLOAT r22 = r2 * r2;
218
219 BGFLOAT cosCBA = (r22 + lenAB2 - r12) / (2.0 * r2 * lenAB);
220 BGFLOAT cosCAB = (r12 + lenAB2 - r22) / (2.0 * r1 * lenAB);
221
222 if (fabs(cosCBA) >= 1.0 || fabs(cosCAB) >= 1.0) {
223 area_(i, j) = 0.0;
224 } else {
225 BGFLOAT angCBA = acos(cosCBA);
226 BGFLOAT angCBD = 2.0 * angCBA;
227
228 BGFLOAT angCAB = acos(cosCAB);
229 BGFLOAT angCAD = 2.0 * angCAB;
230
231 area_(i, j) = 0.5 * (r22 * (angCBD - sin(angCBD)) + r12 * (angCAD - sin(angCAD)));
232 }
233 }
234 }
235 }
236 }
237}

245void ConnGrowth::updateEdgesWeights()
246{
247 int numVertices = Simulator::getInstance().getTotalVertices();
248 AllNeuroEdges &synapses = dynamic_cast<AllNeuroEdges &>(*edges_);
249 Layout &layout = Simulator::getInstance().getModel().getLayout();
250 AllVertices &vertices = layout.getVertices();
251
252 // For now, we just set the weights to equal the areas. We will later
253 // scale it and set its sign (when we index and get its sign).
254 W_ = area_;
255 int adjusted = 0;
256 int couldBeRemoved = 0; // TODO: use this value
257 int removed = 0;
258 int added = 0;
259
260 LOG4CPLUS_INFO(fileLogger_, "Adjusting Synapse weights");
261
262 // Scale and add sign to the areas
263 // visit each neuron 'a'
264 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
265 // and each destination neuron 'b'
266 for (int destVertex = 0; destVertex < numVertices; destVertex++) {
267 if (destVertex == srcVertex) {
268 // we don't create a synapse between the same neuron
269 continue;
270 }
271
272 // visit each synapse at (xa,ya)
273 bool connected = false;
274 edgeType type = layout.edgType(srcVertex, destVertex);
275
276 // for each existing synapse
277 BGSIZE synapseCounts = synapses.edgeCounts_[destVertex];
278 BGSIZE synapse_adjusted = 0;
279 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * destVertex;
280 for (BGSIZE synapseIndex = 0; synapse_adjusted < synapseCounts; synapseIndex++, iEdg++) {
281 if (synapses.inUse_[iEdg]) {
282 // if there is a synapse between a and b
283 if (synapses.sourceVertexIndex_[iEdg] == srcVertex) {
284 connected = true;
285 adjusted++;
286 // adjust the strength of the synapse or remove
287 // it from the synapse map if it has gone below
288 // zero.
289 if (W_(srcVertex, destVertex) <= 0) {
290 removed++;
291 synapses.eraseEdge(destVertex, iEdg);
292 } else {
293 // adjust
294 // SYNAPSE_STRENGTH_ADJUSTMENT is 1.0e-8;
295 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
296 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
297
298 LOG4CPLUS_DEBUG(fileLogger_, "Weight of rgSynapseMap"
299 << "[" << synapseIndex
300 << "]: " << synapses.W_[iEdg]);
301 }
302 }
303 synapse_adjusted++;
304 }
305 }
306
307 // if not connected and weight(a,b) > 0, add a new synapse from a to b
308 if (!connected && (W_(srcVertex, destVertex) > 0)) {
309 added++;
310 BGSIZE iEdg = synapses.addEdge(type, srcVertex, destVertex,
311 Simulator::getInstance().getDeltaT());
312 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
313 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
314 }
315 }
316 }
317
318 LOG4CPLUS_INFO(fileLogger_,
319 "\nAdjusted: " << adjusted << "\nCould have been removed (TODO: calculate this): "
320 << couldBeRemoved << "\nRemoved: " << removed
321 << "\nAdded: " << added);
322}
