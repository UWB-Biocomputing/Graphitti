147 void ConnGrowth::updateConns(IAllVertices &vertices) {
148  AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
149 
150  // Calculate growth cycle firing rate for previous period
151  int maxSpikes = static_cast<int> (Simulator::getInstance().getEpochDuration() *
152  Simulator::getInstance().getMaxFiringRate());
153  for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
154  // Calculate firing rate
155  assert(spNeurons.spikeCount_[i] < maxSpikes);
156  (*rates_)[i] = spNeurons.spikeCount_[i] / Simulator::getInstance().getEpochDuration();
157  }
158 
159  // compute vertex radii change and assign new values
160  (*outgrowth_) =
161  1.0 - 2.0 / (1.0 + exp((growthParams_.epsilon - *rates_ / growthParams_.maxRate) / growthParams_.beta));
162  (*deltaR_) = Simulator::getInstance().getEpochDuration() * growthParams_.rho * *outgrowth_;
163  (*radii_) += (*deltaR_);
164 }

185 void ConnGrowth::updateOverlap(BGFLOAT numVertices, Layout *layout) {
186  LOG4CPLUS_INFO(fileLogger_, "Computing areas of overlap");
187 
188  // Compute areas of overlap; this is only done for overlapping units
189  for (int i = 0; i < numVertices; i++) {
190  for (int j = 0; j < numVertices; j++) {
191  (*area_)(i, j) = 0.0;
192 
193  if ((*delta_)(i, j) < 0) {
194  BGFLOAT lenAB = (*layout->dist_)(i, j);
195  BGFLOAT r1 = (*radii_)[i];
196  BGFLOAT r2 = (*radii_)[j];
197 
198  if (lenAB + min(r1, r2) <= max(r1, r2)) {
199  (*area_)(i, j) = pi * min(r1, r2) * min(r1, r2); // Completely overlapping unit
200 
201  LOG4CPLUS_DEBUG(fileLogger_, "Completely overlapping (i, j, r1, r2, area): "
202  << i << ", " << j << ", " << r1 << ", " << r2 << ", " << *area_ << endl);
203  } else {
204  // Partially overlapping unit
205  BGFLOAT lenAB2 = (*layout->dist2_)(i, j);
206  BGFLOAT r12 = r1 * r1;
207  BGFLOAT r22 = r2 * r2;
208 
209  BGFLOAT cosCBA = (r22 + lenAB2 - r12) / (2.0 * r2 * lenAB);
210  BGFLOAT cosCAB = (r12 + lenAB2 - r22) / (2.0 * r1 * lenAB);
211 
212  if (fabs(cosCBA) >= 1.0 || fabs(cosCAB) >= 1.0) {
213  (*area_)(i, j) = 0.0;
214  } else {
215 
216  BGFLOAT angCBA = acos(cosCBA);
217  BGFLOAT angCBD = 2.0 * angCBA;
218 
219  BGFLOAT angCAB = acos(cosCAB);
220  BGFLOAT angCAD = 2.0 * angCAB;
221 
222  (*area_)(i, j) = 0.5 * (r22 * (angCBD - sin(angCBD)) + r12 * (angCAD - sin(angCAD)));
223  }
224  }
225  }
226  }
227  }
228 }

260  // and each destination neuron 'b'

262  // visit each synapse at (xa,ya)

264  synapseType type = layout->synType(srcVertex, destVertex);
265

270  for (BGSIZE synapseIndex = 0; synapse_adjusted < synapseCounts; synapseIndex++, iEdg++) {
271  if (synapses.inUse_[iEdg] == true) {
272  // if there is a synapse between a and b
273  if (synapses.sourceNeuronIndex_[iEdg] == srcVertex) {

291  }
292  }

294 
295  }
296  }

307  Simulator::getInstance().getDeltaT());

312  }
