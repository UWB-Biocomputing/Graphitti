148void ConnGrowth::updateConns(AllVertices &vertices)
149{
150 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
151
152 // Calculate growth cycle firing rate for previous period
153 int maxSpikes = static_cast<int>(Simulator::getInstance().getEpochDuration()
154 * Simulator::getInstance().getMaxFiringRate());
155 for (int i = 0; i < Simulator::getInstance().getTotalVertices(); i++) {
156 // Calculate firing rate
157 assert(spNeurons.vertexEvents_[i].getNumElementsInEpoch() < maxSpikes);
158 rates_[i] = spNeurons.vertexEvents_[i].getNumElementsInEpoch()
159 / Simulator::getInstance().getEpochDuration();
160 }
161
162 // compute vertex radii change and assign new values
163 outgrowth_ = 1.0
164 - 2.0
165 / (1.0
166 + exp((growthParams_.epsilon - rates_ / growthParams_.maxRate)
167 / growthParams_.beta));
168 deltaR_ = Simulator::getInstance().getEpochDuration() * growthParams_.rho * outgrowth_;
169 radii_ += deltaR_;
170}

173void ConnGrowth::updateFrontiers()
174{
175 LOG4CPLUS_INFO(fileLogger_, "Updating distance between frontiers...");
176 // Update distance between frontiers
177 Layout &layout = Simulator::getInstance().getModel().getLayout();
178 int numVertices = Simulator::getInstance().getTotalVertices();
179 for (int unit = 0; unit < numVertices - 1; unit++) {
180 for (int i = unit + 1; i < numVertices; i++) {
181 delta_(unit, i) = layout.dist_(unit, i) - (radii_[unit] + radii_[i]);
182 delta_(i, unit) = delta_(unit, i);
183 }

188void ConnGrowth::updateOverlap()
189{
190 int numVertices = Simulator::getInstance().getTotalVertices();
191 Layout &layout = Simulator::getInstance().getModel().getLayout();
192
193 LOG4CPLUS_INFO(fileLogger_, "Computing areas of overlap");
194
195 // Compute areas of overlap; this is only done for overlapping units
196 for (int i = 0; i < numVertices; i++) {
197 for (int j = 0; j < numVertices; j++) {
198 area_(i, j) = 0.0;
199
200 if (delta_(i, j) < 0) {
201 BGFLOAT lenAB = layout.dist_(i, j);
202 BGFLOAT r1 = radii_[i];
203 BGFLOAT r2 = radii_[j];
204
205 if (lenAB + min(r1, r2) <= max(r1, r2)) {
206 area_(i, j) = pi * min(r1, r2) * min(r1, r2); // Completely overlapping unit
207
208 LOG4CPLUS_DEBUG(fileLogger_, "Completely overlapping (i, j, r1, r2, area): "
209 << i << ", " << j << ", " << r1 << ", " << r2 << ", "
210 << area_(i, j) << endl);
211 } else {
212 // Partially overlapping unit
213 BGFLOAT lenAB2 = layout.dist2_(i, j);
214 BGFLOAT r12 = r1 * r1;
215 BGFLOAT r22 = r2 * r2;
216
217 BGFLOAT cosCBA = (r22 + lenAB2 - r12) / (2.0 * r2 * lenAB);
218 BGFLOAT cosCAB = (r12 + lenAB2 - r22) / (2.0 * r1 * lenAB);
219
220 if (fabs(cosCBA) >= 1.0 || fabs(cosCAB) >= 1.0) {
221 area_(i, j) = 0.0;
222 } else {
223 BGFLOAT angCBA = acos(cosCBA);
224 BGFLOAT angCBD = 2.0 * angCBA;
225
226 BGFLOAT angCAB = acos(cosCAB);
227 BGFLOAT angCAD = 2.0 * angCAB;
228
229 area_(i, j) = 0.5 * (r22 * (angCBD - sin(angCBD)) + r12 * (angCAD - sin(angCAD)));
230 }
231 }
232 }
233 }
234 }
235}

243void ConnGrowth::updateSynapsesWeights()
244{
245 int numVertices = Simulator::getInstance().getTotalVertices();
246 AllNeuroEdges &synapses = dynamic_cast<AllNeuroEdges &>(*edges_);
247 Layout &layout = Simulator::getInstance().getModel().getLayout();
248 AllVertices &vertices = layout.getVertices();
249
250 // For now, we just set the weights to equal the areas. We will later
251 // scale it and set its sign (when we index and get its sign).
252 W_ = area_;
253 int adjusted = 0;
254 int couldBeRemoved = 0; // TODO: use this value
255 int removed = 0;
256 int added = 0;
257
258 LOG4CPLUS_INFO(fileLogger_, "Adjusting Synapse weights");
259
260 // Scale and add sign to the areas
261 // visit each neuron 'a'
262 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
263 // and each destination neuron 'b'
264 for (int destVertex = 0; destVertex < numVertices; destVertex++) {
265 if (destVertex == srcVertex) {
266 // we don't create a synapse between the same neuron
267 continue;
268 }
269
270 // visit each synapse at (xa,ya)
271 bool connected = false;
272 edgeType type = layout.edgType(srcVertex, destVertex);
273
274 // for each existing synapse
275 BGSIZE synapseCounts = synapses.edgeCounts_[destVertex];
276 BGSIZE synapse_adjusted = 0;
277 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * destVertex;
278 for (BGSIZE synapseIndex = 0; synapse_adjusted < synapseCounts; synapseIndex++, iEdg++) {
279 if (synapses.inUse_[iEdg]) {
280 // if there is a synapse between a and b
281 if (synapses.sourceVertexIndex_[iEdg] == srcVertex) {
282 connected = true;
283 adjusted++;
284 // adjust the strength of the synapse or remove
285 // it from the synapse map if it has gone below
286 // zero.
287 if (W_(srcVertex, destVertex) <= 0) {
288 removed++;
289 synapses.eraseEdge(destVertex, iEdg);
290 } else {
291 // adjust
292 // SYNAPSE_STRENGTH_ADJUSTMENT is 1.0e-8;
293 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
294 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
295
296 LOG4CPLUS_DEBUG(fileLogger_, "Weight of rgSynapseMap"
297 << "[" << synapseIndex
298 << "]: " << synapses.W_[iEdg]);
299 }
300 }
301 synapse_adjusted++;
302 }
303 }
304
305 // if not connected and weight(a,b) > 0, add a new synapse from a to b
306 if (!connected && (W_(srcVertex, destVertex) > 0)) {
307 added++;
308 BGSIZE iEdg = synapses.addEdge(type, srcVertex, destVertex,
309 Simulator::getInstance().getDeltaT());
310 synapses.W_[iEdg] = W_(srcVertex, destVertex) * synapses.edgSign(type)
311 * AllNeuroEdges::SYNAPSE_STRENGTH_ADJUSTMENT;
312 }
313 }
314 }
315
316 LOG4CPLUS_INFO(fileLogger_,
317 "\nAdjusted: " << adjusted << "\nCould have been removed (TODO: calculate this): "
318 << couldBeRemoved << "\nRemoved: " << removed
319 << "\nAdded: " << added);
320}
