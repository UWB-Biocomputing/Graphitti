70void SparseMatrix::HashTable::insert(Element *el)
71{
72 // Don't overfill the table
73 if (capacity == size + 1)
74 throw Matrix_invalid_argument("SparseMatrix hash table full");
75
76 int start = hash(el);
77 int loc = start;
78
79 DEBUG_SPARSE(cerr << "\tInserting value " << el->value << " at (" << el->row << ", "
80 << el->column << ")" << endl;)
81
82 // Find first location to insert (if Element isn't in the table)
83 while ((table[loc] != &deleted) && (table[loc] != nullptr)) {
84 if (*table[loc] == *el)
85 throw Matrix_invalid_argument(
86 "Attempt to insert duplicate Element into SparseMatrix hash table [1]");
87 loc = (loc + 1) % capacity;
88 if (loc == start)
89 throw Matrix_invalid_argument("Hashtable insert wraparound: unable to insert Element [1]");
90 }
91
92 // If that first location is a deleted entry, then we need to
93 // continue searching to ensure the Element isn't in the table
94 if (table[loc] == &deleted) {
95 int loc2 = (loc + 1) % capacity;
96 while (table[loc2] != nullptr) {
97 if (loc2 == start)
98 throw Matrix_invalid_argument(
99 "Hashtable insert wraparound: unable to insert Element [2]");
100 loc2 = (loc2 + 1) % capacity;
101 if (*table[loc2] == *el)
102 throw Matrix_invalid_argument(
103 "Attempt to insert duplicate Element into SparseMatrix hash table [2]");
104 }
105 }
106
107 // At this point, either table[loc] is deleted or nullptr. Note that
108 // loc2 was only used to ensure we aren't inserting duplicates;
109 // table[loc] is the first available storage location.
110 table[loc] = el;
111 size++;
112
113 DEBUG_SPARSE(cerr << "\tInserted at table location " << loc << endl;)
114}

125SparseMatrix::Element *SparseMatrix::HashTable::retrieve(int r, int c)
126{
127 int loc = hash(r, c);
128
129 // Keep probing while we haven't found the Element (or finished the
130 // probing). Also, eliminate any Elements found with zero value.
131 while ((table[loc] != nullptr) && (!table[loc]->is_at(r, c))) {
132 if (table[loc]->value == 0.0) {
133 theMatrix->remove_lists(table[loc]);
134 table[loc] = &deleted;
135 size--;
136 }
137 loc = (loc + 1) % capacity;
138 }
139
140 // At this point, we either have the pointer to the Element or the
141 // nullptr pointer
142 return table[loc];
143}

371// Clear out storage
372void SparseMatrix::clear(void)
373{
374 DEBUG_SPARSE(cerr << "\tclearing " << rows << "X" << columns << " SparseMatrix...";)
375
376 // Since each Element is only allocated once (and shared among the
377 // row and column lists and the hash table), we only need to
378 // traverse all of the row lists and delete the Elements. This
379 // invalidates all of the column list and hash table pointers, but
380 // those lists will go away when the column list destructors are
381 // called in response to us deleting the column array and when
382 // the hash table is cleared.
383 if (theRows != nullptr) {
384 for (int i = 0; i < rows; i++)
385 for (list<Element *>::iterator it = theRows[i].begin(); it != theRows[i].end(); it++)
386 delete *it;
387 delete[] theRows;
388 }
389
390 if (theColumns != nullptr)
391 delete[] theColumns;
392
393 theElements.clear();
394
395 theRows = nullptr;
396 theColumns = nullptr;
397
398 DEBUG_SPARSE(cerr << "done." << endl;)
399}

413// Copy matrix to this one
414void SparseMatrix::copy(const SparseMatrix &source)
415{
416 DEBUG_SPARSE(cerr << "\t\t\tcopying " << source.rows << "X" << source.columns
417 << " SparseMatrix...";)
418
419 // We will access the source row-wise, inserting new Elements into
420 // the current SparseMatrix's row and column lists.
421 for (int i = 0; i < rows; i++) {
422 for (list<Element *>::iterator it = source.theRows[i].begin(); it != source.theRows[i].end();
423 it++) {
424 Element *el = new Element(i, (*it)->column, (*it)->value);
425 theRows[i].push_back(el);
426 theColumns[el->column].push_back(el);
427 try {
428 theElements.insert(el);
429 } catch (Matrix_invalid_argument e) {
430 cerr << "\nFailure during SparseMatrix copy() for element " << el->value << " at ("
431 << el->row << "," << el->column << ")" << endl;
432 cerr << "\twith " << theElements.size << " elements already copied at i=" << i
433 << ", hashed to " << theElements.hash(el) << " in table with capacity "
434 << theElements.capacity << endl;
435 cerr << "\tSource was: " << source << endl << endl;
436 throw e;
437 }
438 }
439 }
440
441 DEBUG_SPARSE(cerr << "\t\tdone." << endl;)
442}

446{
447 int rowNum;

450
451 // Iterate through the entries, inserting them into the row, column,

461 Element *el = new Element(rowNum, colNum, val);
462 theRows[rowNum].push_back(el);
463 theColumns[colNum].push_back(el);

549BGFLOAT &SparseMatrix::operator()(int r, int c)
550{
551 Element *el = theElements.retrieve(r, c);
552
553 // Because we're a mutator, we need to insert a zero-value element
554 // if the element wasn't found. We will rely on other methods to
555 // eliminate zero elements "on the fly".
556 if (el == nullptr) {
557 if ((el = new Element(r, c, 0.0)) == nullptr)
558 throw Matrix_bad_alloc("Failed allocating storage for SparseMatrix.");
559 theRows[r].push_back(el);
560 theColumns[c].push_back(el);
561 try {
562 theElements.insert(el);
563 } catch (Matrix_invalid_argument e) {
564 cerr << "Failure during SparseMatrix operator() at row " << r << " column " << c << ": "
565 << e.what() << endl;
566 exit(-1);
567 }
568 }
569
570 return el->value;
571}

607// Vector times a Sparse matrix
608const VectorMatrix operator*(const VectorMatrix &v, const SparseMatrix &m)
609{
610 if (m.rows != v.size) {
611 throw Matrix_domain_error(
612 "Illegal vector/matrix product. Rows of matrix must equal vector size.");
613 }
614
615 // the result is a vector the same size as m columns
616 VectorMatrix result("complete", "const", 1, m.columns, 0.0, "");
617
618 // To get each element of result, we will iterate down a column of m,
619 // multiplying each element found by the element in v at position
620 // equal to the row position of the m element. The result is the sum
621 // of those products.
622 for (int col = 0; col < m.columns; col++) {
623 BGFLOAT sum = 0.0;
624 for (list<SparseMatrix::Element *>::iterator el = m.theColumns[col].begin();
625 el != m.theColumns[col].end(); el++)
626 sum += (*el)->value * v[(*el)->row];
627 result[col] = sum;
628 }
629
630 return result;
