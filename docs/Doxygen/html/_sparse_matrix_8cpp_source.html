70 void SparseMatrix::HashTable::insert(Element* el)
71 {
72  // Don't overfill the table
73  if (capacity == size+1)
74  throw Matrix_invalid_argument("SparseMatrix hash table full");
75 
76  int start = hash(el);
77  int loc = start;
78 
79  DEBUG_SPARSE(cerr << "\tInserting value " << el->value << " at ("
80  << el->row << ", " << el->column << ")" << endl;)
81 
82  // Find first location to insert (if Element isn't in the table)
83  while ((table[loc] != &deleted) && (table[loc] != nullptr)) {
84  if (*table[loc] == *el)
85  throw Matrix_invalid_argument("Attempt to insert duplicate Element into SparseMatrix hash table [1]");
86  loc = (loc+1)%capacity;
87  if (loc == start)
88  throw Matrix_invalid_argument("Hashtable insert wraparound: unable to insert Element [1]");
89  }
90 
91  // If that first location is a deleted entry, then we need to
92  // continue searching to ensure the Element isn't in the table
93  if (table[loc] == &deleted) {
94  int loc2 = (loc+1)%capacity;
95  while (table[loc2] != nullptr) {
96  if (loc2 == start)
97  throw Matrix_invalid_argument("Hashtable insert wraparound: unable to insert Element [2]");
98  loc2 = (loc2+1)%capacity;
99  if (*table[loc2] == *el)
100  throw Matrix_invalid_argument("Attempt to insert duplicate Element into SparseMatrix hash table [2]");
101  }
102  }
103 
104  // At this point, either table[loc] is deleted or nullptr. Note that
105  // loc2 was only used to ensure we aren't inserting duplicates;
106  // table[loc] is the first available storage location.
107  table[loc] = el;
108  size++;
109 
110  DEBUG_SPARSE(cerr << "\tInserted at table location " << loc << endl;)
111 }

122 SparseMatrix::Element* SparseMatrix::HashTable::retrieve(int r, int c)
123 {
124  int loc = hash(r, c);
125 
126  // Keep probing while we haven't found the Element (or finished the
127  // probing). Also, eliminate any Elements found with zero value.
128  while ((table[loc] != nullptr) && (!table[loc]->is_at(r, c))) {
129  if (table[loc]->value == 0.0) {
130  theMatrix->remove_lists(table[loc]);
131  table[loc] = &deleted;
132  size--;
133  }
134  loc = (loc+1)%capacity;
135  }
136 
137  // At this point, we either have the pointer to the Element or the
138  // nullptr pointer
139  return table[loc];

374 void SparseMatrix::clear(void)
375 {
376  DEBUG_SPARSE(cerr << "\tclearing " << rows << "X" << columns << " SparseMatrix...";)
377 
378  // Since each Element is only allocated once (and shared among the
379  // row and column lists and the hash table), we only need to
380  // traverse all of the row lists and delete the Elements. This
381  // invalidates all of the column list and hash table pointers, but
382  // those lists will go away when the column list destructors are
383  // called in response to us deleting the column array and when
384  // the hash table is cleared.
385  if (theRows != nullptr) {
386  for (int i = 0; i < rows; i++)
387  for (list<Element*>::iterator it = theRows[i].begin();
388  it != theRows[i].end(); it++)
389  delete *it;
390  delete [] theRows;
391  }
392 
393  if (theColumns != nullptr)
394  delete [] theColumns;
395 
396  theElements.clear();
397 
398  theRows = nullptr;
399  theColumns = nullptr;
400 
401  DEBUG_SPARSE(cerr << "done." << endl;)
402 }
