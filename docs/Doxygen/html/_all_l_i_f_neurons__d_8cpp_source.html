93 bool fAllowBackPropagation)
94{
95 // determine which neuron this thread is processing
96 int idx = blockIdx.x * blockDim.x + threadIdx.x;
97 if (idx >= totalVertices)
98 return;
99
100 hasFired_[idx] = false;
101 BGFLOAT &sp = summationPoints_[idx];
102 BGFLOAT &vm = Vm_[idx];
103 BGFLOAT r_sp = sp;
104 BGFLOAT r_vm = vm;
105
106 if (numStepsInRefractoryPeriod_[idx] > 0) { // is neuron refractory?
107 --numStepsInRefractoryPeriod_[idx];
108 } else if (r_vm >= Vthresh_[idx]) { // should it fire?
109 int &spikeCount = allVerticesDevice->numElementsInEpoch_[idx];
110
111 // Note that the neuron has fired!
112 hasFired_[idx] = true;
113
114 // record spike time
115 int &queueEnd = allVerticesDevice->bufferEnd_[idx];
116 //int idxSp = allVerticesDevice->bufferEnd_[idx];
117 allVerticesDevice->spikeHistory_[idx][queueEnd] = simulationStep;
118 spikeCount++;
119
120 queueEnd = (queueEnd + 1) % maxSpikes;
121 // Debug statements to be removed
122 // DEBUG_SYNAPSE(
123 // printf("advanceLIFNeuronsDevice\n");
124 // printf(" index: %d\n", idx);
125 // printf(" simulationStep: %d\n\n", simulationStep);
126 // );
127
128 // calculate the number of steps in the absolute refractory period
129 numStepsInRefractoryPeriod_[idx] = static_cast<int>(Trefract_[idx] / deltaT + 0.5);
130
131 // reset to 'Vreset'
132 vm = Vreset_[idx];
133
134 // notify outgoing synapses of spike
135 BGSIZE synapseCounts = edgeIndexMapDevice->outgoingEdgeCount_[idx];
136 if (synapseCounts != 0) {
137 // get the index of where this neuron's list of synapses are
138 BGSIZE beginIndex = edgeIndexMapDevice->outgoingEdgeBegin_[idx];
139 // get the memory location of where that list begins
140 BGSIZE *outgoingMapBegin = &(edgeIndexMapDevice->outgoingEdgeIndexMap_[beginIndex]);
141
142 // for each synapse, let them know we have fired
143 for (BGSIZE i = 0; i < synapseCounts; i++) {
144 preSpikingSynapsesSpikeHitDevice(outgoingMapBegin[i], allEdgesDevice);
145 }
146 }
147
148 // notify incomming synapses of spike
149 synapseCounts = edgeIndexMapDevice->incomingEdgeCount_[idx];
150 if (fAllowBackPropagation && synapseCounts != 0) {
151 // get the index of where this neuron's list of synapses are
152 BGSIZE beginIndex = edgeIndexMapDevice->incomingEdgeBegin_[idx];
153 // get the memory location of where that list begins
154 BGSIZE *incomingMapBegin = &(edgeIndexMapDevice->incomingEdgeIndexMap_[beginIndex]);
155
156 // for each synapse, let them know we have fired
157 switch (classSynapses_d) {
158 case enumClassSynapses::classAllSTDPSynapses:
159 case enumClassSynapses::classAllDynamicSTDPSynapses:
160 for (BGSIZE i = 0; i < synapseCounts; i++) {
161 postSTDPSynapseSpikeHitDevice(
162 incomingMapBegin[i],
163 static_cast<AllSTDPSynapsesDeviceProperties *>(allEdgesDevice));
164 } // end for
165 break;
166
167 case enumClassSynapses::classAllSpikingSynapses:
168 case enumClassSynapses::classAllDSSynapses:
169 for (BGSIZE i = 0; i < synapseCounts; i++) {
170 postSpikingSynapsesSpikeHitDevice(incomingMapBegin[i], allEdgesDevice);
171 } // end for
172 break;
173
174 default:
175 assert(false);
176 } // end switch
177 }
178 } else {
179 r_sp += I0_[idx]; // add IO
180
181 // Random number alg. goes here
182 r_sp += (randNoise[idx] * Inoise_[idx]); // add cheap noise
183 vm = C1_[idx] * r_vm + C2_[idx] * (r_sp); // decay Vm and add inputs
184 }
185#ifdef VALIDATION_MODE
186 allVerticesDevice->spValidation_[idx] = r_sp;
187#endif
188 // clear synaptic input for next time step
189 sp = 0;
190}
