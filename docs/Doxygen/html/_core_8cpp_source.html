46bool Core::parseCommandLine(string executableName, string cmdLineArguments)
47{
48 ParamContainer cl; // todo: note as third party class.
49 cl.initOptions(false); // don't allow unknown parameters
50 cl.setHelpString(string(
51 "The UW Bothell graph-based simulation environment, for high-performance neural network and other graph-based problems\n Usage: "
52 + executableName + " "));
53
54 // Set up the comment line parser.
55 if ((cl.addParam("configfile", 'c', ParamContainer::filename, "parameter configuration filepath")
56 != ParamContainer::errOk)
57#if defined(USE_GPU)
58 || (cl.addParam("device", 'd', ParamContainer::regular, "CUDA device id")
59 != ParamContainer::errOk)
60#endif // USE_GPU
61 || (cl.addParam("stimulusfile", 's', ParamContainer::filename, "stimulus input filepath")
62 != ParamContainer::errOk)
63 || (cl.addParam("deserializefile", 'r', ParamContainer::filename,
64 "simulation deserialization filepath (enables deserialization)")
65 != ParamContainer::errOk)
66 || (cl.addParam("serializefile", 'w', ParamContainer::filename,
67 "simulation serialization filepath (enables serialization)")
68 != ParamContainer::errOk)
69 || (cl.addParam("version", 'v', ParamContainer::novalue,
70 "output current git commit ID and exit")
71 != ParamContainer::errOk)) {
72 cerr << "Internal error creating command line parser" << endl;
73 return false;
74 }
75
76 // Parse the command line
77 if (cl.parseCommandLine(cmdLineArguments) != ParamContainer::errOk) {
78 cl.dumpHelp(stderr, true, 78);
79 return false;
80 }
81
82 if (cl["version"].compare("") != 0) {
83 cout << "Git commit ID: " << GIT_COMMIT_ID << endl;
84 exit(0);
85 }
86
87 // Get the command line values
88 Simulator::getInstance().setConfigFileName(cl["configfile"]);
89 Simulator::getInstance().setDeserializationFileName(cl["deserializefile"]);
90 Simulator::getInstance().setSerializationFileName(cl["serializefile"]);
91 Simulator::getInstance().setStimulusFileName(cl["stimulusfile"]);
92
93#if defined(USE_GPU)
94 if (EOF == sscanf(cl["device"].c_str(), "%d", &g_deviceId)) {
95 g_deviceId = 0;
96 }
97#endif // USE_GPU
98 return true;
99}

126 // Loads the configuration file into the Parameter Manager.
127 if (!ParameterManager::getInstance().loadParameterFile(simulator.getConfigFileName())) {

129 "ERROR: failed to load config file: " << simulator.getConfigFileName());

133 // Read in simulator specific parameters from configuration file.
