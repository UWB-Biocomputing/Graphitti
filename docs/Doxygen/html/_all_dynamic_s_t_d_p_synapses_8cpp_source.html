174void AllDynamicSTDPSynapses::changePSR(const BGSIZE iEdg, const BGFLOAT deltaT)
175{
176 BGFLOAT &psr = this->psr_[iEdg];
177 BGFLOAT &W = this->W_[iEdg];
178 BGFLOAT &decay = this->decay_[iEdg];
179 uint64_t &lastSpike = this->lastSpike_[iEdg];
180 BGFLOAT &r = this->r_[iEdg];
181 BGFLOAT &u = this->u_[iEdg];
182 BGFLOAT &D = this->D_[iEdg];
183 BGFLOAT &F = this->F_[iEdg];
184 BGFLOAT &U = this->U_[iEdg];
185
186 // adjust synapse parameters
187 if (lastSpike != ULONG_MAX) {
188 BGFLOAT isi = (g_simulationStep - lastSpike) * deltaT;
189 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
190 u = U + u * (1 - U) * exp(-isi / F);
191 }
192 psr += ((W / decay) * u * r); // calculate psr
193 lastSpike = g_simulationStep; // record the time of the spike
194}
