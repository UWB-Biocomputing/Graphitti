162void AllDynamicSTDPSynapses::changePSR(BGSIZE iEdg, BGFLOAT deltaT)
163{
164 BGFLOAT &psr = this->psr_[iEdg];
165 BGFLOAT &W = this->W_[iEdg];
166 BGFLOAT &decay = this->decay_[iEdg];
167 uint64_t &lastSpike = this->lastSpike_[iEdg];
168 BGFLOAT &r = this->r_[iEdg];
169 BGFLOAT &u = this->u_[iEdg];
170 BGFLOAT &D = this->D_[iEdg];
171 BGFLOAT &F = this->F_[iEdg];
172 BGFLOAT &U = this->U_[iEdg];
173
174 // adjust synapse parameters
175 if (lastSpike != ULONG_MAX) {
176 BGFLOAT isi = (g_simulationStep - lastSpike) * deltaT;
177 r = 1 + (r * (1 - u) - 1) * exp(-isi / D);
178 u = U + u * (1 - U) * exp(-isi / F);
179 }
180 psr += ((W / decay) * u * r); // calculate psr
181 lastSpike = g_simulationStep; // record the time of the spike
182}
