66void AllSpikingNeurons::advanceVertices(AllEdges &synapses, const EdgeIndexMap &edgeIndexMap)
67{
68 int maxSpikes = (int)((Simulator::getInstance().getEpochDuration()
69 * Simulator::getInstance().getMaxFiringRate()));
70
71 AllSpikingSynapses &spSynapses = dynamic_cast<AllSpikingSynapses &>(synapses);
72 // For each neuron in the network
73 for (int idx = Simulator::getInstance().getTotalVertices() - 1; idx >= 0; --idx) {
74 // advance neurons
75 advanceNeuron(idx);
76
77 // notify outgoing/incoming synapses if neuron has fired
78 if (hasFired_[idx]) {
79 LOG4CPLUS_DEBUG(vertexLogger_,
80 "Neuron: " << idx << " has fired at time: "
81 << g_simulationStep * Simulator::getInstance().getDeltaT());
82
83 assert(vertexEvents_[idx].getNumElementsInEpoch() < maxSpikes);
84
85 // notify outgoing synapses
86 BGSIZE synapseCounts;
87
88
89 synapseCounts = edgeIndexMap.outgoingEdgeCount_[idx];
90 if (synapseCounts != 0) {
91 int beginIndex = edgeIndexMap.outgoingEdgeBegin_[idx];
92 BGSIZE iEdg;
93 for (BGSIZE i = 0; i < synapseCounts; i++) {
94 iEdg = edgeIndexMap.outgoingEdgeIndexMap_[beginIndex + i];
95 spSynapses.preSpikeHit(iEdg);
96 }
97 }
98
99
100 // notify incoming synapses
101 synapseCounts = spSynapses.edgeCounts_[idx];
102 BGSIZE synapse_notified = 0;
103
104 if (spSynapses.allowBackPropagation()) {
105 for (int z = 0; synapse_notified < synapseCounts; z++) {
106 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * idx + z;
107 if (spSynapses.inUse_[iEdg]) {
108 spSynapses.postSpikeHit(iEdg);
109 synapse_notified++;
110 }
111 }
112 }
113
114 hasFired_[idx] = false;
115 }
116 }
117}

164uint64_t AllSpikingNeurons::getSpikeHistory(int index, int offIndex)
165{
166 // offIndex is a minus offset
167 // This computes the index of a spike in the past (i.e., the most recent spike,
168 // two spikes ago, etc). It starts with `spikeCountOffset_ + spikeCount_`,
169 // which I believe at the end of an epoch should be one past the end of that
170 // neuron's spikes in `spikeHistory_`. Then, the maximum number of spikes per
171 // epoch is added. Then, the `offIndex` parameter is added. The expectation
172 // is that `offIndex` will be a negative number (i.e., a spike in the past);
173 // the reason that the max spikes value is added is to prevent this from
174 // producing a negative total, so that finally taking mod max spikes will
175 // "wrap around backwards" if needed.
176
177 return vertexEvents_[index].getPastEvent(offIndex);
178}
