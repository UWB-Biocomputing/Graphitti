65void AllSpikingNeurons::advanceVertices(AllEdges &synapses, const EdgeIndexMap &edgeIndexMap)
66{
67 int maxSpikes = (int)((Simulator::getInstance().getEpochDuration()
68 * Simulator::getInstance().getMaxFiringRate()));
69
70 AllSpikingSynapses &spSynapses = dynamic_cast<AllSpikingSynapses &>(synapses);
71 // For each neuron in the network
72 for (int idx = Simulator::getInstance().getTotalVertices() - 1; idx >= 0; --idx) {
73 // advance neurons
74 advanceNeuron(idx);
75
76 // notify outgoing/incoming synapses if neuron has fired
77 if (hasFired_[idx]) {
78 LOG4CPLUS_DEBUG(vertexLogger_,
79 "Neuron: " << idx << " has fired at time: "
80 << g_simulationStep * Simulator::getInstance().getDeltaT());
81
82 assert(vertexEvents_[idx].getNumElementsInEpoch() < maxSpikes);
83
84 // notify outgoing synapses
85 BGSIZE synapseCounts;
86
87
88 synapseCounts = edgeIndexMap.outgoingEdgeCount_[idx];
89 if (synapseCounts != 0) {
90 int beginIndex = edgeIndexMap.outgoingEdgeBegin_[idx];
91 BGSIZE iEdg;
92 for (BGSIZE i = 0; i < synapseCounts; i++) {
93 iEdg = edgeIndexMap.outgoingEdgeIndexMap_[beginIndex + i];
94 spSynapses.preSpikeHit(iEdg);
95 }
96 }
97
98
99 // notify incoming synapses
100 synapseCounts = spSynapses.edgeCounts_[idx];
101 BGSIZE synapse_notified = 0;
102
103 if (spSynapses.allowBackPropagation()) {
104 for (int z = 0; synapse_notified < synapseCounts; z++) {
105 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * idx + z;
106 if (spSynapses.inUse_[iEdg]) {
107 spSynapses.postSpikeHit(iEdg);
108 synapse_notified++;
109 }
110 }
111 }
112
113 hasFired_[idx] = false;
114 }
115 }
116}

137uint64_t AllSpikingNeurons::getSpikeHistory(int index, int offIndex)
138{
139 // offIndex is a minus offset
140 // This computes the index of a spike in the past (i.e., the most recent spike,
141 // two spikes ago, etc). It starts with `spikeCountOffset_ + spikeCount_`,
142 // which I believe at the end of an epoch should be one past the end of that
143 // neuron's spikes in `spikeHistory_`. Then, the maximum number of spikes per
144 // epoch is added. Then, the `offIndex` parameter is added. The expectation
145 // is that `offIndex` will be a negative number (i.e., a spike in the past);
146 // the reason that the max spikes value is added is to prevent this from
147 // producing a negative total, so that finally taking mod max spikes will
148 // "wrap around backwards" if needed.
149
150 return vertexEvents_[index].getPastEvent(offIndex);
151}
