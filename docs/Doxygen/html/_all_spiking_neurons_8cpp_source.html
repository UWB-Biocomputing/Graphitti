42void AllSpikingNeurons::advanceVertices(AllEdges &synapses, const EdgeIndexMap *edgeIndexMap)
43{
44 int maxSpikes = (int)((Simulator::getInstance().getEpochDuration()
45 * Simulator::getInstance().getMaxFiringRate()));
46
47 AllSpikingSynapses &spSynapses = dynamic_cast<AllSpikingSynapses &>(synapses);
48 // For each neuron in the network
49 for (int idx = Simulator::getInstance().getTotalVertices() - 1; idx >= 0; --idx) {
50 // advance neurons
51 advanceNeuron(idx);
52
53 // notify outgoing/incoming synapses if neuron has fired
54 if (hasFired_[idx]) {
55 LOG4CPLUS_DEBUG(vertexLogger_,
56 "Neuron: " << idx << " has fired at time: "
57 << g_simulationStep * Simulator::getInstance().getDeltaT());
58
59 assert(vertexEvents_[idx].getNumEventsInEpoch() < maxSpikes);
60
61 // notify outgoing synapses
62 BGSIZE synapseCounts;
63
64 if (edgeIndexMap != nullptr) {
65 synapseCounts = edgeIndexMap->outgoingEdgeCount_[idx];
66 if (synapseCounts != 0) {
67 int beginIndex = edgeIndexMap->outgoingEdgeBegin_[idx];
68 BGSIZE iEdg;
69 for (BGSIZE i = 0; i < synapseCounts; i++) {
70 iEdg = edgeIndexMap->outgoingEdgeIndexMap_[beginIndex + i];
71 spSynapses.preSpikeHit(iEdg);
72 }
73 }
74 }
75
76 // notify incoming synapses
77 synapseCounts = spSynapses.edgeCounts_[idx];
78 BGSIZE synapse_notified = 0;
79
80 if (spSynapses.allowBackPropagation()) {
81 for (int z = 0; synapse_notified < synapseCounts; z++) {
82 BGSIZE iEdg = Simulator::getInstance().getMaxEdgesPerVertex() * idx + z;
83 if (spSynapses.inUse_[iEdg] == true) {
84 spSynapses.postSpikeHit(iEdg);
85 synapse_notified++;
86 }
87 }
88 }
89
90 hasFired_[idx] = false;
91 }
92 }
93}

114uint64_t AllSpikingNeurons::getSpikeHistory(int index, int offIndex)
115{
116 // offIndex is a minus offset
117 // This computes the index of a spike in the past (i.e., the most recent spike,
118 // two spikes ago, etc). It starts with `spikeCountOffset_ + spikeCount_`,
119 // which I believe at the end of an epoch should be one past the end of that
120 // neuron's spikes in `spikeHistory_`. Then, the maximum number of spikes per
121 // epoch is added. Then, the `offIndex` parameter is added. The expectation
122 // is that `offIndex` will be a negative number (i.e., a spike in the past);
123 // the reason that the max spikes value is added is to prevent this from
124 // producing a negative total, so that finally taking mod max spikes will
125 // "wrap around backwards" if needed.
126
127 return vertexEvents_[index].getPastEvent(offIndex);
128}
