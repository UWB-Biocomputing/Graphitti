55void XmlSTDPRecorder::getValues()
56{
57 Simulator &simulator = Simulator::getInstance();
58 Connections &connections = simulator.getModel().getConnections();
59 AllEdges &synapses = connections.getEdges();
60 const int currentStep = simulator.getCurrentStep();
61
62 for (int i = 0; i < simulator.getTotalVertices(); i++) {
63 synapses.W_[i] = weightsHistory_[currentStep][i];
64 synapses.sourceVertexIndex_[i] = sourceNeuronIndexHistory_[currentStep][i];
65 synapses.destVertexIndex_[i] = destNeuronIndexHistory_[currentStep][i];

74 LOG4CPLUS_INFO(fileLogger_, "Compiling STDP HISTORY");

77 Connections &connections = simulator.getModel().getConnections();

79 const int currentStep = simulator.getCurrentStep();
80
81 for (int iNeuron = 0; iNeuron < simulator.getTotalVertices() * simulator.getMaxEdgesPerVertex();
82 iNeuron++) {
83 // record firing rate to history matrix

85 sourceNeuronIndexHistory_[currentStep][iNeuron] = synapses.sourceVertexIndex_[iNeuron];

138void XmlSTDPRecorder::saveSimData(const AllVertices &neurons)
139{
140 Simulator &simulator = Simulator::getInstance();
141
142 // create Neuron Types matrix
143 VectorMatrix neuronTypes(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), EXC);
144 for (int i = 0; i < simulator.getTotalVertices(); i++) {
145 neuronTypes[i] = simulator.getModel().getLayout().vertexTypeMap_[i];
146 }
147
148 // create neuron threshold matrix
149 VectorMatrix neuronThresh(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), 0);
150 for (int i = 0; i < simulator.getTotalVertices(); i++) {
151 neuronThresh[i] = dynamic_cast<const AllIFNeurons &>(neurons).Vthresh_[i];
152 }
153
154 // Write XML header information:
155 resultOut_ << "<?xml version=\"1.0\" standalone=\"no\"?>\n"
156 << "<!-- State output file for the DCT growth modeling-->\n";
157 //stateOut << version; TODO: version
158
159 // Write the core state information:
160 resultOut_ << "<SimState>\n";
161 resultOut_ << " " << toXML("sourceNeuronIndexHistory", sourceNeuronIndexHistory_) << endl;
162 resultOut_ << " " << toXML("destNeuronIndexHistory", destNeuronIndexHistory_) << endl;
163 resultOut_ << " " << toXML("weightsHistory", weightsHistory_) << endl;
164 resultOut_ << " " << spikesHistory_.toXML("spikesHistory") << endl;
165 resultOut_ << " " << simulator.getModel().getLayout().xloc_.toXML("xloc") << endl;
166 resultOut_ << " " << simulator.getModel().getLayout().yloc_.toXML("yloc") << endl;
167 resultOut_ << " " << neuronTypes.toXML("neuronTypes") << endl;
168
169 // create starter neuron matrix
170 int num_starter_neurons
171 = static_cast<int>(simulator.getModel().getLayout().numEndogenouslyActiveNeurons_);
172 if (num_starter_neurons > 0) {
173 VectorMatrix starterNeurons(MATRIX_TYPE, MATRIX_INIT, 1, num_starter_neurons);
174 getStarterNeuronMatrix(starterNeurons, simulator.getModel().getLayout().starterMap_);
175 resultOut_ << " " << starterNeurons.toXML("starterNeurons") << endl;
176 }
177
178 // Write neuron thresold
179 resultOut_ << " " << neuronThresh.toXML("neuronThresh") << endl;
180
181 // write time between growth cycles
182 resultOut_
183 << " <Matrix name=\"Tsim\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
184 << endl;
185 resultOut_ << " " << simulator.getEpochDuration() << endl;
186 resultOut_ << "</Matrix>" << endl;
187
188 // write simulation end time
189 resultOut_
190 << " <Matrix name=\"simulationEndTime\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
191 << endl;
192 resultOut_ << " " << g_simulationStep * simulator.getDeltaT() << endl;
193 resultOut_ << "</Matrix>" << endl;
194 resultOut_ << "</SimState>" << endl;
195}
