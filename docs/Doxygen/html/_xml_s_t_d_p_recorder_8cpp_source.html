147 void XmlSTDPRecorder::saveSimData(const IAllVertices &neurons)
148 {
149  Simulator &simulator = Simulator::getInstance();
150 
151  // create Neuron Types matrix
152  VectorMatrix neuronTypes(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), EXC);
153  for (int i = 0; i < simulator.getTotalVertices(); i++) {
154  neuronTypes[i] = simulator.getModel()->getLayout()->vertexTypeMap_[i];
155  }
156 
157  // create neuron threshold matrix
158  VectorMatrix neuronThresh(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), 0);
159  for (int i = 0; i < simulator.getTotalVertices(); i++) {
160  neuronThresh[i] = dynamic_cast<const AllIFNeurons &>(neurons).Vthresh_[i];
161  }
162 
163  // Write XML header information:
164  stateOut_ << "<?xml version=\"1.0\" standalone=\"no\"?>\n"
165  << "<!-- State output file for the DCT growth modeling-->\n";
166  //stateOut << version; TODO: version
167 
168  // Write the core state information:
169  stateOut_ << "<SimState>\n";
170  stateOut_ << " " <<toXML("sourceNeuronIndexHistory",sourceNeuronIndexHistory_) << endl;
171  stateOut_ << " " << toXML("destNeuronIndexHistory",destNeuronIndexHistory_) << endl;
172  stateOut_ << " " <<toXML("weightsHistory",weightsHistory_) << endl;
173  stateOut_ << " " << burstinessHist_.toXML("burstinessHist") << endl;
174  stateOut_ << " " << spikesHistory_.toXML("spikesHistory") << endl;
175  stateOut_ << " " << simulator.getModel()->getLayout()->xloc_->toXML("xloc") << endl;
176  stateOut_ << " " << simulator.getModel()->getLayout()->yloc_->toXML("yloc") << endl;
177  stateOut_ << " " << neuronTypes.toXML("neuronTypes") << endl;
178 
179  // create starter neuron matrix
180  int num_starter_neurons = static_cast<int>(simulator.getModel()->getLayout()->numEndogenouslyActiveNeurons_);
181  if (num_starter_neurons > 0) {
182  VectorMatrix starterNeurons(MATRIX_TYPE, MATRIX_INIT, 1, num_starter_neurons);
183  getStarterNeuronMatrix(starterNeurons, simulator.getModel()->getLayout()->starterMap_);
184  stateOut_ << " " << starterNeurons.toXML("starterNeurons") << endl;
185  }
186 
187  // Write neuron thresold
188  stateOut_ << " " << neuronThresh.toXML("neuronThresh") << endl;
189 
190  // write time between growth cycles
191  stateOut_ << " <Matrix name=\"Tsim\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">" << endl;
192  stateOut_ << " " << simulator.getEpochDuration() << endl;
193  stateOut_ << "</Matrix>" << endl;
194 
195  // write simulation end time
196  stateOut_ << " <Matrix name=\"simulationEndTime\" type=\"complete\" rows=\"1\" columns=\"1\" multiplier=\"1.0\">"
197  << endl;
198  stateOut_ << " " << g_simulationStep * simulator.getDeltaT() << endl;
199  stateOut_ << "</Matrix>" << endl;
200  stateOut_ << "</SimState>" << endl;
201 }
