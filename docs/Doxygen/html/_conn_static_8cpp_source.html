37void ConnStatic::setup()
38{
39 // we can obtain the Layout, which holds the vertices, from the Model
40 Layout &layout = *Simulator::getInstance().getModel()->getLayout();
41 AllVertices &vertices = *layout.getVertices();
42
43 Simulator &simulator = Simulator::getInstance();
44 int numVertices = simulator.getTotalVertices();
45 vector<DistDestVertex> distDestVertices[numVertices];
46 BGSIZE maxTotalEdges = simulator.getMaxEdgesPerVertex() * simulator.getTotalVertices();
47 WCurrentEpoch_.resize(maxTotalEdges);
48 sourceVertexIndexCurrentEpoch_.resize(maxTotalEdges);
49 destVertexIndexCurrentEpoch_.resize(maxTotalEdges);
50 AllNeuroEdges &neuroEdges = dynamic_cast<AllNeuroEdges &>(*edges_);
51
52 radiiSize_ = numVertices;
53 int added = 0;
54
55 LOG4CPLUS_INFO(fileLogger_, "Initializing connections");
56
57 for (int srcVertex = 0; srcVertex < numVertices; srcVertex++) {
58 distDestVertices[srcVertex].clear();
59
60 // pick the connections shorter than threshConnsRadius
61 for (int destVertex = 0; destVertex < numVertices; destVertex++) {
62 if (srcVertex != destVertex) {
63 BGFLOAT dist = layout.dist_(srcVertex, destVertex);
64 if (dist <= threshConnsRadius_) {
65 DistDestVertex distDestVertex {dist, destVertex};
66 distDestVertices[srcVertex].push_back(distDestVertex);
67 }
68 }
69 }
70
71 // sort ascendant
72 sort(distDestVertices[srcVertex].begin(), distDestVertices[srcVertex].end());
73 // pick the shortest connsPerVertex_ connections
74 for (BGSIZE i = 0; i < distDestVertices[srcVertex].size() && (int)i < connsPerVertex_; i++) {
75 int destVertex = distDestVertices[srcVertex][i].destVertex;
76 edgeType type = layout.edgType(srcVertex, destVertex);
77 BGFLOAT *sumPoint = &vertices.summationMap_[destVertex];
78
79 LOG4CPLUS_DEBUG(fileLogger_,
80 "Source: " << srcVertex << " Dest: " << destVertex
81 << " Dist: " << distDestVertices[srcVertex][i].dist);
82
83 BGSIZE iEdg;
84 edges_->addEdge(iEdg, type, srcVertex, destVertex, sumPoint, simulator.getDeltaT());
85 added++;
86
87
88 // set edge weight
89 // TODO: we need another synaptic weight distibution mode (normal distribution)
90 if (neuroEdges.edgSign(type) > 0) {
91 neuroEdges.W_[iEdg] = initRNG.inRange(excWeight_[0], excWeight_[1]);
92 } else {
93 neuroEdges.W_[iEdg] = initRNG.inRange(inhWeight_[0], inhWeight_[1]);
94 }
95 }
96 }
97
98 string weight_str = "";
99 for (int i = 0; i < maxTotalEdges; i++) {
100 WCurrentEpoch_[i] = neuroEdges.W_[i];
101 sourceVertexIndexCurrentEpoch_[i] = neuroEdges.sourceVertexIndex_[i];
102 destVertexIndexCurrentEpoch_[i] = neuroEdges.destVertexIndex_[i];
103
104 if (WCurrentEpoch_[i] != 0) {
105 // LOG4CPLUS_DEBUG(edgeLogger_,i << WCurrentEpoch_[i]);
106 weight_str += to_string(WCurrentEpoch_[i]) + " ";
107 }
108 }
109 // LOG4CPLUS_DEBUG(edgeLogger_, "Weights are " << weight_str);
110
111 int nRewiring = added * rewiringProbability_;
112
113 LOG4CPLUS_DEBUG(fileLogger_, "Rewiring connections: " << nRewiring);
114
115 LOG4CPLUS_DEBUG(fileLogger_, "Added connections: " << added);
116}
