9/************************************************************
10 norm.cpp -- normally distibuted random numbers
11
12 The following class allows you to create MTRand objects that are
13 independently seeded upon construction and return normally
14 distributed random numbers.
15
16 This function generates normally distributed random numbers
17 with mean of mu and standard deviation of sigma, using the
18 polar method of Marsaglia and Bray, "A Convenient Method for
19 Generating Normal Variables", _SIAM Rev._, 6: 260-264 (1964).
20
21 The algorithm is as follows:
22
23 1. Generate two uniformly distributed numbers, U1 and U2.
24 Let Vi = 2*Ui-1 for i=1,2, and let W = V1*V1 + V2*V2.
25 2. If W > 1, go back to step 1. Otherwise, let
26 Y = sqrt(-2*ln(W)/W), X1 = V1*Y, and X2 = V2*Y. Then
27 X1 and X2 are normally distributed with mean of 0 and
28 variance of 1.
29 3. Random numbers with mean of mu and standard deviation of sigma
30 are calculated by: X_prime = mu + sigma X.
31
32 Note that numbers are generated in pairs. On odd-numbered calls
33 to operator(), pairs are calculated. On even-numbered calls,
34 the second value is returned.
35
36 Modified from norm.c, from xneuron3
37
38************************************************************/

64BGFLOAT Norm::operator()()
65{
66 BGFLOAT U1, U2, // Uniformly distributed.
67 V1, V2, W, Y, //Work variables (see above).
68 X1; // First value computed (returned immediately)
69
70 // Check to see if we need to compute anything, complement indicator.
71 if ((odd_ = !odd_))
72 return (mu_ + sigma_ * X2_);
73
74 // Do the computation step 1 (until W <= 1)
75 do {
76 U1 = MTRand::operator()(); /* Generate U(0,1) */
77 U2 = MTRand::operator()();
78 V1 = 2 * U1 - 1;
79 V2 = 2 * U2 - 1;
80 W = V1 * V1 + V2 * V2;
81 } while (W > 1);
82
83 // Do the computation step 2
84 Y = sqrt(-2 * log(W) / W);
85 X1 = V1 * Y;
86 X2_ = V2 * Y;
87
88 // Return X1 this time, X2 next time
89 return (mu_ + sigma_ * X1);
90}
