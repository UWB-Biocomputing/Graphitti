169void AllSpikingSynapses::createEdge(BGSIZE iEdg, int srcVertex, int destVertex, BGFLOAT deltaT,
170 edgeType type)
171{
172 BGFLOAT delay;
173
174 inUse_[iEdg] = true; // True: 1, False: 0
175 destVertexIndex_[iEdg] = destVertex;
176 sourceVertexIndex_[iEdg] = srcVertex;
177 W_[iEdg] = edgSign(type) * 10.0e-9;
178 type_[iEdg] = type;
179 tau_[iEdg] = DEFAULT_tau;
180
181 BGFLOAT tau;
182 switch (type) {
183 case II:
184 tau = 6e-3;
185 delay = 0.8e-3;
186 break;
187 case IE:
188 tau = 6e-3;
189 delay = 0.8e-3;
190 break;
191 case EI:
192 tau = 3e-3;
193 delay = 0.8e-3;
194 break;
195 case EE:
196 tau = 3e-3;
197 delay = 1.5e-3;
198 break;
199 default:
200 assert(false);
201 break;
202 }
203
204 tau_[iEdg] = tau;
205 totalDelay_[iEdg] = static_cast<int>(delay / deltaT) + 1;
206
207 // initializes the queues for the Synapses
208 initSpikeQueue(iEdg);
209 // reset time varying state vars and recompute decay
210 resetEdge(iEdg, deltaT);
211}

219bool AllSpikingSynapses::isSpikeQueue(BGSIZE iEdg)
220{
221 uint32_t &delayQueue = delayQueue_[iEdg];
222 int &delayIdx = delayIndex_[iEdg];
223 int &ldelayQueue = delayQueueLength_[iEdg];
224
225 bool r = delayQueue & (0x1 << delayIdx);
226 delayQueue &= ~(0x1 << delayIdx);
227 if (++delayIdx >= ldelayQueue) {
228 delayIdx = 0;
229 }
230 return r;
231}

236void AllSpikingSynapses::preSpikeHit(BGSIZE iEdg)
237{
238 uint32_t &delayQueue = delayQueue_[iEdg];
239 int &delayIdx = delayIndex_[iEdg];
240 int &ldelayQueue = delayQueueLength_[iEdg];
241 int &totalDelay = totalDelay_[iEdg];
242
243 // Add to spike queue
244
245 // calculate index where to insert the spike into delayQueue
246 //LOG4CPLUS_TRACE(edgeLogger_,"delayidx "<<delayIdx<<" totalDelay "<<totalDelay<<" ldelayQueue "<<ldelayQueue);
247 int idx = delayIdx + totalDelay;
248 if (idx >= ldelayQueue) {
249 idx -= ldelayQueue;
250 }
251 if ((delayQueue & (0x1 << idx)) != 0) {
252 LOG4CPLUS_ERROR(edgeLogger_, "Delay Queue Error " << setbase(2) << delayQueue << setbase(10)
253 << " index " << idx << " edge ID " << iEdg);
254 assert(false);
255 }
256
257 // set a spike
258 delayQueue |= (0x1 << idx);
259}
