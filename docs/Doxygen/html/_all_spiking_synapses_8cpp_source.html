166void AllSpikingSynapses::createEdge(BGSIZE iEdg, int srcVertex, int destVertex, BGFLOAT deltaT,
167 edgeType type)
168{
169 BGFLOAT delay;
170
171 inUse_[iEdg] = true; // True: 1, False: 0
172 destVertexIndex_[iEdg] = destVertex;
173 sourceVertexIndex_[iEdg] = srcVertex;
174 W_[iEdg] = edgSign(type) * 10.0e-9;
175 type_[iEdg] = type;
176 tau_[iEdg] = DEFAULT_tau;
177
178 BGFLOAT tau;
179 switch (type) {
180 case edgeType::II:
181 tau = 6e-3;
182 delay = 0.8e-3;
183 break;
184 case edgeType::IE:
185 tau = 6e-3;
186 delay = 0.8e-3;
187 break;
188 case edgeType::EI:
189 tau = 3e-3;
190 delay = 0.8e-3;
191 break;
192 case edgeType::EE:
193 tau = 3e-3;
194 delay = 1.5e-3;
195 break;
196 default:
197 assert(false);
198 break;
199 }
200
201 tau_[iEdg] = tau;
202 totalDelay_[iEdg] = static_cast<int>(delay / deltaT) + 1;
203
204 // initializes the queues for the Synapses
205 initSpikeQueue(iEdg);
206 // reset time varying state vars and recompute decay
207 resetEdge(iEdg, deltaT);
208}

216bool AllSpikingSynapses::isSpikeQueue(BGSIZE iEdg)
217{
218 uint32_t &delayQueue = delayQueue_[iEdg];
219 int &delayIdx = delayIndex_[iEdg];
220 int &ldelayQueue = delayQueueLength_[iEdg];
221
222 bool r = delayQueue & (0x1 << delayIdx);
223 delayQueue &= ~(0x1 << delayIdx);
224 if (++delayIdx >= ldelayQueue) {
225 delayIdx = 0;
226 }
227 return r;
228}

233void AllSpikingSynapses::preSpikeHit(BGSIZE iEdg)
234{
235 uint32_t &delayQueue = delayQueue_[iEdg];
236 int &delayIdx = delayIndex_[iEdg];
237 int &ldelayQueue = delayQueueLength_[iEdg];
238 int &totalDelay = totalDelay_[iEdg];
239
240 // Add to spike queue
241
242 // calculate index where to insert the spike into delayQueue
243 //LOG4CPLUS_TRACE(edgeLogger_,"delayidx "<<delayIdx<<" totalDelay "<<totalDelay<<" ldelayQueue "<<ldelayQueue);
244 int idx = delayIdx + totalDelay;
245 if (idx >= ldelayQueue) {
246 idx -= ldelayQueue;
247 }
248 if ((delayQueue & (0x1 << idx)) != 0) {
249 LOG4CPLUS_ERROR(edgeLogger_, "Delay Queue Error " << setbase(2) << delayQueue << setbase(10)
250 << " index " << idx << " edge ID " << iEdg);
251 assert(false);
252 }
253
254 // set a spike
255 delayQueue |= (0x1 << idx);
256}
