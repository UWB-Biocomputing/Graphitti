181void AllSpikingSynapses::createEdge(const BGSIZE iEdg, int srcVertex, int destVertex,
182 BGFLOAT *sumPoint, const BGFLOAT deltaT, edgeType type)
183{
184 BGFLOAT delay;
185
186 inUse_[iEdg] = true;
187 summationPoint_[iEdg] = sumPoint;
188 destVertexIndex_[iEdg] = destVertex;
189 sourceVertexIndex_[iEdg] = srcVertex;
190 W_[iEdg] = edgSign(type) * 10.0e-9;
191 type_[iEdg] = type;
192 tau_[iEdg] = DEFAULT_tau;
193
194 BGFLOAT tau;
195 switch (type) {
196 case II:
197 tau = 6e-3;
198 delay = 0.8e-3;
199 break;
200 case IE:
201 tau = 6e-3;
202 delay = 0.8e-3;
203 break;
204 case EI:
205 tau = 3e-3;
206 delay = 0.8e-3;
207 break;
208 case EE:
209 tau = 3e-3;
210 delay = 1.5e-3;
211 break;
212 default:
213 assert(false);
214 break;
215 }
216
217 tau_[iEdg] = tau;
218 totalDelay_[iEdg] = static_cast<int>(delay / deltaT) + 1;
219
220 // initializes the queues for the Synapses
221 initSpikeQueue(iEdg);
222 // reset time varying state vars and recompute decay
223 resetEdge(iEdg, deltaT);
224}

232bool AllSpikingSynapses::isSpikeQueue(const BGSIZE iEdg)
233{
234 uint32_t &delayQueue = delayQueue_[iEdg];
235 int &delayIdx = delayIndex_[iEdg];
236 int &ldelayQueue = delayQueueLength_[iEdg];
237
238 bool r = delayQueue & (0x1 << delayIdx);
239 delayQueue &= ~(0x1 << delayIdx);
240 if (++delayIdx >= ldelayQueue) {
241 delayIdx = 0;
242 }
243 return r;
244}

249void AllSpikingSynapses::preSpikeHit(const BGSIZE iEdg)
250{
251 uint32_t &delayQueue = delayQueue_[iEdg];
252 int &delayIdx = delayIndex_[iEdg];
253 int &ldelayQueue = delayQueueLength_[iEdg];
254 int &totalDelay = totalDelay_[iEdg];
255
256 // Add to spike queue
257
258 // calculate index where to insert the spike into delayQueue
259 //LOG4CPLUS_TRACE(edgeLogger_,"delayidx "<<delayIdx<<" totalDelay "<<totalDelay<<" ldelayQueue "<<ldelayQueue);
260 int idx = delayIdx + totalDelay;
261 if (idx >= ldelayQueue) {
262 idx -= ldelayQueue;
263 }
264 if ((delayQueue & (0x1 << idx)) != 0) {
265 LOG4CPLUS_ERROR(edgeLogger_, "Delay Queue Error " << setbase(2) << delayQueue << setbase(10)
266 << " index " << idx << " edge ID " << iEdg);
267 assert(false);
268 }
269
270 // set a spike
271 delayQueue |= (0x1 << idx);
272}

328bool AllSpikingSynapses::updateDecay(const BGSIZE iEdg, const BGFLOAT deltaT)
