99 {
100  // Pull a 32-bit integer from the generator state

102

165 void MTRand::seed( uint32_t *const bigSeed, uint32_t seedLength )
166 {
167  // Seed the generator with an array of uint32_t's
168  // There are 2^19937-1 possible initial states. This function allows
169  // all of those to be accessed by providing at least 19937 bits (with a
170  // default seed length of N = 624 uint32_t's). Any bits above the lower 32
171  // in each element are discarded.
172  // Just call seed() if you want to get array from /dev/urandom
173  initialize(19650218UL);
174  register int i = 1;
175  register uint32_t j = 0;
176  register int k = ( N > seedLength ? N : seedLength );
177  for( ; k; --k )
178  {
179  state[i] =
180  state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1664525UL );
181  state[i] += ( bigSeed[j] & 0xffffffffUL ) + j;
182  state[i] &= 0xffffffffUL;
183  ++i; ++j;
184  if( i >= N ) { state[0] = state[N-1]; i = 1; }
185  if( j >= seedLength ) j = 0;
186  }
187  for( k = N - 1; k; --k )
188  {
189  state[i] =
190  state[i] ^ ( (state[i-1] ^ (state[i-1] >> 30)) * 1566083941UL );
191  state[i] -= i;
192  state[i] &= 0xffffffffUL;
193  ++i;
194  if( i >= N ) { state[0] = state[N-1]; i = 1; }
195  }
196  state[0] = 0x80000000UL; // MSB is 1, assuring non-zero initial array
197  reload();
198 }

201 void MTRand::seed()
202 {
203  // Seed the generator with an array from /dev/urandom if available
204  // Otherwise use a hash of time() and clock() values
205 
206  // First try getting an array from /dev/urandom
207  FILE* urandom = fopen( "/dev/urandom", "rb" );
208  if( urandom )
209  {
210  uint32_t bigSeed[N];
211  register uint32_t *s = bigSeed;
212  register int i = N;
213  register bool success = true;
214  while( success && i-- )
215  success = fread( s++, sizeof(uint32_t), 1, urandom );
216  fclose(urandom);
217  if( success ) { seed( bigSeed, N ); return; }
218  }
219

221  seed( hash( time(nullptr), clock() ) );

266 {
267  // Initialize generator state with seed
268  // See Knuth TAOCP Vol 2, 3rd Ed, p.106 for multiplier.
269  // In previous versions, most significant bits (MSBs) of the seed affect
270  // only MSBs of the state array. Modified 9 Jan 2002 by Makoto Matsumoto.
271  register uint32_t *s = state;
272  register uint32_t *r = state;
273  register int i = 1;
274  *s++ = seed & 0xffffffffUL;
275  for( ; i < N; ++i )
276  {

283 void MTRand::reload()
284 {
285  // Generate N new values in state
286  // Made clearer and faster by Matthew Bellew (matthew.bellew@home.com)
287  register uint32_t *p = state;
288  register int i;
289  for( i = N - M; i--; ++p )
290  *p = twist( p[M], p[0], p[1] );
291  for( i = M; --i; ++p )
292  *p = twist( p[M-N], p[0], p[1] );
293  *p = twist( p[M-N], p[0], state[0] );

299 uint32_t MTRand::hash( time_t t, clock_t c )
300 {
301  // Get a uint32_t from t and c
302  // Better than uint32_t(x) in case x is floating point in [0,1]
303  // Based on code by Lawrence Kirby (fred@genesis.demon.co.uk)
304 
305  static uint32_t differ = 0; // guarantee time-based seeds will change
306 
307  uint32_t h1 = 0;
308  unsigned char *p = (unsigned char *) &t;
309  for( size_t i = 0; i < sizeof(t); ++i )
310  {
311  h1 *= UCHAR_MAX + 2U;
312  h1 += p[i];
313  }
314  uint32_t h2 = 0;
315  p = (unsigned char *) &c;
316  for( size_t j = 0; j < sizeof(c); ++j )
317  {
318  h2 *= UCHAR_MAX + 2U;
319  h2 += p[j];
320  }
321  return ( h1 + differ++ ) ^ h2;
322 }
