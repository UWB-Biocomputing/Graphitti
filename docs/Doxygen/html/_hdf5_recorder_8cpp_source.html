217void Hdf5Recorder::compileHistories(AllVertices &vertices)
218{
219 AllSpikingNeurons &spNeurons = dynamic_cast<AllSpikingNeurons &>(vertices);
220 Simulator &simulator = Simulator::getInstance();
221
222 int maxSpikes = (int)((simulator.getEpochDuration() * simulator.getMaxFiringRate()));
223
224 unsigned int iProbe = 0; // index into the probedNeuronsLayout vector
225 bool fProbe = false;
226
227 Model &model = simulator.getModel();
228
229 // output spikes: iterate over each neuron
230 for (int iVertex = 0; iVertex < spNeurons.vertexEvents_.size(); iVertex++) {
231 // true if this is a probed neuron
232 fProbe = ((iProbe < model.getLayout().probedNeuronList_.size())
233 && (iVertex == model.getLayout().probedNeuronList_[iProbe]));
234
235 // iterate over each spike that neuron produced
236 for (int eventIterator = 0;
237 eventIterator < spNeurons.vertexEvents_[iVertex].getNumEventsInEpoch();
238 eventIterator++) {
239 // Single precision (float) gives you 23 bits of significand, 8 bits of exponent,
240 // and 1 sign bit. Double precision (double) gives you 52 bits of significand,
241 // 11 bits of exponent, and 1 sign bit.
242 // Therefore, single precision can only handle 2^23 = 8,388,608 simulation steps
243 // or 8 epochs (1 epoch = 100s, 1 simulation step = 0.1ms).
244
245 // if (idxSp >= maxSpikes) idxSp = 0;
246
247 // compile network wide spike count in 10ms bins
248 int idx2 = static_cast<int>(
249 static_cast<double>(spNeurons.vertexEvents_[iVertex][eventIterator])
250 * simulator.getDeltaT() * 100
251 - ((simulator.getCurrentStep() - 1) * simulator.getEpochDuration() * 100));
252 // make sure idx2 is a valid index of spikesHistory_
253 assert(idx2 >= 0 && idx2 < (simulator.getEpochDuration() * 100));
254 spikesHistory_[idx2] = spikesHistory_[idx2] + 1.0;
255
256 // compile spikes time of the probed neuron (append spikes time)
257 if (fProbe) {
258 spikesProbedNeurons_[iProbe].insert(spikesProbedNeurons_[iProbe].end(),
259 spNeurons.vertexEvents_[iVertex][eventIterator]);
260 }
261 }
262
263 if (fProbe) {
264 iProbe++;
265 }
266 }
267
268 // clear spike count for all neurons (we've captured their spike information)
269 spNeurons.clearSpikeCounts();
270
271 try {
272 // write index
273 hsize_t offset[2], count[2];
274 hsize_t dimsm[2];
275
276 // write network wide spike count in 10ms bins
277 offset[0] = (simulator.getCurrentStep() - 1) * simulator.getEpochDuration() * 100;
278 count[0] = simulator.getEpochDuration() * 100;
279 dimsm[0] = simulator.getEpochDuration() * 100;
280 DataSpace memspace(1, dimsm, nullptr);
281 DataSpace dataspace = dataSetSpikesHist_.getSpace();
282 dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
283 dataSetSpikesHist_.write(spikesHistory_.data(), PredType::NATIVE_INT, memspace, dataspace);
284 spikesHistory_.assign(static_cast<int>(simulator.getEpochDuration() * 100), 0);
285
286 // write spikes data of probed neurons
287 if (model.getLayout().probedNeuronList_.size() > 0) {
288 unsigned int max_size = 0;
289 // iterate over each neuron to find the maximum number of spikes for
290 // this epoch
291 for (unsigned int i = 0; i < model.getLayout().probedNeuronList_.size(); i++) {
292 unsigned int size = spikesProbedNeurons_[i].size() + offsetSpikesProbedNeurons_[i];
293 max_size = (max_size > size) ? max_size : size;
294 }
295 // dataset dimensions
296 dimsm[0] = static_cast<hsize_t>(max_size);
297 dimsm[1] = static_cast<hsize_t>(model.getLayout().probedNeuronList_.size());
298
299 // extend the dataset
300 dataSetSpikesProbedNeurons_.extend(dimsm);
301 dataspace = dataSetSpikesProbedNeurons_.getSpace();
302
303 // write it! Iterate over each neuron's spike data.
304 for (unsigned int i = 0; i < model.getLayout().probedNeuronList_.size(); i++) {
305 dimsm[0] = spikesProbedNeurons_[i].size();
306 dimsm[1] = 1;
307 DataSpace memspace_spike(2, dimsm, nullptr);
308
309 offset[0] = offsetSpikesProbedNeurons_[i];
310 offset[1] = i;
311 count[0] = spikesProbedNeurons_[i].size();
312 count[1] = 1;
313 dataspace.selectHyperslab(H5S_SELECT_SET, count, offset);
314 offsetSpikesProbedNeurons_[i] += spikesProbedNeurons_[i].size();
315
316 dataSetSpikesProbedNeurons_.write(
317 static_cast<uint64_t *>(&(spikesProbedNeurons_[i][0])), PredType::NATIVE_UINT64,
318 memspace_spike, dataspace);
319
320 // clear the probed spike data
321 spikesProbedNeurons_[i].clear();
322 }
323 }
324 }
325
326 // catch failure caused by the H5File operations
327 catch (FileIException error) {
328 error.printErrorStack();
329 return;
330 }
331
332 // catch failure caused by the DataSet operations
333 catch (DataSetIException error) {
334 error.printErrorStack();
335 return;
336 }
337
338 // catch failure caused by the DataSpace operations
339 catch (DataSpaceIException error) {
340 error.printErrorStack();
341 return;
342 }
343
344 // catch failure caused by the DataType operations
345 catch (DataTypeIException error) {
346 error.printErrorStack();
347 return;
348 }

354void Hdf5Recorder::saveSimData(const AllVertices &vertices)
355{
356 Simulator &simulator = Simulator::getInstance();
357 Model &model = simulator.getModel();
358
359 try {
360 // create Neuron Types matrix
361 VectorMatrix neuronTypes(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), EXC);
362 for (int i = 0; i < simulator.getTotalVertices(); i++) {
363 neuronTypes[i] = model.getLayout().vertexTypeMap_[i];
364 }
365
366 // create neuron threshold matrix
367 VectorMatrix neuronThresh(MATRIX_TYPE, MATRIX_INIT, 1, simulator.getTotalVertices(), 0);
368 for (int i = 0; i < simulator.getTotalVertices(); i++) {
369 neuronThresh[i] = dynamic_cast<const AllIFNeurons &>(vertices).Vthresh_[i];
370 }
371
372 // Write the neuron location matrices
373 vector<int> iXloc(simulator.getTotalVertices());
374 vector<int> iYloc(simulator.getTotalVertices());
375 for (int i = 0; i < simulator.getTotalVertices(); i++) {
376 // convert VectorMatrix to int array
377 iXloc[i] = (model.getLayout().xloc_)[i];
378 iYloc[i] = (model.getLayout().yloc_)[i];
379 }
380 dataSetXloc_.write(iXloc.data(), PredType::NATIVE_INT);
381 dataSetYloc_.write(iYloc.data(), PredType::NATIVE_INT);
382
383 int *iNeuronTypes = new int[simulator.getTotalVertices()];
384 for (int i = 0; i < simulator.getTotalVertices(); i++) {
385 iNeuronTypes[i] = neuronTypes[i];
386 }
387 dataSetNeuronTypes_.write(iNeuronTypes, PredType::NATIVE_INT);
388
389 int num_starter_neurons = static_cast<int>(model.getLayout().numEndogenouslyActiveNeurons_);
390 if (num_starter_neurons > 0) {
391 VectorMatrix starterNeurons(MATRIX_TYPE, MATRIX_INIT, 1, num_starter_neurons);
392 getStarterNeuronMatrix(starterNeurons, model.getLayout().starterMap_);
393
394 // create the data space & dataset for starter neurons
395 hsize_t dims[2];
396 dims[0] = static_cast<hsize_t>(starterNeurons.Size());
397 DataSpace dsStarterNeurons(1, dims);
398 dataSetStarterNeurons_
399 = resultOut_.createDataSet(nameStarterNeurons, PredType::NATIVE_INT, dsStarterNeurons);
400
401 vector<int> iStarterNeurons(starterNeurons.Size());
402 for (int i = 0; i < starterNeurons.Size(); i++) {
403 iStarterNeurons[i] = starterNeurons[i];
404 }
405 dataSetStarterNeurons_.write(iStarterNeurons.data(), PredType::NATIVE_INT);
406 }
407
408 // Finalize probed neurons' spikes dataset
409 if (model.getLayout().probedNeuronList_.size() > 0) {
410 // create the data space & dataset for probed neurons
411 hsize_t dims[2];
412
413 vector<int> iProbedNeurons(model.getLayout().probedNeuronList_.size());
414 for (unsigned int i = 0; i < model.getLayout().probedNeuronList_.size(); i++) {
415 iProbedNeurons[i] = model.getLayout().probedNeuronList_[i];
416 }
417 dataSetProbedNeurons_.write(iProbedNeurons.data(), PredType::NATIVE_INT);
418
419 // Create the data space for the attribute (unit of the spikes of probed neurons in second).
420 dims[0] = 1;
421 DataSpace dsAttrPNUnit(1, dims);
422
423 // Create a dataset attribute.
424 Attribute attribute = dataSetSpikesProbedNeurons_.createAttribute(
425 nameAttrPNUnit, H5_FLOAT, dsAttrPNUnit, PropList::DEFAULT);
426
427 // Write the attribute data.
428 float deltaT = Simulator::getInstance().getDeltaT();
429 attribute.write(H5_FLOAT, &deltaT);
430 }
431
432 // Write neuron thresold
433 vector<BGFLOAT> fNeuronThresh(simulator.getTotalVertices());
434 for (int i = 0; i < simulator.getTotalVertices(); i++) {
435 fNeuronThresh[i] = neuronThresh[i];
436 }
437 dataSetNeuronThresh_.write(fNeuronThresh.data(), H5_FLOAT);
438
439 // write time between growth cycles
440 BGFLOAT epochDuration = simulator.getEpochDuration();
441 dataSetTsim_.write(&epochDuration, H5_FLOAT);
442
443
444 // write simulation end time
445 BGFLOAT endTime = g_simulationStep * simulator.getDeltaT();
446 dataSetSimulationEndTime_.write(&endTime, H5_FLOAT);
447
448 }
449
450 // catch failure caused by the DataSet operations
451 catch (DataSetIException error) {
452 error.printErrorStack();
453 return;
454 }
455
456 // catch failure caused by the DataSpace operations
457 catch (DataSpaceIException error) {
458 error.printErrorStack();
459 return;
460 }
