120 bool ParameterManager::getIntByXpath(string xpath, int &referenceVar) {
121  if (!checkDocumentStatus()) return false;
122  string tmp;
123  if (!getStringByXpath(xpath, tmp)) {
124  cerr << "Failed loading simulation parameter for xpath "
125  << xpath << endl;
126  return false;
127  }
128  // Workaround for standard value conversion functions.
129  // stoi() will cast floats to ints.
130  if (regex_match(tmp, regex("\\d+[.]\\d+(e[+-]?\\d+)?f?|\\d+[.]?\\d+(e[+-]?\\d+)?f"))) {
131  cerr << "Parsed parameter is likely a float/double value. "
132  << "Terminating integer cast. Value: "
133  << tmp << endl;
134  return false;
135  } else if (regex_match(tmp, regex(".*[^\\def.]+.*"))) {
136  cerr << "Parsed parameter is likely a string. "
137  << "Terminating integer cast. Value: "
138  << tmp << endl;
139  return false;
140  }
141  try {
142  referenceVar = stoi(tmp);
143  } catch (invalid_argument &arg_exception) {
144  cerr << "Parsed parameter could not be parsed as an integer. Value: "
145  << tmp << endl;
146  return false;
147  } catch (out_of_range &range_exception) {
148  cerr << "Parsed string parameter could not be converted to an integer. Value: "
149  << tmp << endl;
150  return false;
151  }
152  return true;
153 }

308  cerr << "Could not find <" << elementName << "> in vertex list file " << path << endl;
309  return false;

312  // Get list of ints as a string stream

314 
315  // Parse integers out of the string and add them to a list
