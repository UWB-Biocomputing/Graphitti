88bool ParameterManager::getStringByXpath(string xpath, string &referenceVar)
89{
90 if (!checkDocumentStatus())
91 return false;
92
93 // temp string holds evaluation of value at xpath
94 string temp;
95 // raise error if tinyxml cannot compute the xpath's value or returns empty
96 if (!TinyXPath::o_xpath_string(root_, xpath.c_str(), temp) || temp == "") {
97 cerr << "Failed loading simulation parameter for xpath " << xpath << endl;
98 return false;
99 }
100 referenceVar = temp;
101 return true;

111bool ParameterManager::getIntByXpath(string xpath, int &referenceVar)
112{
113 if (!checkDocumentStatus())
114 return false;
115 string tmp;
116 if (!getStringByXpath(xpath, tmp)) {
117 return false;
118 }
119 // Workaround for standard value conversion functions.
120 // stoi() will cast floats to ints.
121 if (regex_match(tmp, regex("\\d+[.]\\d+(e[+-]?\\d+)?f?|\\d+[.]?\\d+(e[+-]?\\d+)?f"))) {
122 cerr << "Parsed parameter is likely a float/double value. "
123 << "Terminating integer cast. Value: " << tmp << endl;
124 return false;
125 } else if (regex_match(tmp, regex(".*[^\\def.]+.*"))) {
126 cerr << "Parsed parameter is likely a string. "
127 << "Terminating integer cast. Value: " << tmp << endl;
128 return false;
129 }
130 try {
131 referenceVar = stoi(tmp);
132 } catch (invalid_argument &arg_exception) {
133 cerr << "Parsed parameter could not be parsed as an integer. Value: " << tmp << endl;
134 return false;
135 } catch (out_of_range &range_exception) {
136 cerr << "Parsed string parameter could not be converted to an integer. Value: " << tmp
137 << endl;
138 return false;
139 }
140 return true;
141}

275bool ParameterManager::getIntVectorByXpath(const string &path, const string &elementName,
276 vector<int> &referenceVar)
277{
278 // Open file using a local XmlDocument object
279 TiXmlDocument xmlDocument(path.c_str());
280 if (!xmlDocument.LoadFile()) {
281 cerr << "Failed to load " << path.c_str() << ":"
282 << "\n\t" << xmlDocument.ErrorDesc() << endl;
283 return false;
284 }
285
286 // Check file for matching element
287 TiXmlNode *xmlNode = nullptr;
288 if ((xmlNode = xmlDocument.FirstChildElement(elementName)) == nullptr) {
289 cerr << "Could not find <" << elementName << "> in vertex list file " << path << endl;
290 return false;
291 }
292
293 // Get list of ints as a string stream
294 std::istringstream valueStream(xmlNode->ToElement()->GetText());
295
296 // Parse integers out of the string and add them to a list
297 int i;
298 while (valueStream.good()) {
299 valueStream >> i;
300 referenceVar.push_back(i);
301 }
302 return true;
303}
